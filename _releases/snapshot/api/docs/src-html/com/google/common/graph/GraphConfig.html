017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkArgument;
020import static com.google.common.base.Preconditions.checkNotNull;
021
022import com.google.common.annotations.Beta;
023import com.google.common.base.Joiner;
024import com.google.common.base.Objects;
025import com.google.common.base.Optional;
026import com.google.common.collect.Lists;
027
028import java.util.List;
029
030import javax.annotation.CheckReturnValue;
031import javax.annotation.Nullable;
032
033/**
034 * A class for configuring different types of graphs.
035 *
036 * <p>{@code Graphs.config()} should be used to get an instance of this class.
037 *
038 * <p>Currently, this class supports the following graph configurations (all combinations
039 * of these properties are valid unless stated otherwise):
040 * <ul>
041 * <li>Multigraphs.
042 * <li>Expected number of nodes/edges.
043 * <li>Self-loop edges.
044 * </ul>
045 *
046 * <p>Default graph configuration:
047 * <ul>
048 * <li>Self-loop edges are allowed.
049 * <li>It is not a multigraph: parallel edges (multiple edges directed from n1
050 *     to n2, or between them in case of undirected graphs) are not allowed.
051 * <li>In case of directed graphs, anti-parallel edges (same incident nodes but
052 *     in opposite direction, e.g. (n1, n2) and (n2, n1)) are allowed.
053 * <li>Nodes and edges are not sorted.
054 * </ul>
055 *
056 * <p>{@code GraphConfig} instances are thread-safe immutable, and are therefore safe to
057 * store as {@code static final} constants.
058 *
059 * @author Joshua O'Madadhain
060 * @see Graphs
061 * @since 20.0
062 */
063// TODO(user): Add support for sorted nodes/edges. Use Object as
064//     the node and edge types in this case: the same scheme used in CacheBuilder.
065// TODO(user): Add support for hypergraphs.
066// TODO(user): Handle sorted nodes/edges and expected number of nodes/edges together,
067//     in case sorted nodes/edges is supported.
068@Beta
069@CheckReturnValue
070public final class GraphConfig {
071  private final boolean multigraph;
072  private final boolean selfLoopsAllowed;
073  private final Optional<Integer> expectedNodeCount;
074  private final Optional<Integer> expectedEdgeCount;
075
076  // Creates an instance of this class with the default graph configuration.
077  GraphConfig() {
078    multigraph = false;
079    selfLoopsAllowed = true;
080    expectedNodeCount = Optional.absent();
081    expectedEdgeCount = Optional.absent();
082  }
083
084  private GraphConfig(
085      boolean multigraph,
086      boolean selfLoopsAllowed,
087      Optional<Integer> expectedNodeCount,
088      Optional<Integer> expectedEdgeCount) {
089    this.multigraph = multigraph;
090    this.selfLoopsAllowed = selfLoopsAllowed;
091    this.expectedNodeCount = expectedNodeCount;
092    this.expectedEdgeCount = expectedEdgeCount;
093  }
094
095  public boolean isMultigraph() {
096    return multigraph;
097  }
098
099  public boolean isSelfLoopsAllowed() {
100    return selfLoopsAllowed;
101  }
102
103  public Optional<Integer> getExpectedNodeCount() {
104    return expectedNodeCount;
105  }
106
107  public Optional<Integer> getExpectedEdgeCount() {
108    return expectedEdgeCount;
109  }
110
111  /**
112   * Specifies the expected number of nodes in the graph configuration.
113   *
114   * @return a new {@code GraphConfig} instance that augments the existing configuration
115   *         by specifying the expected number of nodes.
116   * @throws IllegalArgumentException if {@code expectedNodeCount} is negative
117   */
118  public GraphConfig expectedNodeCount(int expectedNodeCount) {
119    checkArgument(expectedNodeCount >= 0, "The expected number of nodes can't be negative");
120    return new GraphConfig(
121        multigraph, selfLoopsAllowed, Optional.of(expectedNodeCount), expectedEdgeCount);
122  }
123
124  /**
125   * Specifies the expected number of edges in the graph configuration.
126   *
127   * @return a new {@code GraphConfig} instance that augments the existing configuration
128   *         by specifying the expected number of edges.
129   * @throws IllegalArgumentException if {@code expectedEdgeCount} is negative
130   */
131  public GraphConfig expectedEdgeCount(int expectedEdgeCount) {
132    checkArgument(expectedEdgeCount >= 0, "The expected number of edges can't be negative");
133    return new GraphConfig(
134        multigraph, selfLoopsAllowed, expectedNodeCount, Optional.of(expectedEdgeCount));
135  }
136
137  /**
138   * Specifies that this graph is a multigraph (allows parallel edges).
139   *
140   * @return a new {@code GraphConfig} instance that augments the existing configuration
141   *         by allowing parallel edges (multigraph).
142   */
143  public GraphConfig multigraph() {
144    return new GraphConfig(true, selfLoopsAllowed, expectedNodeCount, expectedEdgeCount);
145  }
146
147  /**
148   * Specifies that this graph does not allow self-loop edges.
149   *
150   * @return a new {@code GraphConfig} instance that augments the existing configuration
151   *         by disallowing self-loop edges.
152   */
153  public GraphConfig noSelfLoops() {
154    return new GraphConfig(multigraph, false, expectedNodeCount, expectedEdgeCount);
155  }
156
157  /**
158   * Returns true iff the configuration defined by this object is <i>compatible with</i> the
159   * configuration defined by {@code that}.  Intuitively, this checks for structural properties
160   * and ignores non-structural properties.  Specifically, this method checks whether the
161   * configurations have compatible support for:
162   * <ul>
163   * <li>parallel edges
164   * <li>self-loops
165   * </ul>
166   *
167   * <p>It does not compare expected values for numbers of edges or nodes,
168   * and it is not equivalent to {@code Object.equals}.
169   */
170  public boolean compatibleWith(GraphConfig that) {
171    checkNotNull(that, "that");
172    return this.multigraph == that.multigraph
173        && this.selfLoopsAllowed == that.selfLoopsAllowed;
174  }
175
176  @Override
177  public boolean equals(@Nullable Object object) {
178    if (object instanceof GraphConfig) {
179      GraphConfig that = (GraphConfig) object;
180      return this.multigraph == that.multigraph
181          && this.selfLoopsAllowed == that.selfLoopsAllowed
182          && this.expectedNodeCount.equals(that.expectedNodeCount)
183          && this.expectedEdgeCount.equals(that.expectedEdgeCount);
184    }
185    return false;
186  }
187
188  @Override
189  public int hashCode() {
190    return Objects.hashCode(expectedNodeCount, expectedEdgeCount, multigraph, selfLoopsAllowed);
191  }
192
193  @Override
194  public String toString() {
195    List<String> properties = Lists.newArrayList();
196    if (multigraph) {
197      properties.add("multigraph");
198    }
199
200    if (selfLoopsAllowed) {
201      properties.add("self-loops allowed");
202    } else {
203      properties.add("self-loops disallowed");
204    }
205
206    return Joiner.on(',').join(properties);
207  }
208}
