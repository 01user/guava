042/**
043 * A readable source of bytes, such as a file. Unlike an {@link InputStream}, a
044 * {@code ByteSource} is not an open, stateful stream for input that can be read and closed.
045 * Instead, it is an immutable <i>supplier</i> of {@code InputStream} instances.
046 *
047 * <p>{@code ByteSource} provides two kinds of methods:
048 * <ul>
049 *   <li><b>Methods that return a stream:</b> These methods should return a <i>new</i>, independent
050 *   instance each time they are called. The caller is responsible for ensuring that the returned
051 *   stream is closed.
052 *   <li><b>Convenience methods:</b> These are implementations of common operations that are
053 *   typically implemented by opening a stream using one of the methods in the first category, doing
054 *   something and finally closing the stream that was opened.
055 * </ul>
056 *
057 * @since 14.0
058 * @author Colin Decker
059 */
060public abstract class ByteSource {
061
062  private static final int BUF_SIZE = 0x1000; // 4K
063
064  /**
065   * Constructor for use by subclasses.
066   */
067  protected ByteSource() {}

074    return new AsCharSource(charset);

077  /**
078   * Opens a new {@link InputStream} for reading from this source. This method should return a new,
079   * independent stream each time it is called.
080   *
081   * <p>The caller is responsible for ensuring that the returned stream is closed.
082   *
083   * @throws IOException if an I/O error occurs in the process of opening the stream
084   */
085  public abstract InputStream openStream() throws IOException;

087  /**
088   * Opens a new buffered {@link InputStream} for reading from this source. The returned stream is
089   * not required to be a {@link BufferedInputStream} in order to allow implementations to simply
090   * delegate to {@link #openStream()} when the stream returned by that method does not benefit
091   * from additional buffering (for example, a {@code ByteArrayInputStream}). This method should
092   * return a new, independent stream each time it is called.
093   *
094   * <p>The caller is responsible for ensuring that the returned stream is closed.
095   *
096   * @throws IOException if an I/O error occurs in the process of opening the stream
097   * @since 15.0 (in 14.0 with return type {@link BufferedInputStream})
098   */
099  public InputStream openBufferedStream() throws IOException {
100    InputStream in = openStream();
101    return (in instanceof BufferedInputStream)
102        ? (BufferedInputStream) in
103        : new BufferedInputStream(in);
104  }

116  /**
117   * Returns whether the source has zero bytes. The default implementation returns true if
118   * {@link #sizeIfKnown} returns zero, falling back to opening a stream and checking for
119   * EOF if the size is not known.
120   *
121   * <p>Note that, in cases where {@code sizeIfKnown} returns zero, it is <i>possible</i> that bytes
122   * are actually available for reading. (For example, some special files may return a size of 0
123   * despite actually having content when read.) This means that a source may return {@code true}
124   * from {@code isEmpty()} despite having readable content.
125   *
126   * @throws IOException if an I/O error occurs
127   * @since 15.0
128   */
129  public boolean isEmpty() throws IOException {
130    Optional<Long> sizeIfKnown = sizeIfKnown();
131    if (sizeIfKnown.isPresent() && sizeIfKnown.get() == 0L) {
132      return true;
133    }
134    Closer closer = Closer.create();
135    try {
136      InputStream in = closer.register(openStream());
137      return in.read() == -1;
138    } catch (Throwable e) {
139      throw closer.rethrow(e);
140    } finally {
141      closer.close();
142    }
143  }

145  /**
146   * Returns the size of this source in bytes, if the size can be easily determined without
147   * actually opening the data stream.
148   *
149   * <p>The default implementation returns {@link Optional#absent}. Some sources, such as a file,
150   * may return a non-absent value. Note that in such cases, it is <i>possible</i> that this method
151   * will return a different number of bytes than would be returned by reading all of the bytes (for
152   * example, some special files may return a size of 0 despite actually having content when read).
153   *
154   * <p>Additionally, for mutable sources such as files, a subsequent read may return a different
155   * number of bytes if the contents are changed.
156   *
157   * @since 19.0
158   */
159  @Beta
160  public Optional<Long> sizeIfKnown() {
161    return Optional.absent();
162  }
163
164  /**
165   * Returns the size of this source in bytes, even if doing so requires opening and traversing
166   * an entire stream. To avoid a potentially expensive operation, see {@link #sizeIfKnown}.
167   *
168   * <p>The default implementation calls {@link #sizeIfKnown} and returns the value if present.
169   * If absent, it will fall back to a heavyweight operation that will open a stream, read (or
170   * {@link InputStream#skip(long) skip}, if possible) to the end of the stream and return the total
171   * number of bytes that were read.
172   *
173   * <p>Note that for some sources that implement {@link #sizeIfKnown} to provide a more efficient
174   * implementation, it is <i>possible</i> that this method will return a different number of bytes
175   * than would be returned by reading all of the bytes (for example, some special files may return
176   * a size of 0 despite actually having content when read).
177   *
178   * <p>In either case, for mutable sources such as files, a subsequent read may return a different
179   * number of bytes if the contents are changed.
180   *
181   * @throws IOException if an I/O error occurs in the process of reading the size of this source
182   */
183  public long size() throws IOException {
184    Optional<Long> sizeIfKnown = sizeIfKnown();
185    if (sizeIfKnown.isPresent()) {
186      return sizeIfKnown.get();
187    }
188
189    Closer closer = Closer.create();
190    try {
191      InputStream in = closer.register(openStream());
192      return countBySkipping(in);
193    } catch (IOException e) {
194      // skip may not be supported... at any rate, try reading
195    } finally {
196      closer.close();
197    }
198
199    closer = Closer.create();
200    try {
201      InputStream in = closer.register(openStream());
202      return countByReading(in);
203    } catch (Throwable e) {
204      throw closer.rethrow(e);
205    } finally {
206      closer.close();
207    }
208  }
209
210  /**
211   * Counts the bytes in the given input stream using skip if possible. Returns SKIP_FAILED if the
212   * first call to skip threw, in which case skip may just not be supported.
213   */
214  private long countBySkipping(InputStream in) throws IOException {
215    long count = 0;
216    while (true) {
217      // don't try to skip more than available()
218      // things may work really wrong with FileInputStream otherwise
219      long skipped = in.skip(Math.min(in.available(), Integer.MAX_VALUE));
220      if (skipped <= 0) {
221        if (in.read() == -1) {
222          return count;
223        } else if (count == 0 && in.available() == 0) {
224          // if available is still zero after reading a single byte, it
225          // will probably always be zero, so we should countByReading
226          throw new IOException();
227        }
228        count++;
229      } else {
230        count += skipped;
231      }
232    }
233  }
234
235  private static final byte[] countBuffer = new byte[BUF_SIZE];
236
237  private long countByReading(InputStream in) throws IOException {
238    long count = 0;
239    long read;
240    while ((read = in.read(countBuffer)) != -1) {
241      count += read;
242    }
243    return count;
244  }

247   * Copies the contents of this byte source to the given {@code OutputStream}. Does not close
248   * {@code output}.
249   *
250   * @throws IOException if an I/O error occurs in the process of reading from this source or

254    checkNotNull(output);

256    Closer closer = Closer.create();
257    try {
258      InputStream in = closer.register(openStream());
259      return ByteStreams.copy(in, output);
260    } catch (Throwable e) {

306   * Reads the contents of this byte source using the given {@code processor} to process bytes as
307   * they are read. Stops when all bytes have been read or the consumer returns {@code false}.
308   * Returns the result produced by the processor.
309   *
310   * @throws IOException if an I/O error occurs in the process of reading from this source or if
311   *     {@code processor} throws an {@code IOException}
312   * @since 16.0
313   */
314  @Beta
315  public <T> T read(ByteProcessor<T> processor) throws IOException {
316    checkNotNull(processor);
317
318    Closer closer = Closer.create();
319    try {
320      InputStream in = closer.register(openStream());
321      return ByteStreams.readBytes(in, processor);
322    } catch (Throwable e) {
323      throw closer.rethrow(e);
324    } finally {
325      closer.close();
326    }

330   * Hashes the contents of this byte source using the given hash function.
331   *
332   * @throws IOException if an I/O error occurs in the process of reading from this source
333   */
334  public HashCode hash(HashFunction hashFunction) throws IOException {
335    Hasher hasher = hashFunction.newHasher();
336    copyTo(Funnels.asOutputStream(hasher));
337    return hasher.hash();

340  /**
341   * Checks that the contents of this byte source are equal to the contents of the given byte
342   * source.
343   *
344   * @throws IOException if an I/O error occurs in the process of reading from this source or
345   *     {@code other}
346   */
347  public boolean contentEquals(ByteSource other) throws IOException {
348    checkNotNull(other);
349
350    byte[] buf1 = new byte[BUF_SIZE];
351    byte[] buf2 = new byte[BUF_SIZE];
352
353    Closer closer = Closer.create();
354    try {
355      InputStream in1 = closer.register(openStream());
356      InputStream in2 = closer.register(other.openStream());
357      while (true) {
358        int read1 = ByteStreams.read(in1, buf1, 0, BUF_SIZE);
359        int read2 = ByteStreams.read(in2, buf2, 0, BUF_SIZE);
360        if (read1 != read2 || !Arrays.equals(buf1, buf2)) {
361          return false;
362        } else if (read1 != BUF_SIZE) {
363          return true;
364        }
365      }
366    } catch (Throwable e) {
367      throw closer.rethrow(e);
368    } finally {
369      closer.close();
370    }
371  }

468  /**
469   * A view of a subsection of the containing byte source.
470   */
471  private final class SlicedByteSource extends ByteSource {
472
473    final long offset;
474    final long length;
475
476    SlicedByteSource(long offset, long length) {
477      checkArgument(offset >= 0, "offset (%s) may not be negative", offset);
478      checkArgument(length >= 0, "length (%s) may not be negative", length);
479      this.offset = offset;
480      this.length = length;
481    }
482
483    @Override
484    public InputStream openStream() throws IOException {
485      return sliceStream(ByteSource.this.openStream());
486    }
487
488    @Override
489    public InputStream openBufferedStream() throws IOException {
490      return sliceStream(ByteSource.this.openBufferedStream());
491    }
492
493    private InputStream sliceStream(InputStream in) throws IOException {
494      if (offset > 0) {
495        try {
496          ByteStreams.skipFully(in, offset);
497        } catch (Throwable e) {
498          Closer closer = Closer.create();
499          closer.register(in);
500          try {
501            throw closer.rethrow(e);
502          } finally {
503            closer.close();
504          }
505        }
506      }
507      return ByteStreams.limit(in, length);
508    }
509
510    @Override
511    public ByteSource slice(long offset, long length) {
512      checkArgument(offset >= 0, "offset (%s) may not be negative", offset);
513      checkArgument(length >= 0, "length (%s) may not be negative", length);
514      long maxLength = this.length - offset;
515      return ByteSource.this.slice(this.offset + offset, Math.min(length, maxLength));
516    }
517
518    @Override
519    public boolean isEmpty() throws IOException {
520      return length == 0 || super.isEmpty();
521    }
522
523    @Override
524    public Optional<Long> sizeIfKnown() {
525      Optional<Long> unslicedSize = ByteSource.this.sizeIfKnown();
526      if (unslicedSize.isPresent()) {
527        return Optional.of(Math.min(offset + length, unslicedSize.get()) - offset);
528      }
529      return Optional.absent();
530    }
531
532    @Override
533    public String toString() {
534      return ByteSource.this.toString() + ".slice(" + offset + ", " + length + ")";
535    }
536  }

538  private static class ByteArrayByteSource extends ByteSource {
539
540    final byte[] bytes;
541    final int offset;
542    final int length;
543
544    ByteArrayByteSource(byte[] bytes) {
545      this(bytes, 0, bytes.length);
546    }
547
548    // NOTE: Preconditions are enforced by slice, the only non-trivial caller.
549    ByteArrayByteSource(byte[] bytes, int offset, int length) {
550      this.bytes = bytes;
551      this.offset = offset;
552      this.length = length;
553    }
554
555    @Override
556    public InputStream openStream() {
557      return new ByteArrayInputStream(bytes, offset, length);
558    }
559
560    @Override
561    public InputStream openBufferedStream() throws IOException {
562      return openStream();
563    }
564
565    @Override
566    public boolean isEmpty() {
567      return length == 0;
568    }
569
570    @Override
571    public long size() {
572      return length;
573    }
574
575    @Override
576    public Optional<Long> sizeIfKnown() {
577      return Optional.of((long) length);
578    }
579
580    @Override
581    public byte[] read() {
582      return Arrays.copyOfRange(bytes, offset, offset + length);
583    }
584
585    @Override
586    public long copyTo(OutputStream output) throws IOException {
587      output.write(bytes, offset, length);
588      return length;
589    }
590
591    @Override
592    public <T> T read(ByteProcessor<T> processor) throws IOException {
593      processor.processBytes(bytes, offset, length);
594      return processor.getResult();
595    }
596
597    @Override
598    public HashCode hash(HashFunction hashFunction) throws IOException {
599      return hashFunction.hashBytes(bytes, offset, length);
600    }
601
602    @Override
603    public ByteSource slice(long offset, long length) {
604      checkArgument(offset >= 0, "offset (%s) may not be negative", offset);
605      checkArgument(length >= 0, "length (%s) may not be negative", length);
606
607      offset = Math.min(offset, this.length);
608      length = Math.min(length, this.length - offset);
609      int newOffset = this.offset + (int) offset;
610      return new ByteArrayByteSource(bytes, newOffset, (int) length);
611    }
612
613    @Override
614    public String toString() {
615      return "ByteSource.wrap("
616          + Ascii.truncate(BaseEncoding.base16().encode(bytes, offset, length), 30, "...") + ")";
617    }
618  }

645  private static final class ConcatenatedByteSource extends ByteSource {
646
647    final Iterable<? extends ByteSource> sources;
648
649    ConcatenatedByteSource(Iterable<? extends ByteSource> sources) {
650      this.sources = checkNotNull(sources);
651    }
652
653    @Override
654    public InputStream openStream() throws IOException {
655      return new MultiInputStream(sources.iterator());
656    }
657
658    @Override
659    public boolean isEmpty() throws IOException {
660      for (ByteSource source : sources) {
661        if (!source.isEmpty()) {
662          return false;
663        }
664      }
665      return true;
666    }
667
668    @Override
669    public Optional<Long> sizeIfKnown() {
670      long result = 0L;
671      for (ByteSource source : sources) {
672        Optional<Long> sizeIfKnown = source.sizeIfKnown();
673        if (!sizeIfKnown.isPresent()) {
674          return Optional.absent();
675        }
676        result += sizeIfKnown.get();
677      }
678      return Optional.of(result);
679    }
680
681    @Override
682    public long size() throws IOException {
683      long result = 0L;
684      for (ByteSource source : sources) {
685        result += source.size();
686      }
687      return result;
688    }
689
690    @Override
691    public String toString() {
692      return "ByteSource.concat(" + sources + ")";
693    }
694  }
