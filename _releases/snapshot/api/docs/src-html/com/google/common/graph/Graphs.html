047  /**
048   * Returns the node at the other end of {@code e} from {@code n}.
049   */
050  @SuppressWarnings("unchecked")
051  public static <N> N oppositeNode(
052      UndirectedGraph<N, ?> undirectedGraph, Object edge, Object node) {
053    checkNotNull(edge, "edge");
054    checkNotNull(node, "node");
055    Set<N> incidentNodes = undirectedGraph.incidentNodes(edge);
056    checkArgument(incidentNodes.contains(node), "Edge %s is not incident to node %s", edge, node);
057    for (N incidentNode : incidentNodes) {
058      if (!incidentNode.equals(node)) {
059        return incidentNode;
060      }
061    }
062    // Reaching this point means that incidentNodes contains only one node,
063    // which is node, hence edge is a self-loop for node, and node is its own opposite
064    return (N) node;
065  }

067  /**
068   * Adds {@code edge} to {@code graph} with the specified incident {@code nodes}, in the order
069   * returned by {@code nodes}' iterator.
070   */
071  @CanIgnoreReturnValue
072  public static <N, E> boolean addEdge(Graph<N, E> graph, E edge, Iterable<N> nodes) {
073    checkNotNull(graph, "graph");
074    checkNotNull(nodes, "nodes");
075    checkNotNull(edge, "edge");
076    if (graph instanceof Hypergraph) {
077      return ((Hypergraph<N, E>) graph).addEdge(edge, nodes);
078    }
079
080    Iterator<N> it = nodes.iterator();
081    checkArgument(
082        it.hasNext(), "'graph' is not a Hypergraph, and 'nodes' has < 1 elements: %s", nodes);
083    N n1 = it.next();
084    if (it.hasNext()) {
085      N n2 = it.next();
086      checkArgument(
087          !it.hasNext(), "'graph' is not a Hypergraph, and 'nodes' has > 2 elements: %s", nodes);
088      return graph.addEdge(edge, n1, n2);
089    } else {
090      return graph.addEdge(edge, n1, n1);
091    }
092  }

094  /**
095   * Creates a mutable copy of {@code graph}, using the same node and edge elements.
096   */
097  public static <N, E> DirectedGraph<N, E> copyOf(DirectedGraph<N, E> graph) {
098    checkNotNull(graph, "graph");
099    DirectedGraph<N, E> copy = createDirected(graph.config()
100            .expectedNodeCount(graph.nodes().size())
101            .expectedEdgeCount(graph.edges().size()));
102    mergeNodesFrom(graph, copy);
103    mergeEdgesFrom(graph, copy);
104    return copy;
105  }

107  /**
108   * Creates a mutable copy of {@code graph}, using all of its elements that satisfy
109   * {@code nodePredicate} and {@code edgePredicate}.
110   */
111  public static <N, E> DirectedGraph<N, E> copyOf(
112      DirectedGraph<N, E> graph,
113      Predicate<? super N> nodePredicate,
114      Predicate<? super E> edgePredicate) {
115    checkNotNull(graph, "graph");
116    checkNotNull(nodePredicate, "nodePredicate");
117    checkNotNull(edgePredicate, "edgePredicate");
118    DirectedGraph<N, E> copy = createDirected(graph.config()
119            .expectedNodeCount(graph.nodes().size())
120            .expectedEdgeCount(graph.edges().size()));
121    mergeNodesFrom(graph, copy, nodePredicate);
122    // We can't just call mergeEdgesFrom(graph, copy, edgePredicate) because addEdge() can add
123    // the edge's incident nodes if they are not present; we need to run them past nodePredicate.
124    if (edgePredicate.equals(Predicates.<E>alwaysFalse())) {
125      return copy; // no edges to add
126    }
127
128    for (E edge : graph.edges()) {
129      if (edgePredicate.apply(edge)) {
130        N source = graph.source(edge);
131        N target = graph.target(edge);
132        if (nodePredicate.apply(source) && nodePredicate.apply(target)) {
133          copy.addEdge(edge, source, target);
134        }
135      }
136    }
137
138    return copy;
139  }

141  /**
142   * Copies all nodes from {@code original} into {@code copy}.
143   */
144  public static <N, E> void mergeNodesFrom(Graph<N, E> original, Graph<N, E> copy) {
145    checkNotNull(original, "original");
146    checkNotNull(copy, "copy");
147    for (N node : original.nodes()) {
148      copy.addNode(node);
149    }
150  }

152  /**
153   * Copies all nodes from {@code original} into {@code copy} that satisfy {@code nodePredicate}.
154   */
155  public static <N, E> void mergeNodesFrom(
156      Graph<N, E> original, Graph<N, E> copy, Predicate<? super N> nodePredicate) {
157    checkNotNull(original, "original");
158    checkNotNull(copy, "copy");
159    checkNotNull(nodePredicate, "nodePredicate");
160    if (nodePredicate.equals(Predicates.<N>alwaysFalse())) {
161      return; // nothing to do
162    }
163
164    if (nodePredicate.equals(Predicates.<N>alwaysTrue())) {
165      mergeNodesFrom(original, copy); // optimization
166    } else {
167      for (N node : original.nodes()) {
168        if (nodePredicate.apply(node)) {
169          copy.addNode(node);
170        }
171      }
172    }
173  }

175  /**
176   * Copies all edges from {@code original} into {@code copy}.  Also copies all nodes incident
177   * to these edges.
178   */
179  public static <N, E> void mergeEdgesFrom(DirectedGraph<N, E> original, DirectedGraph<N, E> copy) {
180    checkNotNull(original, "original");
181    checkNotNull(copy, "copy");
182    for (E edge : original.edges()) {
183      copy.addEdge(edge, original.source(edge), original.target(edge));
184    }
185  }

224  /**
225   * Creates a mutable copy of {@code graph}, using all of its elements that satisfy
226   * {@code nodePredicate} and {@code edgePredicate}.
227   */
228  public static <N, E> UndirectedGraph<N, E> copyOf(
229      UndirectedGraph<N, E> graph,
230      Predicate<? super N> nodePredicate,
231      Predicate<? super E> edgePredicate) {
232    checkNotNull(graph, "graph");
233    checkNotNull(nodePredicate, "nodePredicate");
234    checkNotNull(edgePredicate, "edgePredicate");
235    UndirectedGraph<N, E> copy = createUndirected(graph.config()
236            .expectedNodeCount(graph.nodes().size())
237            .expectedEdgeCount(graph.edges().size()));
238    mergeNodesFrom(graph, copy, nodePredicate);
239
240    // We can't just call mergeEdgesFrom(graph, copy, edgePredicate) because addEdge() can add
241    // the edge's incident nodes if they are not present; we need to run them past nodePredicate.
242    for (E edge : graph.edges()) {
243      if (edgePredicate.apply(edge)) {
244        boolean nodesOk = true;
245        Set<N> incidentNodes = graph.incidentNodes(edge);
246        for (N node : incidentNodes) {
247          nodesOk &= nodePredicate.apply(node);
248        }
249        if (nodesOk) {
250          addEdge(copy, edge, incidentNodes);
251        }
252      }
253    }
254
255    return copy;
256  }

258  /**
259   * Copies all edges from {@code original} into {@code copy}.  Also copies all nodes incident
260   * to these edges.
261   */
262  public static <N, E> void mergeEdgesFrom(Graph<N, E> original, Graph<N, E> copy) {
263    checkNotNull(original, "original");
264    checkNotNull(copy, "copy");
265    for (E edge : original.edges()) {
266      addEdge(copy, edge, original.incidentNodes(edge));
267    }
268  }

271   * Copies all edges from {@code original} into {@code copy} that satisfy {@code edgePredicate}.
272   * Also copies all nodes incident to these edges.

274  // NOTE: this is identical to mergeEdgesFrom(DirectedGraph) except for the call to addEdge
275  public static <N, E> void mergeEdgesFrom(Graph<N, E> original, Graph<N, E> copy,
276      Predicate<? super E> edgePredicate) {
277    checkNotNull(original, "original");
278    checkNotNull(copy, "copy");
279    checkNotNull(edgePredicate, "edgePredicate");
280    if (edgePredicate.equals(Predicates.<E>alwaysFalse())) {
281      return; // nothing to do
282    }
283
284    if (edgePredicate.equals(Predicates.<E>alwaysTrue())) {
285      mergeEdgesFrom(original, copy); // optimization
286    } else {
287      for (E edge : original.edges()) {
288        if (edgePredicate.apply(edge)) {
289          addEdge(copy, edge, original.incidentNodes(edge));
290        }
291      }
292    }
293  }

298   */
299  public static <N, E> void copyFrom(
300      Graph<N, E> original,
301      Graph<N, E> copy,

357  /**
358   * Returns true iff {@code g1} and {@code g2} have the same node and edge sets and each edge
359   * has the same source and target in both graphs.
360   *
361   * @see Graph#equals(Object)
362   */
363  public static <N, E> boolean equal(
364      @Nullable DirectedGraph<?, ?> g1, @Nullable DirectedGraph<?, ?> g2) {
365    if (g1 == g2) {
366      return true;
367    }
368
369    if (g1 == null || g2 == null) {
370      return false;
371    }
372
373    if (!g1.nodes().equals(g2.nodes()) || !g1.edges().equals(g2.edges())) {
374      return false;
375    }
376
377    for (Object e : g1.edges()) {
378      if (!g1.source(e).equals(g2.source(e)) || !g1.target(e).equals(g2.target(e))) {
379        return false;
380      }
381    }
382    return true;
383  }
384
385  /**
386   * Returns true iff {@code g1} and {@code g2} have the same node and edge sets and each edge
387   * has the same incident node set in both graphs.
388   *
389   * @see Graph#equals(Object)
390   */
391  public static boolean equal(@Nullable Graph<?, ?> g1, @Nullable Graph<?, ?> g2) {
392    if (g1 == g2) {
393      return true;
394    }
395
396    if (g1 == null || g2 == null) {
397      return false;
398    }
399
400    if (!g1.nodes().equals(g2.nodes()) || !g1.edges().equals(g2.edges())) {
401      return false;
402    }
403
404    for (Object e : g1.edges()) {
405      if (!g1.incidentNodes(e).equals(g2.incidentNodes(e))) {
406        return false;
407      }
408    }
409    return true;
410  }

412  /**
413   * Returns a {@link Predicate} that returns {@code true} if the input edge is not a self-loop in
414   * {@code graph}. A self-loop is defined as an edge whose set of incident nodes has exactly one
415   * element. The predicate's {@code apply} method will throw a {@link IllegalStateException} if
416   * {@code graph} does not contain {@code edge}.
417   */
418  public static <E> Predicate<E> noSelfLoopPredicate(final Graph<?, E> graph) {
419    checkNotNull(graph, "graph");
420    return new Predicate<E>() {
421      @Override
422      public boolean apply(E edge) {
423        checkState(graph.edges().contains(edge), "Graph does not contain edge %s", edge);
424        return graph.incidentNodes(edge).size() != 1;
425      }
426    };
427  }
