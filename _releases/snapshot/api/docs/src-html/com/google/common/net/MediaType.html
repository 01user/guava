054/**
055 * Represents an <a href="http://en.wikipedia.org/wiki/Internet_media_type">Internet Media Type</a>
056 * (also known as a MIME Type or Content Type). This class also supports the concept of media ranges
057 * <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1">defined by HTTP/1.1</a>.
058 * As such, the {@code *} character is treated as a wildcard and is used to represent any acceptable
059 * type or subtype value. A media type may not have wildcard type with a declared subtype. The
060 * {@code *} character has no special meaning as part of a parameter. All values for type, subtype,
061 * parameter attributes or parameter values must be valid according to RFCs
062 * <a href="http://www.ietf.org/rfc/rfc2045.txt">2045</a> and
063 * <a href="http://www.ietf.org/rfc/rfc2046.txt">2046</a>.
064 *
065 * <p>All portions of the media type that are case-insensitive (type, subtype, parameter attributes)
066 * are normalized to lowercase. The value of the {@code charset} parameter is normalized to
067 * lowercase, but all others are left as-is.
068 *
069 * <p>Note that this specifically does <strong>not</strong> represent the value of the MIME
070 * {@code Content-Type} header and as such has no support for header-specific considerations such as
071 * line folding and comments.
072 *
073 * <p>For media types that take a charset the predefined constants default to UTF-8 and have a
074 * "_UTF_8" suffix. To get a version without a character set, use {@link #withoutParameters}.
075 *
076 * @since 12.0
077 *
078 * @author Gregory Kick
079 */
080@Beta
081@GwtCompatible
082@Immutable
083public final class MediaType {
084  private static final String CHARSET_ATTRIBUTE = "charset";
085  private static final ImmutableListMultimap<String, String> UTF_8_CONSTANT_PARAMETERS =
086      ImmutableListMultimap.of(CHARSET_ATTRIBUTE, Ascii.toLowerCase(UTF_8.name()));

088  /** Matcher for type, subtype and attributes. */
089  private static final CharMatcher TOKEN_MATCHER = ascii().and(javaIsoControl().negate())
090      .and(CharMatcher.isNot(' '))
091      .and(CharMatcher.noneOf("()<>@,;:\\\"/[]?="));
092  private static final CharMatcher QUOTED_TEXT_MATCHER = ascii()
093      .and(CharMatcher.noneOf("\"\\\r"));

095   * This matches the same characters as linear-white-space from RFC 822, but we make no effort to
096   * enforce any particular rules with regards to line folding as stated in the class docs.
097   */
098  private static final CharMatcher LINEAR_WHITE_SPACE = CharMatcher.anyOf(" \t\r\n");

109  private static final Map<MediaType, MediaType> KNOWN_TYPES = Maps.newHashMap();

111  private static MediaType createConstant(String type, String subtype) {
112    return addKnownType(new MediaType(type, subtype, ImmutableListMultimap.<String, String>of()));
113  }
114
115  private static MediaType createConstantUtf8(String type, String subtype) {
116    return addKnownType(new MediaType(type, subtype, UTF_8_CONSTANT_PARAMETERS));
117  }
118
119  private static MediaType addKnownType(MediaType mediaType) {
120    KNOWN_TYPES.put(mediaType, mediaType);
121    return mediaType;
122  }

526  /**
527   * Returns an optional charset for the value of the charset parameter if it is specified.
528   *
529   * @throws IllegalStateException if multiple charset values have been set for this media type
530   * @throws IllegalCharsetNameException if a charset value is present, but illegal
531   * @throws UnsupportedCharsetException if a charset value is present, but no support is available
532   *     in this instance of the Java virtual machine
533   */
534  public Optional<Charset> charset() {
535    ImmutableSet<String> charsetValues = ImmutableSet.copyOf(parameters.get(CHARSET_ATTRIBUTE));
536    switch (charsetValues.size()) {
537      case 0:
538        return Optional.absent();
539      case 1:
540        return Optional.of(Charset.forName(Iterables.getOnlyElement(charsetValues)));
541      default:
542        throw new IllegalStateException("Multiple charset values defined: " + charsetValues);
543    }
544  }

563  /**
564   * <em>Replaces</em> all parameters with the given attribute with a single parameter with the
565   * given value. If multiple parameters with the same attributes are necessary use
566   * {@link #withParameters}. Prefer {@link #withCharset} for setting the {@code charset} parameter
567   * when using a {@link Charset} object.
568   *
569   * @throws IllegalArgumentException if either {@code attribute} or {@code value} is invalid
570   */
571  public MediaType withParameter(String attribute, String value) {
572    checkNotNull(attribute);
573    checkNotNull(value);
574    String normalizedAttribute = normalizeToken(attribute);
575    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
576    for (Entry<String, String> entry : parameters.entries()) {
577      String key = entry.getKey();
578      if (!normalizedAttribute.equals(key)) {
579        builder.put(key, entry.getValue());
580      }
581    }
582    builder.put(normalizedAttribute, normalizeParameterValue(normalizedAttribute, value));
583    MediaType mediaType = new MediaType(type, subtype, builder.build());
584    // Return one of the constants if the media type is a known type.
585    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
586  }

607  /**
608   * Returns {@code true} if this instance falls within the range (as defined by
609   * <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html">the HTTP Accept header</a>)
610   * given by the argument according to three criteria:
611   *
612   * <ol>
613   * <li>The type of the argument is the wildcard or equal to the type of this instance.
614   * <li>The subtype of the argument is the wildcard or equal to the subtype of this instance.
615   * <li>All of the parameters present in the argument are present in this instance.
616   * </ol>
617   *
618   * <p>For example: <pre>   {@code
619   *   PLAIN_TEXT_UTF_8.is(PLAIN_TEXT_UTF_8) // true
620   *   PLAIN_TEXT_UTF_8.is(HTML_UTF_8) // false
621   *   PLAIN_TEXT_UTF_8.is(ANY_TYPE) // true
622   *   PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE) // true
623   *   PLAIN_TEXT_UTF_8.is(ANY_IMAGE_TYPE) // false
624   *   PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_8)) // true
625   *   PLAIN_TEXT_UTF_8.withoutParameters().is(ANY_TEXT_TYPE.withCharset(UTF_8)) // false
626   *   PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_16)) // false}</pre>
627   *
628   * <p>Note that while it is possible to have the same parameter declared multiple times within a
629   * media type this method does not consider the number of occurrences of a parameter.  For
630   * example, {@code "text/plain; charset=UTF-8"} satisfies
631   * {@code "text/plain; charset=UTF-8; charset=UTF-8"}.
632   */
633  public boolean is(MediaType mediaTypeRange) {
634    return (mediaTypeRange.type.equals(WILDCARD) || mediaTypeRange.type.equals(this.type))
635        && (mediaTypeRange.subtype.equals(WILDCARD) || mediaTypeRange.subtype.equals(this.subtype))
636        && this.parameters.entries().containsAll(mediaTypeRange.parameters.entries());
637  }

694  private static MediaType create(String type, String subtype,
695      Multimap<String, String> parameters) {
696    checkNotNull(type);
697    checkNotNull(subtype);
698    checkNotNull(parameters);
699    String normalizedType = normalizeToken(type);
700    String normalizedSubtype = normalizeToken(subtype);
701    checkArgument(!WILDCARD.equals(normalizedType) || WILDCARD.equals(normalizedSubtype),
702        "A wildcard type cannot be used with a non-wildcard subtype");
703    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
704    for (Entry<String, String> entry : parameters.entries()) {
705      String attribute = normalizeToken(entry.getKey());
706      builder.put(attribute, normalizeParameterValue(attribute, entry.getValue()));
707    }
708    MediaType mediaType = new MediaType(normalizedType, normalizedSubtype, builder.build());
709    // Return one of the constants if the media type is a known type.
710    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
711  }

713  private static String normalizeToken(String token) {
714    checkArgument(TOKEN_MATCHER.matchesAllOf(token));
715    return Ascii.toLowerCase(token);
716  }
717
718  private static String normalizeParameterValue(String attribute, String value) {
719    return CHARSET_ATTRIBUTE.equals(attribute) ? Ascii.toLowerCase(value) : value;
720  }

722  /**
723   * Parses a media type from its string representation.
724   *
725   * @throws IllegalArgumentException if the input is not parsable
726   */
727  public static MediaType parse(String input) {
728    checkNotNull(input);
729    Tokenizer tokenizer = new Tokenizer(input);
730    try {
731      String type = tokenizer.consumeToken(TOKEN_MATCHER);
732      tokenizer.consumeCharacter('/');
733      String subtype = tokenizer.consumeToken(TOKEN_MATCHER);
734      ImmutableListMultimap.Builder<String, String> parameters = ImmutableListMultimap.builder();
735      while (tokenizer.hasMore()) {
736        tokenizer.consumeCharacter(';');
737        tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);
738        String attribute = tokenizer.consumeToken(TOKEN_MATCHER);
739        tokenizer.consumeCharacter('=');
740        final String value;
741        if ('"' == tokenizer.previewChar()) {
742          tokenizer.consumeCharacter('"');
743          StringBuilder valueBuilder = new StringBuilder();
744          while ('"' != tokenizer.previewChar()) {
745            if ('\\' == tokenizer.previewChar()) {
746              tokenizer.consumeCharacter('\\');
747              valueBuilder.append(tokenizer.consumeCharacter(ascii()));
748            } else {
749              valueBuilder.append(tokenizer.consumeToken(QUOTED_TEXT_MATCHER));
750            }
751          }
752          value = valueBuilder.toString();
753          tokenizer.consumeCharacter('"');
754        } else {
755          value = tokenizer.consumeToken(TOKEN_MATCHER);
756        }
757        parameters.put(attribute, value);
758      }
759      return create(type, subtype, parameters.build());
760    } catch (IllegalStateException e) {
761      throw new IllegalArgumentException("Could not parse '" + input + "'", e);
762    }
763  }
764
765  private static final class Tokenizer {
766    final String input;
767    int position = 0;
768
769    Tokenizer(String input) {
770      this.input = input;
771    }
772
773    String consumeTokenIfPresent(CharMatcher matcher) {
774      checkState(hasMore());
775      int startPosition = position;
776      position = matcher.negate().indexIn(input, startPosition);
777      return hasMore() ? input.substring(startPosition, position) : input.substring(startPosition);
778    }
779
780    String consumeToken(CharMatcher matcher) {
781      int startPosition = position;
782      String token = consumeTokenIfPresent(matcher);
783      checkState(position != startPosition);
784      return token;
785    }
786
787    char consumeCharacter(CharMatcher matcher) {
788      checkState(hasMore());
789      char c = previewChar();
790      checkState(matcher.matches(c));
791      position++;
792      return c;
793    }
794
795    char consumeCharacter(char c) {
796      checkState(hasMore());
797      checkState(previewChar() == c);
798      position++;
799      return c;
800    }
801
802    char previewChar() {
803      checkState(hasMore());
804      return input.charAt(position);
805    }
806
807    boolean hasMore() {
808      return (position >= 0) && (position < input.length());
809    }
810  }

812  @Override public boolean equals(@Nullable Object obj) {
813    if (obj == this) {
814      return true;
815    } else if (obj instanceof MediaType) {
816      MediaType that = (MediaType) obj;
817      return this.type.equals(that.type)
818          && this.subtype.equals(that.subtype)
819          // compare parameters regardless of order
820          && this.parametersAsMap().equals(that.parametersAsMap());
821    } else {
822      return false;
823    }
824  }

827    // racy single-check idiom
828    int h = hashCode;
829    if (h == 0) {
830      h = Objects.hashCode(type, subtype, parametersAsMap());
831      hashCode = h;
832    }
833    return h;

836  private static final MapJoiner PARAMETER_JOINER = Joiner.on("; ").withKeyValueSeparator("=");
837
838  /**
839   * Returns the string representation of this media type in the format described in <a
840   * href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045</a>.
841   */
842  @Override public String toString() {
843    // racy single-check idiom, safe because String is immutable
844    String result = toString;
845    if (result == null) {
846      result = computeToString();
847      toString = result;
848    }
849    return result;
850  }
851
852  private String computeToString() {
853    StringBuilder builder = new StringBuilder().append(type).append('/').append(subtype);
854    if (!parameters.isEmpty()) {
855      builder.append("; ");
856      Multimap<String, String> quotedParameters = Multimaps.transformValues(parameters,
857          new Function<String, String>() {
858            @Override public String apply(String value) {
859              return TOKEN_MATCHER.matchesAllOf(value) ? value : escapeAndQuote(value);
860            }
861          });
862      PARAMETER_JOINER.appendTo(builder, quotedParameters.entries());
863    }
864    return builder.toString();
865  }
866
867  private static String escapeAndQuote(String value) {
868    StringBuilder escaped = new StringBuilder(value.length() + 16).append('"');
869    for (int i = 0; i < value.length(); i++) {
870      char ch = value.charAt(i);
871      if (ch == '\r' || ch == '\\' || ch == '"') {
872        escaped.append('\\');
873      }
874      escaped.append(ch);
875    }
876    return escaped.append('"').toString();
877  }
