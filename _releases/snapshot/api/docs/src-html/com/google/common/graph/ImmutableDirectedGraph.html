020import static com.google.common.base.Preconditions.checkNotNull;
021import static com.google.common.graph.GraphErrorMessageUtils.EDGE_NOT_IN_GRAPH;
022import static com.google.common.graph.GraphErrorMessageUtils.NODE_NOT_IN_GRAPH;

024import com.google.common.annotations.Beta;
025import com.google.common.collect.ImmutableMap;

035/**
036 * Implementation of an immutable directed graph consisting of nodes
037 * of type N and edges of type E.
038 *
039 * <p>This class maintains the following data structures:
040 * <ul>
041 * <li>For each node: sets of incoming and outgoing edges.
042 * <li>For each edge: references to the source and target nodes.
043 * </ul>
044 *
045 * <p>Some invariants/assumptions are maintained in this implementation:
046 * <ul>
047 * <li>An edge has exactly two end-points (source node and target node), which
048 *     may or may not be distinct.
049 * <li>By default, this is not a multigraph, that is, parallel edges (multiple
050 *     edges directed from node1 to node2) are not allowed.  If you want a multigraph,
051 *     build the graph with the 'multigraph' option:
052 *     <pre>ImmutableDirectedGraph.builder(Graphs.config().multigraph()).build();</pre>
053 * <li>Anti-parallel edges (same incident nodes but in opposite direction,
054 *     e.g. (node1, node2) and (node2, node1)) are always allowed.
055 * <li>Edges are not adjacent to themselves by definition. In the case of a
056 *     self-loop, a node can be adjacent to itself, but an edge will never be.
057 * </ul>
058 *
059 * <p>Most of the {@code Set}-returning accessors return an immutable set which is an internal
060 * data structure, hence they have a time complexity of O(1). The rest of these accessors build
061 * and return an immutable set that is <b>derived</b> from the internal data structures, hence
062 * they have the following time complexities:
063 * <ul>
064 * <li>{@code incidentEdges(node)}: O(d_node).
065 * <li>Methods that ask for adjacent nodes:
066 *     <ul>
067 *     <li>{@code adjacentNodes(node)}: O(d_node).
068 *     <li>{@code predecessors(node)}: O(inD_node).
069 *     <li>{@code successors(node)}: O(outD_node).
070 *     </ul>
071 * <li>{@code adjacentEdges(edge)}: O(d_node1 + d_node2), where node1 and node2 are
072 *     {@code edge}'s incident nodes.
073 * <li>{@code edgesConnecting(node1, node2)}: O(min(outD_node1, inD_node2)).
074 * </ul>
075 * where d_node is the degree of node, inD_node is the in-degree of node, and outD_node is the
076 * out-degree of node. The set returned by these methods is <b>not</b> cached, so every time the
077 * user calls the method, the same set will be reconstructed again.
078 *
079 * <p>All other accessors have a time complexity of O(1), except for
080 * {@code degree(node)}, whose time complexity is linear in the minimum of
081 * the out-degree and in-degree of {@code node}. This is due to a call to {@code edgesConnecting}.
082 *
083 * @author Joshua O'Madadhain
084 * @param <N> Node parameter type
085 * @param <E> Edge parameter type
086 * @see Graphs
087 * @since 20.0
088 */
089// TODO(user): Add support for sorted nodes/edges and/or hypergraphs.
090@Beta
091public final class ImmutableDirectedGraph<N, E> extends AbstractImmutableGraph<N, E>
092    implements DirectedGraph<N, E> {
093
094  // All nodes in the graph exist in this map
095  private final ImmutableMap<N, IncidentEdges<E>> nodeToIncidentEdges;
096  // All edges in the graph exist in this map
097  private final ImmutableMap<E, IncidentNodes<N>> edgeToIncidentNodes;
098  private final GraphConfig config;
099
100  private ImmutableDirectedGraph(Builder<N, E> builder) {
101    DirectedGraph<N, E> directedGraph = builder.directedGraph;
102    ImmutableMap.Builder<N, IncidentEdges<E>> nodeToIncidentEdgesBuilder = ImmutableMap.builder();
103    for (N node : directedGraph.nodes()) {
104      IncidentEdges<E> incidentEdges =
105          IncidentEdges.ofImmutable(directedGraph.inEdges(node), directedGraph.outEdges(node));
106      nodeToIncidentEdgesBuilder.put(node, incidentEdges);
107    }
108    nodeToIncidentEdges = nodeToIncidentEdgesBuilder.build();
109    ImmutableMap.Builder<E, IncidentNodes<N>> edgeToIncidentNodesBuilder = ImmutableMap.builder();
110    for (E edge : directedGraph.edges()) {
111      IncidentNodes<N> incidentNodes =
112          IncidentNodes.of(directedGraph.source(edge), directedGraph.target(edge));
113      edgeToIncidentNodesBuilder.put(edge, incidentNodes);
114    }
115    edgeToIncidentNodes = edgeToIncidentNodesBuilder.build();
116    this.config = directedGraph.config();
117  }
118
119  @Override
120  public Set<N> nodes() {
121    return nodeToIncidentEdges.keySet();
122  }
123
124  @Override
125  public Set<E> edges() {
126    return edgeToIncidentNodes.keySet();
127  }
128
129  @Override
130  public GraphConfig config() {
131    return config;
132  }
133
134  @Override
135  public Set<E> incidentEdges(Object node) {
136    return Sets.union(inEdges(node), outEdges(node)).immutableCopy();
137  }
138
139  @Override
140  public Set<N> incidentNodes(Object edge) {
141    checkNotNull(edge, "edge");
142    IncidentNodes<N> endpoints = edgeToIncidentNodes.get(edge);
143    checkArgument(endpoints != null, EDGE_NOT_IN_GRAPH, edge);
144    return endpoints.asImmutableSet();
145  }
146
147  @Override
148  public Set<N> adjacentNodes(Object node) {
149    return Sets.union(predecessors(node), successors(node)).immutableCopy();
150  }
151
152  @Override
153  public Set<E> adjacentEdges(Object edge) {
154    checkNotNull(edge, "edge");
155    IncidentNodes<N> endpoints = edgeToIncidentNodes.get(edge);
156    checkArgument(endpoints != null, EDGE_NOT_IN_GRAPH, edge);
157    Set<E> adjacentEdges = Sets.newLinkedHashSet();
158    adjacentEdges.addAll(incidentEdges(endpoints.target()));
159    adjacentEdges.addAll(incidentEdges(endpoints.source()));
160    // Edges are not adjacent to themselves by definition.
161    adjacentEdges.remove(edge);
162    return Collections.unmodifiableSet(adjacentEdges);
163  }
164
165  /**
166   * Returns the intersection of these two sets, using {@code Sets.intersection}:
167   * <ol>
168   * <li>Outgoing edges of {@code node1}.
169   * <li>Incoming edges of {@code node2}.
170   * </ol>
171   * The first argument passed to {@code Sets.intersection} is the smaller of the two sets.
172   *
173   * @see Sets#intersection
174   */
175  @Override
176  public Set<E> edgesConnecting(Object node1, Object node2) {
177    checkNotNull(node1, "node1");
178    checkNotNull(node2, "node2");
179    Set<E> sourceOutEdges = outEdges(node1);
180    Set<E> targetInEdges = inEdges(node2);
181    return sourceOutEdges.size() <= targetInEdges.size()
182        ? Sets.intersection(sourceOutEdges, targetInEdges).immutableCopy()
183        : Sets.intersection(targetInEdges, sourceOutEdges).immutableCopy();
184  }
185
186  @Override
187  public Set<E> inEdges(Object node) {
188    checkNotNull(node, "node");
189    IncidentEdges<E> incidentEdges = nodeToIncidentEdges.get(node);
190    checkArgument(incidentEdges != null, NODE_NOT_IN_GRAPH, node);
191    return incidentEdges.inEdges();
192  }
193
194  @Override
195  public Set<E> outEdges(Object node) {
196    checkNotNull(node, "node");
197    IncidentEdges<E> incidentEdges = nodeToIncidentEdges.get(node);
198    checkArgument(incidentEdges != null, NODE_NOT_IN_GRAPH, node);
199    return incidentEdges.outEdges();
200  }
201
202  @Override
203  public Set<N> predecessors(Object node) {
204    ImmutableSet.Builder<N> predecessorsBuilder = ImmutableSet.builder();
205    for (E edge : inEdges(node)) {
206      predecessorsBuilder.add(source(edge));
207    }
208    return predecessorsBuilder.build();
209  }
210
211  @Override
212  public Set<N> successors(Object node) {
213    ImmutableSet.Builder<N> successorsBuilder = ImmutableSet.builder();
214    for (E edge : outEdges(node)) {
215      successorsBuilder.add(target(edge));
216    }
217    return successorsBuilder.build();
218  }
219
220  @Override
221  public long degree(Object node) {
222    return inDegree(node) + outDegree(node) - edgesConnecting(node, node).size();
223  }
224
225  @Override
226  public long inDegree(Object node) {
227    return inEdges(node).size();
228  }
229
230  @Override
231  public long outDegree(Object node) {
232    return outEdges(node).size();
233  }
234
235  @Override
236  public N source(Object edge) {
237    checkNotNull(edge, "edge");
238    IncidentNodes<N> endpoints = edgeToIncidentNodes.get(edge);
239    checkArgument(endpoints != null, EDGE_NOT_IN_GRAPH, edge);
240    return endpoints.source();
241  }
242
243  @Override
244  public N target(Object edge) {
245    checkNotNull(edge, "edge");
246    IncidentNodes<N> endpoints = edgeToIncidentNodes.get(edge);
247    checkArgument(endpoints != null, EDGE_NOT_IN_GRAPH, edge);
248    return endpoints.target();
249  }
250
251  @Override
252  public boolean equals(@Nullable Object object) {
253    return (object instanceof DirectedGraph) && Graphs.equal(this, (DirectedGraph) object);
254  }
255
256  @Override
257  public int hashCode() {
258    // This map encapsulates all of the structural relationships of this graph, so its hash code
259    // is consistent with the above definition of equals().
260    return nodeToIncidentEdges.hashCode();
261  }
262
263  /**
264   * Returns a new builder. The generated builder is equivalent to the builder
265   * created by the {@code Builder} constructor.
266   */
267  @CheckReturnValue
268  public static <N, E> Builder<N, E> builder() {
269    return new Builder<N, E>();
270  }
271
272  /**
273   * Returns a new builder. The generated builder is equivalent to the builder
274   * created by the {@code Builder} constructor.
275   *
276   * @param config an instance of {@code GraphConfig} with the intended
277   *        graph configuration.
278   */
279  @CheckReturnValue
280  public static <N, E> Builder<N, E> builder(GraphConfig config) {
281    return new Builder<N, E>(config);
282  }
283
284  /**
285   * Returns an immutable copy of the input graph.
286   */
287  @CheckReturnValue
288  public static <N, E> ImmutableDirectedGraph<N, E> copyOf(DirectedGraph<N, E> graph) {
289    return new Builder<N, E>(graph).build();
290  }
291
292  @Override
293  public String toString() {
294    return String.format("config: %s, nodes: %s, edges: %s",
295        config,
296        nodeToIncidentEdges.keySet(),
297        edgeToIncidentNodes);
298  }

300  /**
301   * A builder for creating immutable directed graph instances.
302   *
303   * @param <N> Node parameter type
304   * @param <E> edge parameter type
305   * @see GraphConfig
306   */
307  public static final class Builder<N, E> implements AbstractImmutableGraph.Builder<N, E> {
308
309    private final DirectedGraph<N, E> directedGraph;
310
311    /**
312     * Creates a new builder with the default graph configuration.
313     */
314    public Builder() {
315      this(Graphs.<N, E>createDirected());
316    }
317
318    /**
319     * Creates a new builder with the specified graph configuration.
320     */
321    public Builder(GraphConfig config) {
322      this(Graphs.<N, E>createDirected(config));
323    }
324
325    /**
326     * Creates a new builder whose internal state is that of {@code graph}.
327     *
328     * <p>NOTE: This constructor should only be used in the case where it will be immediately
329     * followed by a call to {@code build}, to ensure that the input graph will not be modified.
330     * Currently the only such context is {@code Immutable*Graph.copyOf()}, which use these
331     * constructors to avoid making an extra copy of the graph state.
332     * @see ImmutableDirectedGraph#copyOf(DirectedGraph)
333     */
334    private Builder(DirectedGraph<N, E> graph) {
335      this.directedGraph = checkNotNull(graph, "graph");
336    }
337
338    @Override
339    public Builder<N, E> addNode(N node) {
340      directedGraph.addNode(node);
341      return this;
342    }
343
344    @Override
345    public Builder<N, E> addEdge(E edge, N node1, N node2) {
346      directedGraph.addEdge(edge, node1, node2);
347      return this;
348    }
349
350    /**
351     * Adds all elements of {@code graph} to the graph being built.
352     *
353     * @throws IllegalArgumentException under either of two conditions:
354     *     (1) the {@code GraphConfig} objects held by the graph being built and by {@code graph}
355     *     are not compatible
356     *     (2) calling {@code Graph.addEdge(e, n1, n2)} on the graph being built throws IAE
357     * @see Graph#addEdge(e, n1, n2)
358     */
359    public Builder<N, E> addGraph(DirectedGraph<N, E> graph) {
360      checkArgument(
361          directedGraph.config().compatibleWith(graph.config()),
362          "GraphConfigs for input and for graph being built are not compatible: input: %s, "
363              + "this graph: %s",
364          graph.config(),
365          directedGraph.config());
366
367      for (N node : graph.nodes()) {
368        directedGraph.addNode(node);
369      }
370      for (E edge : graph.edges()) {
371        directedGraph.addEdge(edge, graph.source(edge), graph.target(edge));
372      }
373
374      return this;
375    }
376
377    @Override
378    @CheckReturnValue
379    public ImmutableDirectedGraph<N, E> build() {
380      return new ImmutableDirectedGraph<N, E>(this);
381    }
382  }
