077  // A note on memory visibility.
078  // Many of the utilities in this class (transform, withFallback, withTimeout, asList, combine)
079  // have two requirements that significantly complicate their design.
080  // 1. Cancellation should propagate from the returned future to the input future(s).
081  // 2. The returned futures shouldn't unnecessarily 'pin' their inputs after completion.
082  //
083  // A consequence of these requirements is that the delegate futures cannot be stored in
084  // final fields.
085  //
086  // For simplicity the rest of this description will discuss Futures.catching since it is the
087  // simplest instance, though very similar descriptions apply to many other classes in this file.
088  //
089  // In the constructor of AbstractCatchingFuture, the delegate future is assigned to a field
090  // 'inputFuture'. That field is non-final and non-volatile.  There are 2 places where the
091  // 'inputFuture' field is read and where we will have to consider visibility of the write
092  // operation in the constructor.
093  //
094  // 1. In the listener that performs the callback.  In this case it is fine since inputFuture is
095  //    assigned prior to calling addListener, and addListener happens-before any invocation of the
096  //    listener. Notably, this means that 'volatile' is unnecessary to make 'inputFuture' visible
097  //    to the listener.
098  //
099  // 2. In done() where we may propagate cancellation to the input.  In this case it is _not_ fine.
100  //    There is currently nothing that enforces that the write to inputFuture in the constructor is
101  //    visible to done().  This is because there is no happens before edge between the write and a
102  //    (hypothetical) unsafe read by our caller. Note: adding 'volatile' does not fix this issue,
103  //    it would just add an edge such that if done() observed non-null, then it would also
104  //    definitely observe all earlier writes, but we still have no guarantee that done() would see
105  //    the inital write (just stronger guarantees if it does).
106  //
107  // See: http://cs.oswego.edu/pipermail/concurrency-interest/2015-January/013800.html
108  // For a (long) discussion about this specific issue and the general futility of life.
109  //
110  // For the time being we are OK with the problem discussed above since it requires a caller to
111  // introduce a very specific kind of data-race.  And given the other operations performed by these
112  // methods that involve volatile read/write operations, in practice there is no issue.  Also, the
113  // way in such a visibility issue would surface is most likely as a failure of cancel() to
114  // propagate to the input.  Cancellation propagation is fundamentally racy so this is fine.
115  //
116  // Future versions of the JMM may revise safe construction semantics in such a way that we can
117  // safely publish these objects and we won't need this whole discussion.
118  // TODO(user,lukes): consider adding volatile to all these fields since in current known JVMs
119  // that should resolve the issue.  This comes at the cost of adding more write barriers to the
120  // implementations.

143  private abstract static class ImmediateFuture<V>
144      implements ListenableFuture<V> {
145
146    private static final Logger log =
147        Logger.getLogger(ImmediateFuture.class.getName());
148
149    @Override
150    public void addListener(Runnable listener, Executor executor) {
151      checkNotNull(listener, "Runnable was null.");
152      checkNotNull(executor, "Executor was null.");
153      try {
154        executor.execute(listener);
155      } catch (RuntimeException e) {
156        // ListenableFuture's contract is that it will not throw unchecked
157        // exceptions, so log the bad runnable and/or executor and swallow it.
158        log.log(Level.SEVERE, "RuntimeException while executing runnable "
159            + listener + " with executor " + executor, e);
160      }
161    }
162
163    @Override
164    public boolean cancel(boolean mayInterruptIfRunning) {
165      return false;
166    }
167
168    @Override
169    public abstract V get() throws ExecutionException;
170
171    @Override
172    public V get(long timeout, TimeUnit unit) throws ExecutionException {
173      checkNotNull(unit);
174      return get();
175    }
176
177    @Override
178    public boolean isCancelled() {
179      return false;
180    }
181
182    @Override
183    public boolean isDone() {
184      return true;
185    }
186  }

371  /**
372   * Returns a {@code Future} whose result is taken from the given primary
373   * {@code input} or, if the primary input fails, from the {@code Future}
374   * provided by the {@code fallback}. {@link FutureFallback#create} is not
375   * invoked until the primary input has failed, so if the primary input
376   * succeeds, it is never invoked. If, during the invocation of {@code
377   * fallback}, an exception is thrown, this exception is used as the result of
378   * the output {@code Future}.
379   *
380   * <p>Below is an example of a fallback that returns a default value if an
381   * exception occurs:
382   *
383   * <pre>   {@code
384   *   ListenableFuture<Integer> fetchCounterFuture = ...;
385   *
386   *   // Falling back to a zero counter in case an exception happens when
387   *   // processing the RPC to fetch counters.
388   *   ListenableFuture<Integer> faultTolerantFuture = Futures.withFallback(
389   *       fetchCounterFuture, new FutureFallback<Integer>() {
390   *         public ListenableFuture<Integer> create(Throwable t) {
391   *           // Returning "0" as the default for the counter when the
392   *           // exception happens.
393   *           return immediateFuture(0);
394   *         }
395   *       });}</pre>
396   *
397   * <p>The fallback can also choose to propagate the original exception when
398   * desired:
399   *
400   * <pre>   {@code
401   *   ListenableFuture<Integer> fetchCounterFuture = ...;
402   *
403   *   // Falling back to a zero counter only in case the exception was a
404   *   // TimeoutException.
405   *   ListenableFuture<Integer> faultTolerantFuture = Futures.withFallback(
406   *       fetchCounterFuture, new FutureFallback<Integer>() {
407   *         public ListenableFuture<Integer> create(Throwable t) {
408   *           if (t instanceof TimeoutException) {
409   *             return immediateFuture(0);
410   *           }
411   *           return immediateFailedFuture(t);
412   *         }
413   *       });}</pre>
414   *
415   * <p>This overload, which does not accept an executor, uses {@code
416   * directExecutor}, a dangerous choice in some cases. See the discussion in
417   * the {@link ListenableFuture#addListener ListenableFuture.addListener}
418   * documentation. The documentation's warnings about "lightweight listeners"
419   * refer here to the work done during {@code FutureFallback.create}, not to
420   * any work done to complete the returned {@code Future}.
421   *
422   * @param input the primary input {@code Future}
423   * @param fallback the {@link FutureFallback} implementation to be called if
424   *     {@code input} fails
425   * @since 14.0
426   * @deprecated Use {@link #catchingAsync(ListenableFuture, Class,
427   *     AsyncFunction) catchingAsync(input, Throwable.class,
428   *     fallbackImplementedAsAnAsyncFunction)}, usually replacing {@code
429   *     Throwable.class} with the specific type you want to handle. This method
430   *     will be removed in Guava release 20.0.

752  private abstract static class AbstractCatchingFuture<V, X extends Throwable, F>
753      extends AbstractFuture.TrustedFuture<V> implements Runnable {
754    @Nullable ListenableFuture<? extends V> inputFuture;
755    @Nullable Class<X> exceptionType;
756    @Nullable F fallback;
757
758    AbstractCatchingFuture(
759        ListenableFuture<? extends V> inputFuture, Class<X> exceptionType, F fallback) {
760      this.inputFuture = checkNotNull(inputFuture);
761      this.exceptionType = checkNotNull(exceptionType);
762      this.fallback = checkNotNull(fallback);
763    }
764
765    @Override public final void run() {
766      ListenableFuture<? extends V> localInputFuture = inputFuture;
767      Class<X> localExceptionType = exceptionType;
768      F localFallback = fallback;
769      if (localInputFuture == null | localExceptionType == null | localFallback == null
770          | isCancelled()) {
771        return;
772      }
773      inputFuture = null;
774      exceptionType = null;
775      fallback = null;
776
777      Throwable throwable;
778      try {
779        set(getUninterruptibly(localInputFuture));
780        return;
781      } catch (ExecutionException e) {
782        throwable = e.getCause();
783      } catch (Throwable e) {  // this includes cancellation exception
784        throwable = e;
785      }
786      try {
787        if (isInstanceOfThrowableClass(throwable, localExceptionType)) {
788          @SuppressWarnings("unchecked") // verified safe by isInstance
789          X castThrowable = (X) throwable;
790          doFallback(localFallback, castThrowable);
791        } else {
792          setException(throwable);
793        }
794      } catch (Throwable e) {
795        setException(e);
796      }
797    }
798
799    /** Template method for subtypes to actually run the fallback. */
800    abstract void doFallback(F fallback, X throwable) throws Exception;
801
802    @Override final void done() {
803      maybePropagateCancellation(inputFuture);
804      this.inputFuture = null;
805      this.exceptionType = null;
806      this.fallback = null;
807    }
808  }

873  /**
874   * Future that delegates to another but will finish early (via a {@link
875   * TimeoutException} wrapped in an {@link ExecutionException}) if the
876   * specified duration expires.
877   * The delegate future is interrupted and cancelled if it times out.
878   */
879  private static final class TimeoutFuture<V> extends AbstractFuture.TrustedFuture<V> {
880    // Memory visibility of these fields.
881    // There are two cases to consider.
882    // 1. visibility of the writes to these fields to Fire.run
883    //    The initial write to delegateRef is made definitely visible via the semantics of
884    //    addListener/SES.schedule.  The later racy write in cancel() is not guaranteed to be
885    //    observed, however that is fine since the correctness is based on the atomic state in
886    //    our base class.
887    //    The initial write to timer is never definitely visible to Fire.run since it is assigned
888    //    after SES.schedule is called. Therefore Fire.run has to check for null.  However, it
889    //    should be visible if Fire.run is called by delegate.addListener since addListener is
890    //    called after the assignment to timer, and importantly this is the main situation in which
891    //    we need to be able to see the write.
892    // 2. visibility of the writes to cancel
893    //    Since these fields are non-final that means that TimeoutFuture is not being 'safely
894    //    published', thus a motivated caller may be able to expose the reference to another thread
895    //    that would then call cancel() and be unable to cancel the delegate.
896    //    There are a number of ways to solve this, none of which are very pretty, and it is
897    //    currently believed to be a purely theoretical problem (since the other actions should
898    //    supply sufficient write-barriers).
899
900    @Nullable ListenableFuture<V> delegateRef;
901    @Nullable Future<?> timer;
902
903    TimeoutFuture(ListenableFuture<V> delegate) {
904      this.delegateRef = Preconditions.checkNotNull(delegate);
905    }
906
907    /** A runnable that is called when the delegate or the timer completes. */
908    private static final class Fire<V> implements Runnable {
909      @Nullable TimeoutFuture<V> timeoutFutureRef;
910
911      Fire(TimeoutFuture<V> timeoutFuture) {
912        this.timeoutFutureRef = timeoutFuture;
913      }
914
915      @Override public void run() {
916        // If either of these reads return null then we must be after a successful cancel
917        // or another call to this method.
918        TimeoutFuture<V> timeoutFuture = timeoutFutureRef;
919        if (timeoutFuture == null) {
920          return;
921        }
922        ListenableFuture<V> delegate = timeoutFuture.delegateRef;
923        if (delegate == null) {
924          return;
925        }
926
927        /*
928         * If we're about to complete the TimeoutFuture, we want to release our reference to it.
929         * Otherwise, we'll pin it (and its result) in memory until the timeout task is GCed. (The
930         * need to clear our reference to the TimeoutFuture is the reason we use a *static* nested
931         * class with a manual reference back to the "containing" class.)
932         *
933         * This has the nice-ish side effect of limiting reentrancy: run() calls
934         * timeoutFuture.setException() calls run(). That reentrancy would already be harmless,
935         * since timeoutFuture can be set (and delegate cancelled) only once. (And "set only once"
936         * is important for other reasons: run() can still be invoked concurrently in different
937         * threads, even with the above null checks.)
938         */
939        timeoutFutureRef = null;
940        if (delegate.isDone()) {
941          timeoutFuture.setFuture(delegate);
942        } else {
943          try {
944            // TODO(lukes): this stack trace is particularly useless (all it does is point at the
945            // scheduledexecutorservice thread), consider eliminating it altogether?
946            timeoutFuture.setException(new TimeoutException("Future timed out: " + delegate));
947          } finally {
948            delegate.cancel(true);
949          }
950        }
951      }
952    }
953
954    @Override void done() {
955      maybePropagateCancellation(delegateRef);
956
957      Future<?> localTimer = timer;
958      // Try to cancel the timer as an optimization
959      // timer may be null if this call to run was by the timer task since there is no
960      // happens-before edge between the assignment to timer and an execution of the timer task.
961      if (localTimer != null) {
962        localTimer.cancel(false);
963      }
964
965      delegateRef = null;
966      timer = null;
967    }
968  }

1158  /**
1159   * Returns an Executor that will propagate {@link RejectedExecutionException} from the delegate
1160   * executor to the given {@code future}.
1161   *
1162   * <p>Note, the returned executor can only be used once.
1163   */
1164  private static Executor rejectionPropagatingExecutor(
1165      final Executor delegate, final AbstractFuture<?> future) {
1166    checkNotNull(delegate);
1167    if (delegate == directExecutor()) {
1168      // directExecutor() cannot throw RejectedExecutionException
1169      return delegate;
1170    }
1171    return new Executor() {
1172      volatile boolean thrownFromDelegate = true;
1173      @Override public void execute(final Runnable command) {
1174        try {
1175          delegate.execute(new Runnable() {
1176            @Override public void run() {
1177              thrownFromDelegate = false;
1178              command.run();
1179            }
1180          });
1181        } catch (RejectedExecutionException e) {
1182          if (thrownFromDelegate) {
1183            // wrap exception?
1184            future.setException(e);
1185          }

1352  /**
1353   * An implementation of {@code ListenableFuture} that also implements
1354   * {@code Runnable} so that it can be used to nest ListenableFutures.
1355   * Once the passed-in {@code ListenableFuture} is complete, it calls the
1356   * passed-in {@code Function} to generate the result.
1357   *
1358   * <p>For historical reasons, this class has a special case in its exception
1359   * handling: If the given {@code AsyncFunction} throws an {@code
1360   * UndeclaredThrowableException}, {@code ChainingListenableFuture} unwraps it
1361   * and uses its <i>cause</i> as the output future's exception, rather than
1362   * using the {@code UndeclaredThrowableException} itself as it would for other
1363   * exception types. The reason for this is that {@code Futures.transform} used
1364   * to require a {@code Function}, whose {@code apply} method is not allowed to
1365   * throw checked exceptions. Nowadays, {@code Futures.transform} has an
1366   * overload that accepts an {@code AsyncFunction}, whose {@code apply} method
1367   * <i>is</i> allowed to throw checked exception. Users who wish to throw
1368   * checked exceptions should use that overload instead, and <a
1369   * href="http://code.google.com/p/guava-libraries/issues/detail?id=1548">we
1370   * should remove the {@code UndeclaredThrowableException} special case</a>.
1371   */
1372  private abstract static class AbstractChainingFuture<I, O, F>
1373      extends AbstractFuture.TrustedFuture<O> implements Runnable {
1374    // In theory, this field might not be visible to a cancel() call in certain circumstances. For
1375    // details, see the comments on the fields of TimeoutFuture.
1376    @Nullable ListenableFuture<? extends I> inputFuture;
1377    @Nullable F function;
1378
1379    AbstractChainingFuture(ListenableFuture<? extends I> inputFuture, F function) {
1380      this.inputFuture = checkNotNull(inputFuture);
1381      this.function = checkNotNull(function);
1382    }
1383
1384    @Override
1385    public final void run() {
1386      try {
1387        ListenableFuture<? extends I> localInputFuture = inputFuture;
1388        F localFunction = function;
1389        if (isCancelled() | localInputFuture == null | localFunction == null) {
1390          return;
1391        }
1392        inputFuture = null;
1393        function = null;
1394
1395        I sourceResult;
1396        try {
1397          sourceResult = getUninterruptibly(localInputFuture);
1398        } catch (CancellationException e) {
1399          // Cancel this future and return.
1400          // At this point, inputFuture is cancelled and outputFuture doesn't
1401          // exist, so the value of mayInterruptIfRunning is irrelevant.
1402          cancel(false);
1403          return;
1404        } catch (ExecutionException e) {
1405          // Set the cause of the exception as this future's exception
1406          setException(e.getCause());
1407          return;
1408        }
1409        doTransform(localFunction, sourceResult);
1410      } catch (UndeclaredThrowableException e) {
1411        // Set the cause of the exception as this future's exception
1412        setException(e.getCause());
1413      } catch (Throwable t) {
1414        // This exception is irrelevant in this thread, but useful for the
1415        // client
1416        setException(t);
1417      }
1418    }
1419
1420    /** Template method for subtypes to actually run the transform. */
1421    abstract void doTransform(F function, I result) throws Exception;
1422
1423    @Override final void done() {
1424      maybePropagateCancellation(inputFuture);
1425      this.inputFuture = null;
1426      this.function = null;
1427    }
1428  }
