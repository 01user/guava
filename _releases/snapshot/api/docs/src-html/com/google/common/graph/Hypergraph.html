017package com.google.common.graph;
018
019import com.google.common.annotations.Beta;
020import com.google.errorprone.annotations.CanIgnoreReturnValue;
021
022/**
023 * A subinterface of {@code Graph} which specifies that all edges are hyperedges, that is,
024 * they connect arbitrary sets of nodes rather than pairs of nodes.
025 *
026 * <p>A few notes about how hyperedges and connectivity:
027 * <ul>
028 * <li>Hyperedges, like undirected edges, are both incoming and outgoing edges.
029 * <li>Hyperedges incident to a single node {@code node} connect {@code node} to itself; such edges
030 *     are analogous to self-loops in graphs.  Hyperedges incident to > 1 nodes do not connect any
031 *     of their incident nodes to themselves.
032 * </ul>
033 *
034 * @author Joshua O'Madadhain
035 * @param <N> Node parameter type
036 * @param <E> Edge parameter type
037 * @since 20.0
038 */
039@Beta
040public interface Hypergraph<N, E> extends Graph<N, E> {
041  /**
042   * Adds {@code edge} to this graph, connecting {@code nodes}.
043   *
044   * @return {@code true} iff the graph was modified as a result of this call
045   * @throws UnsupportedOperationException if the add operation is not supported by this graph
046   */
047  @CanIgnoreReturnValue
048  boolean addEdge(E edge, N... nodes);
049
050  /**
051   * Adds {@code edge} to this graph, connecting {@code nodes}.
052   *
053   * @return {@code true} iff the graph was modified as a result of this call
054   * @throws UnsupportedOperationException if the add operation is not supported by this graph
055   */
056  @CanIgnoreReturnValue
057  boolean addEdge(E edge, Iterable<N> nodes);
058}
