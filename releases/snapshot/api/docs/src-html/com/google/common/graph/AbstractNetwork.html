017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkArgument;
020import static com.google.common.graph.GraphConstants.EDGE_CONNECTING_NOT_IN_GRAPH;
021import static com.google.common.graph.GraphConstants.GRAPH_STRING_FORMAT;
022
023import com.google.common.annotations.Beta;
024import com.google.common.base.Function;
025import com.google.common.collect.ImmutableSet;
026import com.google.common.collect.Iterators;
027import com.google.common.collect.Maps;
028import com.google.common.collect.Sets;
029import com.google.common.math.IntMath;
030import java.util.AbstractSet;
031import java.util.Iterator;
032import java.util.Map;
033import java.util.Set;
034import javax.annotation.Nullable;
035
036/**
037 * This class provides a skeletal implementation of {@link Network}. It is recommended to extend
038 * this class rather than implement {@link Network} directly, to ensure consistent {@link
039 * #equals(Object)} and {@link #hashCode()} results across different network implementations.
040 *
041 * @author James Sexton
042 * @param <N> Node parameter type
043 * @param <E> Edge parameter type
044 * @since 20.0
045 */
046@Beta
047public abstract class AbstractNetwork<N, E> implements Network<N, E> {
048
049  @Override
050  public Graph<N, Set<E>> asGraph() {
051    return new AbstractGraph<N, Set<E>>() {
052      @Override
053      public Set<N> nodes() {
054        return AbstractNetwork.this.nodes();
055      }
056
057      @Override
058      public Set<Endpoints<N>> edges() {
059        if (allowsParallelEdges()) {
060          return super.edges(); // Defer to AbstractGraph implementation.
061        }
062
063        // Optimized implementation assumes no parallel edges (1:1 edge to Endpoints mapping).
064        return new AbstractSet<Endpoints<N>>() {
065          @Override
066          public Iterator<Endpoints<N>> iterator() {
067            return Iterators.transform(
068                AbstractNetwork.this.edges().iterator(),
069                new Function<E, Endpoints<N>>() {
070                  @Override
071                  public Endpoints<N> apply(E edge) {
072                    return incidentNodes(edge);
073                  }
074                });
075          }
076
077          @Override
078          public int size() {
079            return AbstractNetwork.this.edges().size();
080          }
081
082          @Override
083          public boolean contains(Object obj) {
084            if (!(obj instanceof Endpoints)) {
085              return false;
086            }
087            Endpoints<?> endpoints = (Endpoints<?>) obj;
088            return isDirected() == endpoints.isDirected()
089                && !edgesConnecting(endpoints.nodeA(), endpoints.nodeB()).isEmpty();
090          }
091        };
092      }
093
094      @Override
095      public ElementOrder<N> nodeOrder() {
096        return AbstractNetwork.this.nodeOrder();
097      }
098
099      @Override
100      public boolean isDirected() {
101        return AbstractNetwork.this.isDirected();
102      }
103
104      @Override
105      public boolean allowsSelfLoops() {
106        return AbstractNetwork.this.allowsSelfLoops();
107      }
108
109      @Override
110      public Set<N> adjacentNodes(Object node) {
111        return AbstractNetwork.this.adjacentNodes(node);
112      }
113
114      @Override
115      public Set<N> predecessors(Object node) {
116        return AbstractNetwork.this.predecessors(node);
117      }
118
119      @Override
120      public Set<N> successors(Object node) {
121        return AbstractNetwork.this.successors(node);
122      }
123
124      @Override
125      public Set<E> edgeValue(Object nodeA, Object nodeB) {
126        Set<E> edges = edgesConnecting(nodeA, nodeB);
127        checkArgument(!edges.isEmpty(), EDGE_CONNECTING_NOT_IN_GRAPH, nodeA, nodeB);
128        return edges;
129      }
130
131      @Override
132      public Set<E> edgeValueOrDefault(Object nodeA, Object nodeB, Set<E> defaultValue) {
133        Set<E> edges = edgesConnecting(nodeA, nodeB);
134        return edges.isEmpty() ? defaultValue : edges;
135      }
136    };
137  }
138
139  @Override
140  public int degree(Object node) {
141    if (isDirected()) {
142      return IntMath.saturatedAdd(inEdges(node).size(), outEdges(node).size());
143    } else {
144      return IntMath.saturatedAdd(incidentEdges(node).size(), edgesConnecting(node, node).size());
145    }
146  }
147
148  @Override
149  public int inDegree(Object node) {
150    return isDirected() ? inEdges(node).size() : degree(node);
151  }
152
153  @Override
154  public int outDegree(Object node) {
155    return isDirected() ? outEdges(node).size() : degree(node);
156  }
157
158  @Override
159  public Set<E> adjacentEdges(Object edge) {
160    Endpoints<?> endpoints = incidentNodes(edge); // Verifies that edge is in this network.
161    Set<E> endpointsIncidentEdges =
162        Sets.union(incidentEdges(endpoints.nodeA()), incidentEdges(endpoints.nodeB()));
163    return Sets.difference(endpointsIncidentEdges, ImmutableSet.of(edge));
164  }
165
166  @Override
167  public final boolean equals(@Nullable Object obj) {
168    if (obj == this) {
169      return true;
170    }
171    if (!(obj instanceof Network)) {
172      return false;
173    }
174    Network<?, ?> other = (Network<?, ?>) obj;
175
176    if (isDirected() != other.isDirected()
177        || !nodes().equals(other.nodes())
178        || !edges().equals(other.edges())) {
179      return false;
180    }
181
182    for (E edge : edges()) {
183      if (!incidentNodes(edge).equals(other.incidentNodes(edge))) {
184        return false;
185      }
186    }
187
188    return true;
189  }
190
191  @Override
192  public final int hashCode() {
193    return edgeEndpointsMap().hashCode();
194  }
195
196  /**
197   * Returns a string representation of this network.
198   */
199  @Override
200  public String toString() {
201    String propertiesString = String.format(
202        "isDirected: %s, allowsParallelEdges: %s, allowsSelfLoops: %s",
203        isDirected(), allowsParallelEdges(), allowsSelfLoops());
204    return String.format(GRAPH_STRING_FORMAT,
205        propertiesString,
206        nodes(),
207        edgeEndpointsMap());
208  }
209
210  private Map<E, Endpoints<N>> edgeEndpointsMap() {
211    Function<E, Endpoints<N>> edgeToEndpointsFn = new Function<E, Endpoints<N>>() {
212      @Override
213      public Endpoints<N> apply(E edge) {
214        return incidentNodes(edge);
215      }
216    };
217    return Maps.asMap(edges(), edgeToEndpointsFn);
218  }
219}
