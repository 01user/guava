017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkNotNull;
020import static com.google.common.graph.GraphConstants.NOT_AVAILABLE_ON_UNDIRECTED;
021
022import com.google.common.annotations.Beta;
023import com.google.common.base.Objects;
024import com.google.common.collect.UnmodifiableIterator;
025
026import java.util.AbstractCollection;
027import java.util.Collection;
028import java.util.NoSuchElementException;
029
030/**
031 * An immutable {@link Collection} to represent the two (possibly equal, in the case of a self-loop)
032 * endpoints of an edge in a graph.
033 *
034 * <p>The {@link Endpoints} of a directed edge are an ordered pair of nodes (source and target).
035 * The {@link Endpoints} of an undirected edge are an unordered pair of nodes. The nodes can be
036 * accessed through the {@link #iterator()}, and in the directed case, will iterate in the order
037 * {@link #source()}, {@link #target()}.
038 *
039 * @author James Sexton
040 * @since 20.0
041 */
042@Beta
043public abstract class Endpoints<N> extends AbstractCollection<N> {
044  private final N nodeA;
045  private final N nodeB;
046
047  private Endpoints(N nodeA, N nodeB) {
048    this.nodeA = checkNotNull(nodeA);
049    this.nodeB = checkNotNull(nodeB);
050  }
051
052  /**
053   * Returns {@link Endpoints} representing the endpoints of an edge in {@code graph}.
054   */
055  public static <N> Endpoints<N> of(Graph<?> graph, N nodeA, N nodeB) {
056    return graph.isDirected() ? ofDirected(nodeA, nodeB) : ofUndirected(nodeA, nodeB);
057  }
058
059  /**
060   * Returns {@link Endpoints} representing the endpoints of a directed edge.
061   */
062  static <N> Endpoints.Directed<N> ofDirected(N source, N target) {
063    return new Directed<N>(source, target);
064  }
065
066  /**
067   * Returns {@link Endpoints} representing the endpoints of an undirected edge.
068   */
069  static <N> Endpoints.Undirected<N> ofUndirected(N nodeA, N nodeB) {
070    return new Undirected<N>(nodeA, nodeB);
071  }
072
073  /**
074   * If these are the {@link Endpoints} of a directed edge, returns the node which is the source of
075   * that edge.
076   *
077   * @throws UnsupportedOperationException if these are the {@link Endpoints} of a undirected edge
078   */
079  public abstract N source();
080
081  /**
082   * If these are the {@link Endpoints} of a directed edge, returns the node which is the target of
083   * that edge.
084   *
085   * @throws UnsupportedOperationException if these are the {@link Endpoints} of a undirected edge
086   */
087  public abstract N target();
088
089  /**
090   * If these are the {@link Endpoints} of a directed edge, returns the {@link #source()};
091   * otherwise, returns an arbitrary (but consistent) endpoint of the origin edge.
092   */
093  final N nodeA() {
094    return nodeA;
095  }
096
097  /**
098   * Returns the node that is adjacent to {@link #nodeA()} via the origin edge.
099   * If these are the {@link Endpoints} of a directed edge, it is equal to the {@link #target()}.
100   */
101  final N nodeB() {
102    return nodeB;
103  }
104
105  /**
106   * Returns the node that is adjacent to {@code node} via the origin edge.
107   *
108   * @throws IllegalArgumentException if the origin edge is not incident to {@code node}
109   */
110  public final N otherNode(Object node) {
111    checkNotNull(node, "node");
112    if (node.equals(nodeA())) {
113      return nodeB();
114    } else if (node.equals(nodeB())) {
115      return nodeA();
116    } else {
117      throw new IllegalArgumentException(
118          String.format("Endpoints %s does not contain node %s", this, node));
119    }
120  }
121
122  @Override
123  public final UnmodifiableIterator<N> iterator() {
124    return new UnmodifiableIterator<N>() {
125      private int pos = 0;
126
127      @Override
128      public boolean hasNext() {
129        return pos < 2;
130      }
131
132      @Override
133      public N next() {
134        switch (pos++) {
135          case 0:
136            return nodeA;
137          case 1:
138            return nodeB;
139          default:
140            pos = 2;
141            throw new NoSuchElementException();
142        }
143      }
144    };
145  }
146
147  @Override
148  public final int size() {
149    return 2;
150  }
151
152  @Override
153  public final boolean contains(Object obj) {
154    return nodeA.equals(obj) || nodeB.equals(obj);
155  }
156
157  /**
158   * The {@link Endpoints} of two directed edges are equal if their {@link #source()} and
159   * {@link #target()} are equal. The {@link Endpoints} of two undirected edges are equal if they
160   * contain the same nodes. The {@link Endpoints} of a directed edge are never equal to the
161   * {@link Endpoints} of an undirected edge.
162   */
163  @Override
164  public abstract boolean equals(Object obj);
165
166  @Override
167  public abstract int hashCode();
168
169  /**
170   * The {@link Endpoints} of a directed edge. It is guaranteed that all {@link Endpoints} of
171   * directed edges will be an instance of this class.
172   */
173  static final class Directed<N> extends Endpoints<N> {
174    private Directed(N source, N target) {
175      super(source, target);
176    }
177
178    @Override
179    public N source() {
180      return nodeA();
181    }
182
183    @Override
184    public N target() {
185      return nodeB();
186    }
187
188    @Override
189    public boolean equals(Object obj) {
190      if (obj == this) {
191        return true;
192      }
193      if (!(obj instanceof Directed)) {
194        return false;
195      }
196
197      Directed<?> other = (Directed<?>) obj;
198      return source().equals(other.source()) && target().equals(other.target());
199    }
200
201    @Override
202    public int hashCode() {
203      return Objects.hashCode(source(), target());
204    }
205
206    @Override
207    public String toString() {
208      return String.format("<%s -> %s>", source(), target());
209    }
210  }
211
212  /**
213   * The {@link Endpoints} of an undirected edge. It is guaranteed that all {@link Endpoints} of
214   * undirected edges will be an instance of this class.
215   */
216  static final class Undirected<N> extends Endpoints<N> {
217    private Undirected(N nodeA, N nodeB) {
218      super(nodeA, nodeB);
219    }
220
221    @Override
222    public N source() {
223      throw new UnsupportedOperationException(NOT_AVAILABLE_ON_UNDIRECTED);
224    }
225
226    @Override
227    public N target() {
228      throw new UnsupportedOperationException(NOT_AVAILABLE_ON_UNDIRECTED);
229    }
230
231    @Override
232    public boolean equals(Object obj) {
233      if (obj == this) {
234        return true;
235      }
236      if (!(obj instanceof Undirected)) {
237        return false;
238      }
239
240      Undirected<?> other = (Undirected<?>) obj;
241      // Equivalent to the following simple implementation:
242      // boolean condition1 = nodeA().equals(other.nodeA()) && nodeB().equals(other.nodeB());
243      // boolean condition2 = nodeA().equals(other.nodeB()) && nodeB().equals(other.nodeA());
244      // return condition1 || condition2;
245      if (nodeA().equals(other.nodeA())) { // check condition1
246        // Here's the tricky bit. We don't have to explicitly check for condition2 in this case.
247        // Why? The second half of condition2 requires that nodeB equals other.nodeA.
248        // We already know that nodeA equals other.nodeA. Combined with the earlier statement,
249        // and the transitive property of equality, this implies that nodeA equals nodeB.
250        // If nodeA equals nodeB, condition1 == condition2, so checking condition1 is sufficient.
251        return nodeB().equals(other.nodeB());
252      }
253      return nodeA().equals(other.nodeB()) && nodeB().equals(other.nodeA()); // check condition2
254    }
255
256    @Override
257    public int hashCode() {
258      return nodeA().hashCode() ^ nodeB().hashCode();
259    }
260  }
