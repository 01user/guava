017package com.google.common.graph;
018
019import com.google.common.annotations.Beta;
020import com.google.errorprone.annotations.CanIgnoreReturnValue;
021
022/**
023 * A subtype of {@link Graph} which permits mutations.
024 * Users should generally use the {@link Graph} interface where possible.
025 *
026 * @author James Sexton
027 * @author Joshua O'Madadhain
028 * @param <N> Node parameter type
029 * @since 20.0
030 */
031@Beta
032public interface MutableGraph<N> extends Graph<N> {
033
034  /**
035   * Adds {@code node} to this graph (optional operation).
036   *
037   * <p><b>Nodes must be unique</b>, just as {@code Map} keys must be; they must also be non-null.
038   *
039   * @return {@code true} iff the graph was modified as a result of this call
040   * @throws UnsupportedOperationException if the add operation is not supported by this graph
041   */
042  @CanIgnoreReturnValue
043  boolean addNode(N node);
044
045  /**
046   * Adds an (implicit) edge to this graph connecting {@code nodeA} to {@code nodeB}
047   * (optional operation).
048   *
049   * <p>Behavior if {@code nodeA} and {@code nodeB} are not already elements of the graph is
050   * unspecified. Suggested behaviors include (a) silently adding {@code nodeA} and {@code nodeB}
051   * to the graph or (b) throwing {@code IllegalArgumentException}.
052   *
053   * <p>Currently, this type does not support parallel edges.  {@code addEdge(nodeA, nodeB)} will
054   * simply return false on any future calls with the same arguments (analogous to the behavior of
055   * {@code Network.addEdge(e, nodeA, nodeB)}).  A hypothetical instance that supported parallel
056   * edges would add a new edge between {@code nodeA} and {@code nodeB} for every call to
057   * {@code addEdge(nodeA, nodeB)}, and return {@code true} every time.
058   *
059   * @return {@code true} iff the graph was modified as a result of this call
060   * @throws UnsupportedOperationException if the add operation is not supported by this graph
061   */
062  @CanIgnoreReturnValue
063  boolean addEdge(N nodeA, N nodeB);
064
065  /**
066   * Removes {@code node} from this graph, if it is present (optional operation).
067   * All edges incident to {@code node} in this graph will also be removed.
068   *
069   * @return {@code true} iff the graph was modified as a result of this call
070   * @throws UnsupportedOperationException if the remove operation is not supported by this graph
071   */
072  @CanIgnoreReturnValue
073  boolean removeNode(Object node);
074
075  /**
076   * Removes an edge connecting {@code nodeA} to {@code nodeB} from this graph, if one is present
077   * (optional operation).
078   *
079   * <p>In general, the input nodes are unaffected (although implementations may choose
080   * to disallow certain configurations, e.g., isolated nodes).
081   *
082   * @return {@code true} iff the graph was modified as a result of this call
083   * @throws UnsupportedOperationException if the remove operation is not supported by this graph
084   */
085  @CanIgnoreReturnValue
086  boolean removeEdge(Object nodeA, Object nodeB);
087}
