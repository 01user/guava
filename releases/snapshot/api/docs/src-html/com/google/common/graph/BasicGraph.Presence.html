017package com.google.common.graph;
018
019import com.google.common.annotations.Beta;
020import java.util.ConcurrentModificationException;
021import java.util.Map;
022import java.util.Set;
023
024/**
025 * An interface for <a href="https://en.wikipedia.org/wiki/Graph_(discrete_mathematics)">graph</a>
026 * data structures. A graph is composed of a set of nodes (sometimes called vertices) and a set of
027 * edges connecting pairs of nodes. Graphs are useful for modeling many kinds of relations. If the
028 * relation to be modeled is symmetric (such as "distance between cities"), that can be represented
029 * with an undirected graph, where an edge that connects node A to node B also connects node B to
030 * node A. If the relation to be modeled is asymmetric (such as "employees managed"), that can be
031 * represented with a directed graph, where edges are strictly one-way.
032 *
033 * <p>There are three main interfaces provided to represent graphs. In order of increasing
034 * complexity they are: {@link BasicGraph}, {@link Graph}, and {@link Network}. You should generally
035 * prefer the simplest interface that satisfies your use case.
036 *
037 * <p>To choose the right interface, answer these questions:
038 *
039 * <ol>
040 * <li>Do you have data (objects) that you wish to associate with edges?
041 *     <p>Yes: Go to question 2. No: Use {@link BasicGraph}.
042 * <li>Are the objects you wish to associate with edges unique within the scope of a graph? That is,
043 *     no two objects would be {@link Object#equals(Object) equal} to each other. A common example
044 *     where this would <i>not</i> be the case is with weighted graphs.
045 *     <p>Yes: Go to question 3. No: Use {@link Graph}.
046 * <li>Do you need to be able to query the graph for an edge associated with a particular object?
047 *     For example, do you need to query what nodes an edge associated with a particular object
048 *     connects, or whether an edge associated with that object exists in the graph?
049 *     <p>Yes: Use {@link Network}. No: Go to question 4.
050 * <li>Do you need explicit support for parallel edges? For example, do you need to remove one edge
051 *     connecting a pair of nodes while leaving other edges connecting those same nodes intact?
052 *     <p>Yes: Use {@link Network}. No: Use {@link Graph}.
053 * </ol>
054 *
055 * <p>In all three interfaces, nodes have all the same requirements as keys in a {@link Map}.
056 *
057 * <p>All mutation methods live on the subinterface {@link MutableBasicGraph}. If you do not need to
058 * mutate a graph (e.g. if you write a method than runs a read-only algorithm on the graph), you
059 * should prefer the non-mutating {@link BasicGraph} interface.
060 *
061 * <p>The {@link BasicGraph} interface extends {@link Graph}. When storing references, it is
062 * preferable to store them as {@link BasicGraph}s so you do not have to worry about the value type.
063 * However, when writing methods that operate on graphs but do not care about edge values, it is
064 * preferable to accept {@code Graph<N, ?>} to allow the widest variety of valid input.
065 *
066 * <p>We provide an efficient implementation of this interface via {@link BasicGraphBuilder}. When
067 * using the implementation provided, all {@link Set}-returning methods provide live, unmodifiable
068 * views of the graph. In other words, you cannot add an element to the {@link Set}, but if an
069 * element is added to the {@link BasicGraph} that would affect the result of that set, it will be
070 * updated automatically. This also means that you cannot modify a {@link BasicGraph} in a way that
071 * would affect a {#link Set} while iterating over that set. For example, you cannot remove the
072 * nodes from a {@link BasicGraph} while iterating over {@link #nodes} (unless you first make a copy
073 * of the nodes), just as you could not remove the keys from a {@link Map} while iterating over its
074 * {@link Map#keySet()}. This will either throw a {@link ConcurrentModificationException} or risk
075 * undefined behavior.
076 *
077 * <p>Example of use:
078 *
079 * <pre><code>
080 * MutableBasicGraph<String> managementGraph = BasicGraphBuilder.directed().build();
081 * managementGraph.putEdge("Big Boss", "Middle Manager Jack");
082 * managementGraph.putEdge("Big Boss", "Middle Manager Jill");
083 * managementGraph.putEdge("Middle Manager Jack", "Joe");
084 * managementGraph.putEdge("Middle Manager Jack", "Schmoe");
085 * managementGraph.putEdge("Middle Manager Jill", "Jane");
086 * managementGraph.putEdge("Middle Manager Jill", "Doe");
087 * for (String employee : managementGraph.nodes()) {
088 *   Set<String> reports = managementGraph.successors(employee);
089 *   if (!reports.isEmpty()) {
090 *     System.out.format("%s has the following direct reports: %s%n", employee, reports);
091 *   }
092 * }
093 * </code></pre>
094 *
095 * @author James Sexton
096 * @author Joshua O'Madadhain
097 * @param <N> Node parameter type
098 * @since 20.0
099 */
100@Beta
101public interface BasicGraph<N> extends Graph<N, BasicGraph.Presence> {
102
103  /**
104   * A placeholder for the (generally ignored) value type of a {@link BasicGraph}. Users shouldn't
105   * have to reference this enum unless they are implementing the {@link BasicGraph} interface.
106   */
107  public enum Presence {
108    EDGE_EXISTS
109  }
110}
