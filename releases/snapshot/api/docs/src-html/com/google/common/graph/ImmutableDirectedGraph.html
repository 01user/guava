017package com.google.common.graph;
018
019import com.google.common.annotations.Beta;
020import com.google.errorprone.annotations.CanIgnoreReturnValue;
021
022/**
023 * Implementation of an immutable directed graph consisting of nodes of type N
024 * and edges of type E.
025 *
026 * <p>Some invariants/assumptions are maintained in this implementation:
027 * <ul>
028 * <li>An edge has exactly two end-points (source node and target node), which
029 *     may or may not be distinct.
030 * <li>By default, this is not a multigraph, that is, parallel edges (multiple
031 *     edges directed from node1 to node2) are not allowed.  If you want a multigraph,
032 *     build the graph with the 'multigraph' option:
033 *     <pre>ImmutableDirectedGraph.builder(Graphs.config().multigraph()).build();</pre>
034 * <li>Anti-parallel edges (same incident nodes but in opposite direction,
035 *     e.g. (node1, node2) and (node2, node1)) are always allowed.
036 * <li>Edges are not adjacent to themselves by definition. In the case of a
037 *     self-loop, a node can be adjacent to itself, but an edge will never be.
038 * </ul>
039 *
040 * @author James Sexton
041 * @author Joshua O'Madadhain
042 * @author Omar Darwish
043 * @param <N> Node parameter type
044 * @param <E> Edge parameter type
045 * @see AbstractConfigurableGraph
046 * @see AbstractImmutableGraph
047 * @since 20.0
048 */
049@Beta
050public final class ImmutableDirectedGraph<N, E> extends AbstractImmutableGraph<N, E>
051    implements DirectedGraph<N, E> {
052
053  private ImmutableDirectedGraph(Builder<N, E> builder) {
054    super(builder);
055  }
056
057  @Override
058  public N source(Object edge) {
059    return checkedIncidentNodes(edge).node1();
060  }
061
062  @Override
063  public N target(Object edge) {
064    return checkedIncidentNodes(edge).node2();
065  }
066
067  /**
068   * Returns a new builder. The generated builder is equivalent to the builder
069   * created by the {@code Builder} constructor.
070   */
071  public static <N, E> Builder<N, E> builder() {
072    return new Builder<N, E>();
073  }
074
075  /**
076   * Returns a new builder. The generated builder is equivalent to the builder
077   * created by the {@code Builder} constructor.
078   *
079   * @param config an instance of {@code GraphConfig} with the intended
080   *        graph configuration.
081   */
082  public static <N, E> Builder<N, E> builder(GraphConfig config) {
083    return new Builder<N, E>(config);
084  }
085
086  /**
087   * Returns an immutable copy of the input graph.
088   */
089  public static <N, E> ImmutableDirectedGraph<N, E> copyOf(DirectedGraph<N, E> graph) {
090    return new Builder<N, E>(graph).build();
091  }
092
093  /**
094   * A builder for creating immutable directed graph instances.
095   *
096   * @param <N> Node parameter type
097   * @param <E> edge parameter type
098   * @see GraphConfig
099   */
100  public static final class Builder<N, E> extends AbstractImmutableGraph.Builder<N, E> {
101
102    /**
103     * Creates a new builder with the default graph configuration.
104     */
105    public Builder() {
106      super(Graphs.<N, E>createDirected());
107    }
108
109    /**
110     * Creates a new builder with the specified graph configuration.
111     */
112    public Builder(GraphConfig config) {
113      super(Graphs.<N, E>createDirected(config));
114    }
115
116    /**
117     * Creates a new builder whose internal state is that of {@code graph}.
118     *
119     * <p>NOTE: This constructor should only be used in the case where it will be immediately
120     * followed by a call to {@code build}, to ensure that the input graph will not be modified.
121     * Currently the only such context is {@code Immutable*Graph.copyOf()}, which use these
122     * constructors to avoid making an extra copy of the graph state.
123     * @see ImmutableDirectedGraph#copyOf(DirectedGraph)
124     */
125    private Builder(DirectedGraph<N, E> graph) {
126      super(graph);
127    }
128
129    @Override
130    NodeConnections<N, E> nodeConnectionsOf(N node) {
131      return DirectedNodeConnections.ofImmutable(
132          graph.predecessors(node), graph.successors(node),
133          graph.inEdges(node), graph.outEdges(node));
134    }
135
136    @Override
137    public ImmutableDirectedGraph<N, E> build() {
138      return new ImmutableDirectedGraph<N, E>(this);
139    }
140
141    @Override
142    @CanIgnoreReturnValue
143    public Builder<N, E> addNode(N node) {
144      return (Builder<N, E>) super.addNode(node); // Refine the return type
145    }
146
147    @Override
148    @CanIgnoreReturnValue
149    public Builder<N, E> addEdge(E edge, N node1, N node2) {
150      return (Builder<N, E>) super.addEdge(edge, node1, node2); // Refine the return type
151    }
152
153    @Override
154    @CanIgnoreReturnValue
155    public Builder<N, E> addGraph(Graph<N, E> graphToAdd) {
156      return (Builder<N, E>) super.addGraph(graphToAdd); // Refine the return type
157    }
158  }
159}
