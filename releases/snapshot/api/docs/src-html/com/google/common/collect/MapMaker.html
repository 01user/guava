048/**
049 * <p>A builder of {@link ConcurrentMap} instances having any combination of the following features:
050 *
051 * <ul>
052 * <li>keys or values automatically wrapped in {@linkplain WeakReference weak} references
053 * </ul>
054 *
055 * <p>Usage example: <pre>   {@code
056 *
057 *   ConcurrentMap<Request, Stopwatch> timers = new MapMaker()
058 *       .concurrencyLevel(4)
059 *       .weakKeys()
060 *       .makeMap();}</pre>
061 *
062 * <p>These features are all optional; {@code new MapMaker().makeMap()} returns a valid concurrent
063 * map that behaves similarly to a {@link ConcurrentHashMap}.
064 *
065 * <p>The returned map is implemented as a hash table with similar performance characteristics to
066 * {@link ConcurrentHashMap}. It supports all optional operations of the {@code ConcurrentMap}
067 * interface. It does not permit null keys or values.
068 *
069 * <p><b>Note:</b> by default, the returned map uses equality comparisons (the {@link Object#equals
070 * equals} method) to determine equality for keys or values. However, if {@link #weakKeys} was
071 * specified, the map uses identity ({@code ==}) comparisons instead for keys. Likewise, if
072 * {@link #weakValues}
073 * was specified, the map uses identity comparisons for values.
074 *
075 * <p>The view collections of the returned map have <i>weakly consistent iterators</i>. This means
076 * that they are safe for concurrent use, but if other threads modify the map after the iterator is
077 * created, it is undefined which of these changes, if any, are reflected in that iterator. These
078 * iterators never throw {@link ConcurrentModificationException}.
079 *
080 * <p>If {@link #weakKeys}/{@link #weakValues}
081 * are requested, it is possible for a key or value present in the map to be reclaimed by the
082 * garbage collector. Entries with reclaimed keys or values may be removed from the map on each map
083 * modification or on occasional map accesses; such entries may be counted by {@link Map#size}, but
084 * will never be visible to read or write operations. A partially-reclaimed entry is never exposed
085 * to the user. Any {@link java.util.Map.Entry} instance retrieved from the map's
086 * {@linkplain Map#entrySet entry set} is a snapshot of that entry's state at the time of retrieval;
087 * such entries do, however, support {@link java.util.Map.Entry#setValue}, which simply calls
088 * {@link Map#put} on the entry's key.
089 *
090 * <p>The maps produced by {@code MapMaker} are serializable, and the deserialized maps retain all
091 * the configuration properties of the original map. During deserialization, if the original map had
092 * used weak
093 * references, the entries are reconstructed as they were, but it's not unlikely they'll be quickly
094 * garbage-collected before they are ever accessed.
095 *
096 * <p>{@code new MapMaker().weakKeys().makeMap()} is a recommended replacement for
097 * {@link java.util.WeakHashMap}, but note that it compares keys using object identity whereas
098 * {@code WeakHashMap} uses {@link Object#equals}.
099 *
100 * @author Bob Lee
101 * @author Charles Fry
102 * @author Kevin Bourrillion
103 * @since 2.0
104 */

183  /**
184   * Guides the allowed concurrency among update operations. Used as a hint for internal sizing. The
185   * table is internally partitioned to try to permit the indicated number of concurrent updates
186   * without contention. Because assignment of entries to these partitions is not necessarily
187   * uniform, the actual concurrency observed may vary. Ideally, you should choose a value to
188   * accommodate as many threads as will ever concurrently modify the table. Using a significantly
189   * higher value than you need can waste space and time, and a significantly lower value can lead
190   * to thread contention. But overestimates and underestimates within an order of magnitude do not
191   * usually have much noticeable impact. A value of one permits only one thread to modify the map
192   * at a time, but since read operations can proceed concurrently, this still yields higher
193   * concurrency than full synchronization. Defaults to 4.
194   *
195   * <p><b>Note:</b> Prior to Guava release 9.0, the default was 16. It is possible the default will
196   * change again in the future. If you care about this value, you should always choose it
197   * explicitly.
198   *
199   * @throws IllegalArgumentException if {@code concurrencyLevel} is nonpositive
200   * @throws IllegalStateException if a concurrency level was already set
201   */
202  @CanIgnoreReturnValue
203  @Override
204  public MapMaker concurrencyLevel(int concurrencyLevel) {
205    checkState(
206        this.concurrencyLevel == UNSET_INT,
207        "concurrency level was already set to %s",
208        this.concurrencyLevel);
209    checkArgument(concurrencyLevel > 0);
210    this.concurrencyLevel = concurrencyLevel;
211    return this;
212  }
213
214  int getConcurrencyLevel() {
215    return (concurrencyLevel == UNSET_INT) ? DEFAULT_CONCURRENCY_LEVEL : concurrencyLevel;
216  }

236  MapMaker setKeyStrength(Strength strength) {
237    checkState(keyStrength == null, "Key strength was already set to %s", keyStrength);
238    keyStrength = checkNotNull(strength);
239    checkArgument(keyStrength != SOFT, "Soft keys are not supported");
240    if (strength != Strength.STRONG) {

317  /**
318   * Specifies that each entry should be automatically removed from the map once a fixed duration
319   * has elapsed after the entry's creation, or the most recent replacement of its value.
320   *
321   * <p>When {@code duration} is zero, elements can be successfully added to the map, but are
322   * evicted immediately. It can be useful in testing, or to disable caching temporarily without
323   * a code change.
324   *
325   * <p>Expired entries may be counted by {@link Map#size}, but will never be visible to read or
326   * write operations. Expired entries are currently cleaned up during write operations, or during
327   * occasional read operations in the absense of writes; though this behavior may change in the
328   * future.
329   *
330   * @param duration the length of time after an entry is created that it should be automatically
331   *     removed
332   * @param unit the unit that {@code duration} is expressed in
333   * @throws IllegalArgumentException if {@code duration} is negative
334   * @throws IllegalStateException if the time to live or time to idle was already set
335   * @deprecated Caching functionality in {@code MapMaker} has been moved to
336   *     {@link com.google.common.cache.CacheBuilder}, with {@link #expireAfterWrite} being replaced
337   *     by {@link com.google.common.cache.CacheBuilder#expireAfterWrite}. Note that
338   *     {@code CacheBuilder} is simply an enhanced API for an implementation which was branched
339   *     from {@code MapMaker}.
340   */
341  @CanIgnoreReturnValue
342  @Deprecated
343  @Override
344  MapMaker expireAfterWrite(long duration, TimeUnit unit) {
345    checkExpiration(duration, unit);
346    this.expireAfterWriteNanos = unit.toNanos(duration);
347    if (duration == 0) {
348      this.evictImmediately = true;
349    }
350    useCustomMap = true;
351    return this;
352  }
353
354  private void checkExpiration(long duration, TimeUnit unit) {
355    checkState(
356        expireAfterWriteNanos == UNSET_INT,
357        "expireAfterWrite was already set to %s ns",
358        expireAfterWriteNanos);
359    checkArgument(duration >= 0, "duration cannot be negative: %s %s", duration, unit);
360  }
361
362  long getExpireAfterWriteNanos() {
363    return (expireAfterWriteNanos == UNSET_INT) ? DEFAULT_EXPIRATION_NANOS : expireAfterWriteNanos;
364  }

370  /**
371   * Builds a thread-safe map. This method does not alter the state of this {@code MapMaker}
372   * instance, so it can be invoked again to create multiple independent maps.
373   *
374   * <p>The bulk operations {@code putAll}, {@code equals}, and {@code clear} are not guaranteed to
375   * be performed atomically on the returned map. Additionally, {@code size} and
376   * {@code containsValue} are implemented as bulk read operations, and thus may fail to observe
377   * concurrent writes.
378   *
379   * @return a serializable concurrent map having the requested features
380   */
381  @Override
382  public <K, V> ConcurrentMap<K, V> makeMap() {
383    if (!useCustomMap) {
384      return new ConcurrentHashMap<K, V>(getInitialCapacity(), 0.75f, getConcurrencyLevel());
385    }
386    return evictImmediately ? new NullConcurrentMap<K, V>() : new MapMakerInternalMap<K, V>(this);
387  }

399  /**
400   * Builds a map that supports atomic, on-demand computation of values. {@link Map#get} either
401   * returns an already-computed value for the given key, atomically computes it using the supplied
402   * function, or, if another thread is currently computing the value for this key, simply waits for
403   * that thread to finish and returns its computed value. Note that the function may be executed
404   * concurrently by multiple threads, but only for distinct keys.
405   *
406   * <p>New code should use {@link com.google.common.cache.CacheBuilder}, which supports
407   * {@linkplain com.google.common.cache.CacheStats statistics} collection, introduces the
408   * {@link com.google.common.cache.CacheLoader} interface for loading entries into the cache
409   * (allowing checked exceptions to be thrown in the process), and more cleanly separates
410   * computation from the cache's {@code Map} view.
411   *
412   * <p>If an entry's value has not finished computing yet, query methods besides {@code get} return
413   * immediately as if an entry doesn't exist. In other words, an entry isn't externally visible
414   * until the value's computation completes.
415   *
416   * <p>{@link Map#get} on the returned map will never return {@code null}. It may throw:
417   *
418   * <ul>
419   * <li>{@link NullPointerException} if the key is null or the computing function returns a null
420   * result
421   * <li>{@link ComputationException} if an exception was thrown by the computing function. If that
422   * exception is already of type {@link ComputationException} it is propagated directly; otherwise
423   * it is wrapped.
424   * </ul>
425   *
426   * <p><b>Note:</b> Callers of {@code get} <i>must</i> ensure that the key argument is of type
427   * {@code K}. The {@code get} method accepts {@code Object}, so the key type is not checked at
428   * compile time. Passing an object of a type other than {@code K} can result in that object being
429   * unsafely passed to the computing function as type {@code K}, and unsafely stored in the map.
430   *
431   * <p>If {@link Map#put} is called before a computation completes, other threads waiting on the
432   * computation will wake up and return the stored value.
433   *
434   * <p>This method does not alter the state of this {@code MapMaker} instance, so it can be invoked
435   * again to create multiple independent maps.
436   *
437   * <p>Insertion, removal, update, and access operations on the returned map safely execute
438   * concurrently by multiple threads. Iterators on the returned map are weakly consistent,
439   * returning elements reflecting the state of the map at some point at or since the creation of
440   * the iterator. They do not throw {@link ConcurrentModificationException}, and may proceed
441   * concurrently with other operations.
442   *
443   * <p>The bulk operations {@code putAll}, {@code equals}, and {@code clear} are not guaranteed to
444   * be performed atomically on the returned map. Additionally, {@code size} and
445   * {@code containsValue} are implemented as bulk read operations, and thus may fail to observe
446   * concurrent writes.
447   *
448   * @param computingFunction the function used to compute new values
449   * @return a serializable concurrent map having the requested features
450   * @deprecated Caching functionality in {@code MapMaker} has been moved to
451   *     {@link com.google.common.cache.CacheBuilder}, with {@link #makeComputingMap} being replaced
452   *     by {@link com.google.common.cache.CacheBuilder#build}. See the
453   *     <a href="https://github.com/google/guava/wiki/MapMakerMigration">MapMaker Migration
454   *     Guide</a> for more details.
455   */
456  @Deprecated
457  @Override
458  <K, V> ConcurrentMap<K, V> makeComputingMap(Function<? super K, ? extends V> computingFunction) {
459    return evictImmediately
460        ? new NullComputingConcurrentMap<K, V>(computingFunction)
461        : new MapMaker.ComputingMapAdapter<K, V>(this, computingFunction);
462  }

464  /**
465   * Returns a string representation for this MapMaker instance. The exact form of the returned
466   * string is not specificed.
467   */
468  @Override
469  public String toString() {
470    MoreObjects.ToStringHelper s = MoreObjects.toStringHelper(this);
471    if (initialCapacity != UNSET_INT) {
472      s.add("initialCapacity", initialCapacity);
473    }
474    if (concurrencyLevel != UNSET_INT) {
475      s.add("concurrencyLevel", concurrencyLevel);
476    }
477    if (expireAfterWriteNanos != UNSET_INT) {
478      s.add("expireAfterWrite", expireAfterWriteNanos + "ns");
479    }
480    if (keyStrength != null) {
481      s.add("keyStrength", Ascii.toLowerCase(keyStrength.toString()));
482    }
483    if (valueStrength != null) {
484      s.add("valueStrength", Ascii.toLowerCase(valueStrength.toString()));
485    }
486    if (keyEquivalence != null) {
487      s.addValue("keyEquivalence");
488    }
489    return s.toString();
490  }

546      checkNotNull(newValue);

556  /** Computes on retrieval and evicts the result. */
557  static final class NullComputingConcurrentMap<K, V> extends NullConcurrentMap<K, V> {
558    private static final long serialVersionUID = 0;
559
560    final Function<? super K, ? extends V> computingFunction;
561
562    NullComputingConcurrentMap(Function<? super K, ? extends V> computingFunction) {
563      this.computingFunction = checkNotNull(computingFunction);
564    }
565
566    @SuppressWarnings("unchecked") // unsafe, which is why Cache is preferred
567    @Override
568    public V get(Object k) {
569      K key = (K) k;
570      V value = compute(key);
571      checkNotNull(value, "%s returned null for key %s.", computingFunction, key);
572      return value;
573    }
574
575    private V compute(K key) {
576      checkNotNull(key);
577      try {
578        return computingFunction.apply(key);
579      } catch (ComputationException e) {
580        throw e;
581      } catch (Throwable t) {
582        throw new ComputationException(t);
583      }
584    }
585  }

587  /**
588   * Overrides get() to compute on demand. Also throws an exception when {@code null} is returned
589   * from a computation.
590   */
591  /*
592   * This might make more sense in ComputingConcurrentHashMap, but it causes a javac crash in some
593   * cases there: http://code.google.com/p/guava-libraries/issues/detail?id=950
594   */
595  static final class ComputingMapAdapter<K, V> extends ComputingConcurrentHashMap<K, V>
596      implements Serializable {
597    private static final long serialVersionUID = 0;
598
599    ComputingMapAdapter(MapMaker mapMaker, Function<? super K, ? extends V> computingFunction) {
600      super(mapMaker, computingFunction);
601    }
602
603    @SuppressWarnings("unchecked") // unsafe, which is one advantage of Cache over Map
604    @Override
605    public V get(Object key) {
606      V value;
607      try {
608        value = getOrCompute((K) key);
609      } catch (ExecutionException e) {
610        Throwable cause = e.getCause();
611        Throwables.propagateIfInstanceOf(cause, ComputationException.class);
612        throw new ComputationException(cause);
613      }
614
615      if (value == null) {
616        throw new NullPointerException(computingFunction + " returned null for key " + key + ".");
617      }
618      return value;
619    }
620  }
