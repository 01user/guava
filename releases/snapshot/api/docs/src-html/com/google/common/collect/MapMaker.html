039/**
040 * <p>A builder of {@link ConcurrentMap} instances that can have keys or values automatically
041 * wrapped in {@linkplain WeakReference weak} references.
042 *
043 * <p>Usage example: <pre>   {@code
044 *
045 *   ConcurrentMap<Request, Stopwatch> timers = new MapMaker()
046 *       .concurrencyLevel(4)
047 *       .weakKeys()
048 *       .makeMap();}</pre>
049 *
050 * <p>These features are all optional; {@code new MapMaker().makeMap()} returns a valid concurrent
051 * map that behaves similarly to a {@link ConcurrentHashMap}.
052 *
053 * <p>The returned map is implemented as a hash table with similar performance characteristics to
054 * {@link ConcurrentHashMap}. It supports all optional operations of the {@code ConcurrentMap}
055 * interface. It does not permit null keys or values.
056 *
057 * <p><b>Note:</b> by default, the returned map uses equality comparisons (the {@link Object#equals
058 * equals} method) to determine equality for keys or values. However, if {@link #weakKeys} was
059 * specified, the map uses identity ({@code ==}) comparisons instead for keys. Likewise, if
060 * {@link #weakValues} was specified, the map uses identity comparisons for values.
061 *
062 * <p>The view collections of the returned map have <i>weakly consistent iterators</i>. This means
063 * that they are safe for concurrent use, but if other threads modify the map after the iterator is
064 * created, it is undefined which of these changes, if any, are reflected in that iterator. These
065 * iterators never throw {@link ConcurrentModificationException}.
066 *
067 * <p>If {@link #weakKeys} or {@link #weakValues} are requested, it is possible for a key or value
068 * present in the map to be reclaimed by the
069 * garbage collector. Entries with reclaimed keys or values may be removed from the map on each map
070 * modification or on occasional map accesses; such entries may be counted by {@link Map#size}, but
071 * will never be visible to read or write operations. A partially-reclaimed entry is never exposed
072 * to the user. Any {@link java.util.Map.Entry} instance retrieved from the map's
073 * {@linkplain Map#entrySet entry set} is a snapshot of that entry's state at the time of retrieval;
074 * such entries do, however, support {@link java.util.Map.Entry#setValue}, which simply calls
075 * {@link Map#put} on the entry's key.
076 *
077 * <p>The maps produced by {@code MapMaker} are serializable, and the deserialized maps retain all
078 * the configuration properties of the original map. During deserialization, if the original map had
079 * used weak references, the entries are reconstructed as they were, but it's not unlikely they'll
080 * be quickly garbage-collected before they are ever accessed.
081 *
082 * <p>{@code new MapMaker().weakKeys().makeMap()} is a recommended replacement for
083 * {@link java.util.WeakHashMap}, but note that it compares keys using object identity whereas
084 * {@code WeakHashMap} uses {@link Object#equals}.
085 *
086 * @author Bob Lee
087 * @author Charles Fry
088 * @author Kevin Bourrillion
089 * @since 2.0
090 */
091@GwtCompatible(emulated = true)
092public final class MapMaker {
093  private static final int DEFAULT_INITIAL_CAPACITY = 16;
094  private static final int DEFAULT_CONCURRENCY_LEVEL = 4;
095
096  static final int UNSET_INT = -1;
097
098  // TODO(kevinb): dispense with this after benchmarking
099  boolean useCustomMap;
100
101  int initialCapacity = UNSET_INT;
102  int concurrencyLevel = UNSET_INT;
103
104  Strength keyStrength;
105  Strength valueStrength;
106
107  Equivalence<Object> keyEquivalence;
108
109  /**
110   * Constructs a new {@code MapMaker} instance with default settings, including strong keys, strong
111   * values, and no automatic eviction of any kind.
112   */
113  public MapMaker() {}
114
115  /**
116   * Sets a custom {@code Equivalence} strategy for comparing keys.
117   *
118   * <p>By default, the map uses {@link Equivalence#identity} to determine key equality when
119   * {@link #weakKeys} is specified, and {@link Equivalence#equals()} otherwise. The only place this
120   * is used is in {@link Interners.WeakInterner}.
121   */
122  @CanIgnoreReturnValue
123  @GwtIncompatible // To be supported
124  MapMaker keyEquivalence(Equivalence<Object> equivalence) {
125    checkState(keyEquivalence == null, "key equivalence was already set to %s", keyEquivalence);
126    keyEquivalence = checkNotNull(equivalence);
127    this.useCustomMap = true;
128    return this;
129  }
130
131  Equivalence<Object> getKeyEquivalence() {
132    return MoreObjects.firstNonNull(keyEquivalence, getKeyStrength().defaultEquivalence());
133  }
134
135  /**
136   * Sets the minimum total size for the internal hash tables. For example, if the initial capacity
137   * is {@code 60}, and the concurrency level is {@code 8}, then eight segments are created, each
138   * having a hash table of size eight. Providing a large enough estimate at construction time
139   * avoids the need for expensive resizing operations later, but setting this value unnecessarily
140   * high wastes memory.
141   *
142   * @throws IllegalArgumentException if {@code initialCapacity} is negative
143   * @throws IllegalStateException if an initial capacity was already set
144   */
145  @CanIgnoreReturnValue
146  public MapMaker initialCapacity(int initialCapacity) {
147    checkState(
148        this.initialCapacity == UNSET_INT,
149        "initial capacity was already set to %s",
150        this.initialCapacity);
151    checkArgument(initialCapacity >= 0);
152    this.initialCapacity = initialCapacity;
153    return this;
154  }
155
156  int getInitialCapacity() {
157    return (initialCapacity == UNSET_INT) ? DEFAULT_INITIAL_CAPACITY : initialCapacity;
158  }
159
160  /**
161   * Guides the allowed concurrency among update operations. Used as a hint for internal sizing. The
162   * table is internally partitioned to try to permit the indicated number of concurrent updates
163   * without contention. Because assignment of entries to these partitions is not necessarily
164   * uniform, the actual concurrency observed may vary. Ideally, you should choose a value to
165   * accommodate as many threads as will ever concurrently modify the table. Using a significantly
166   * higher value than you need can waste space and time, and a significantly lower value can lead
167   * to thread contention. But overestimates and underestimates within an order of magnitude do not
168   * usually have much noticeable impact. A value of one permits only one thread to modify the map
169   * at a time, but since read operations can proceed concurrently, this still yields higher
170   * concurrency than full synchronization. Defaults to 4.
171   *
172   * <p><b>Note:</b> Prior to Guava release 9.0, the default was 16. It is possible the default will
173   * change again in the future. If you care about this value, you should always choose it
174   * explicitly.
175   *
176   * @throws IllegalArgumentException if {@code concurrencyLevel} is nonpositive
177   * @throws IllegalStateException if a concurrency level was already set
178   */
179  @CanIgnoreReturnValue
180  public MapMaker concurrencyLevel(int concurrencyLevel) {
181    checkState(
182        this.concurrencyLevel == UNSET_INT,
183        "concurrency level was already set to %s",
184        this.concurrencyLevel);
185    checkArgument(concurrencyLevel > 0);
186    this.concurrencyLevel = concurrencyLevel;
187    return this;
188  }
189
190  int getConcurrencyLevel() {
191    return (concurrencyLevel == UNSET_INT) ? DEFAULT_CONCURRENCY_LEVEL : concurrencyLevel;
192  }
193
194  /**
195   * Specifies that each key (not value) stored in the map should be wrapped in a
196   * {@link WeakReference} (by default, strong references are used).
197   *
198   * <p><b>Warning:</b> when this method is used, the resulting map will use identity ({@code ==})
199   * comparison to determine equality of keys, which is a technical violation of the {@link Map}
200   * specification, and may not be what you expect.
201   *
202   * @throws IllegalStateException if the key strength was already set
203   * @see WeakReference
204   */
205  @CanIgnoreReturnValue
206  @GwtIncompatible // java.lang.ref.WeakReference
207  public MapMaker weakKeys() {
208    return setKeyStrength(Strength.WEAK);
209  }
210
211  MapMaker setKeyStrength(Strength strength) {
212    checkState(keyStrength == null, "Key strength was already set to %s", keyStrength);
213    keyStrength = checkNotNull(strength);
214    if (strength != Strength.STRONG) {
215      // STRONG could be used during deserialization.
216      useCustomMap = true;
217    }
218    return this;
219  }
220
221  Strength getKeyStrength() {
222    return MoreObjects.firstNonNull(keyStrength, Strength.STRONG);
223  }
224
225  /**
226   * Specifies that each value (not key) stored in the map should be wrapped in a
227   * {@link WeakReference} (by default, strong references are used).
228   *
229   * <p>Weak values will be garbage collected once they are weakly reachable. This makes them a poor
230   * candidate for caching.
231   *
232   * <p><b>Warning:</b> when this method is used, the resulting map will use identity ({@code ==})
233   * comparison to determine equality of values. This technically violates the specifications of the
234   * methods {@link Map#containsValue containsValue}, {@link ConcurrentMap#remove(Object, Object)
235   * remove(Object, Object)} and {@link ConcurrentMap#replace(Object, Object, Object) replace(K, V,
236   * V)}, and may not be what you expect.
237   *
238   * @throws IllegalStateException if the value strength was already set
239   * @see WeakReference
240   */
241  @CanIgnoreReturnValue
242  @GwtIncompatible // java.lang.ref.WeakReference
243  public MapMaker weakValues() {
244    return setValueStrength(Strength.WEAK);
245  }
246
247  MapMaker setValueStrength(Strength strength) {
248    checkState(valueStrength == null, "Value strength was already set to %s", valueStrength);
249    valueStrength = checkNotNull(strength);
250    if (strength != Strength.STRONG) {
251      // STRONG could be used during deserialization.
252      useCustomMap = true;
253    }
254    return this;
255  }
256
257  Strength getValueStrength() {
258    return MoreObjects.firstNonNull(valueStrength, Strength.STRONG);
259  }
260
261  /**
262   * Builds a thread-safe map. This method does not alter the state of this {@code MapMaker}
263   * instance, so it can be invoked again to create multiple independent maps.
264   *
265   * <p>The bulk operations {@code putAll}, {@code equals}, and {@code clear} are not guaranteed to
266   * be performed atomically on the returned map. Additionally, {@code size} and
267   * {@code containsValue} are implemented as bulk read operations, and thus may fail to observe
268   * concurrent writes.
269   *
270   * @return a serializable concurrent map having the requested features
271   */
272  public <K, V> ConcurrentMap<K, V> makeMap() {
273    if (!useCustomMap) {
274      return new ConcurrentHashMap<K, V>(getInitialCapacity(), 0.75f, getConcurrencyLevel());
275    }
276    return new MapMakerInternalMap<K, V>(this);
277  }
278
279  /**
280   * Returns a MapMakerInternalMap for the benefit of internal callers that use features of that
281   * class not exposed through ConcurrentMap.
282   */
283  @GwtIncompatible // MapMakerInternalMap
284  <K, V> MapMakerInternalMap<K, V> makeCustomMap() {
285    return new MapMakerInternalMap<K, V>(this);
286  }

288  /**
289   * Builds a map that supports atomic, on-demand computation of values. {@link Map#get} either
290   * returns an already-computed value for the given key, atomically computes it using the supplied
291   * function, or, if another thread is currently computing the value for this key, simply waits for
292   * that thread to finish and returns its computed value. Note that the function may be executed
293   * concurrently by multiple threads, but only for distinct keys.
294   *
295   * <p>New code should use {@link com.google.common.cache.CacheBuilder}, which supports
296   * {@linkplain com.google.common.cache.CacheStats statistics} collection, introduces the
297   * {@link com.google.common.cache.CacheLoader} interface for loading entries into the cache
298   * (allowing checked exceptions to be thrown in the process), and more cleanly separates
299   * computation from the cache's {@code Map} view.
300   *
301   * <p>If an entry's value has not finished computing yet, query methods besides {@code get} return
302   * immediately as if an entry doesn't exist. In other words, an entry isn't externally visible
303   * until the value's computation completes.
304   *
305   * <p>{@link Map#get} on the returned map will never return {@code null}. It may throw:
306   *
307   * <ul>
308   * <li>{@link NullPointerException} if the key is null or the computing function returns a null
309   * result
310   * <li>{@link ComputationException} if an exception was thrown by the computing function. If that
311   * exception is already of type {@link ComputationException} it is propagated directly; otherwise
312   * it is wrapped.
313   * </ul>
314   *
315   * <p><b>Note:</b> Callers of {@code get} <i>must</i> ensure that the key argument is of type
316   * {@code K}. The {@code get} method accepts {@code Object}, so the key type is not checked at
317   * compile time. Passing an object of a type other than {@code K} can result in that object being
318   * unsafely passed to the computing function as type {@code K}, and unsafely stored in the map.
319   *
320   * <p>If {@link Map#put} is called before a computation completes, other threads waiting on the
321   * computation will wake up and return the stored value.
322   *
323   * <p>This method does not alter the state of this {@code MapMaker} instance, so it can be invoked
324   * again to create multiple independent maps.
325   *
326   * <p>Insertion, removal, update, and access operations on the returned map safely execute
327   * concurrently by multiple threads. Iterators on the returned map are weakly consistent,
328   * returning elements reflecting the state of the map at some point at or since the creation of
329   * the iterator. They do not throw {@link ConcurrentModificationException}, and may proceed
330   * concurrently with other operations.
331   *
332   * <p>The bulk operations {@code putAll}, {@code equals}, and {@code clear} are not guaranteed to
333   * be performed atomically on the returned map. Additionally, {@code size} and
334   * {@code containsValue} are implemented as bulk read operations, and thus may fail to observe
335   * concurrent writes.
336   *
337   * @param computingFunction the function used to compute new values
338   * @return a serializable concurrent map having the requested features
339   * @deprecated Caching functionality in {@code MapMaker} has been moved to
340   *     {@link com.google.common.cache.CacheBuilder}, with {@link #makeComputingMap} being replaced
341   *     by {@link com.google.common.cache.CacheBuilder#build}. See the
342   *     <a href="https://github.com/google/guava/wiki/MapMakerMigration">MapMaker Migration
343   *     Guide</a> for more details.
344   */
345  @Deprecated
346  <K, V> ConcurrentMap<K, V> makeComputingMap(Function<? super K, ? extends V> computingFunction) {
347    return new MapMaker.ComputingMapAdapter<K, V>(this, computingFunction);
348  }
349
350  /**
351   * Returns a string representation for this MapMaker instance. The exact form of the returned
352   * string is not specificed.
353   */
354  @Override
355  public String toString() {
356    MoreObjects.ToStringHelper s = MoreObjects.toStringHelper(this);
357    if (initialCapacity != UNSET_INT) {
358      s.add("initialCapacity", initialCapacity);
359    }
360    if (concurrencyLevel != UNSET_INT) {
361      s.add("concurrencyLevel", concurrencyLevel);
362    }
363    if (keyStrength != null) {
364      s.add("keyStrength", Ascii.toLowerCase(keyStrength.toString()));
365    }
366    if (valueStrength != null) {
367      s.add("valueStrength", Ascii.toLowerCase(valueStrength.toString()));
368    }
369    if (keyEquivalence != null) {
370      s.addValue("keyEquivalence");
371    }
372    return s.toString();
373  }

375  /**
376   * Overrides get() to compute on demand. Also throws an exception when {@code null} is returned
377   * from a computation.
378   */
379  /*
380   * This might make more sense in ComputingConcurrentHashMap, but it causes a javac crash in some
381   * cases there: http://code.google.com/p/guava-libraries/issues/detail?id=950
382   */
383  static final class ComputingMapAdapter<K, V> extends ComputingConcurrentHashMap<K, V>
384      implements Serializable {
385    private static final long serialVersionUID = 0;
386
387    ComputingMapAdapter(MapMaker mapMaker, Function<? super K, ? extends V> computingFunction) {
388      super(mapMaker, computingFunction);
389    }
390
391    @SuppressWarnings("unchecked") // unsafe, which is one advantage of Cache over Map
392    @Override
393    public V get(Object key) {
394      V value;
395      try {
396        value = getOrCompute((K) key);
397      } catch (ExecutionException e) {
398        Throwable cause = e.getCause();
399        Throwables.propagateIfInstanceOf(cause, ComputationException.class);
400        throw new ComputationException(cause);
401      }
402
403      if (value == null) {
404        throw new NullPointerException(computingFunction + " returned null for key " + key + ".");
405      }
406      return value;
407    }
408  }
