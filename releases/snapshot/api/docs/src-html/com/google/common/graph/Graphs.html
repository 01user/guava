055  /**
056   * Returns the node at the other end of {@code edge} from {@code node}.
057   *
058   * @throws UnsupportedOperationException if {@code graph} is a {@link Hypergraph}
059   * @throws IllegalArgumentException if {@code edge} is not incident to {@code node}
060   */
061  public static <N> N oppositeNode(Network<N, ?> graph, Object edge, Object node) {
062    if (graph instanceof Hypergraph) {
063      throw new UnsupportedOperationException();
064    }
065
066    checkNotNull(node, "node");
067    Iterator<N> incidentNodesIterator = graph.incidentNodes(edge).iterator();
068    N node1 = incidentNodesIterator.next();
069    N node2 = incidentNodesIterator.hasNext() ? incidentNodesIterator.next() : node1;
070    if (node.equals(node1)) {
071      return node2;
072    } else {
073      checkArgument(node.equals(node2), "Edge %s is not incident to node %s", edge, node);
074      return node1;
075    }
076  }

078  /**
079   * Returns an unmodifiable view of edges that are parallel to {@code edge}, i.e. the set of edges
080   * that connect the same nodes in the same direction (if any). An edge is not parallel to itself.
081   *
082   * @throws UnsupportedOperationException if {@code graph} is a {@link Hypergraph}
083   * @throws IllegalArgumentException if {@code edge} is not present in {@code graph}
084   */
085  public static <N, E> Set<E> parallelEdges(Network<N, E> graph, Object edge) {
086    if (graph instanceof Hypergraph) {
087      throw new UnsupportedOperationException();
088    }
089
090    Set<N> incidentNodes = graph.incidentNodes(edge); // Verifies that edge is in graph
091    if (!graph.allowsParallelEdges()) {
092      return ImmutableSet.of();
093    }
094    Iterator<N> incidentNodesIterator = incidentNodes.iterator();
095    N node1 = incidentNodesIterator.next();
096    N node2 = incidentNodesIterator.hasNext() ? incidentNodesIterator.next() : node1;
097    return Sets.difference(graph.edgesConnecting(node1, node2), ImmutableSet.of(edge));
098  }

100  /**
101   * Adds {@code edge} to {@code graph} with the specified incident {@code nodes}, in the order
102   * returned by {@code nodes}' iterator.
103   */
104  @CanIgnoreReturnValue
105  public static <N, E> boolean addEdge(MutableNetwork<N, E> graph, E edge, Iterable<N> nodes) {
106    checkNotNull(graph, "graph");
107    checkNotNull(edge, "edge");
108    checkNotNull(nodes, "nodes");
109    if (graph instanceof Hypergraph) {
110      return ((Hypergraph<N, E>) graph).addEdge(edge, nodes);
111    }
112
113    Iterator<N> nodesIterator = nodes.iterator();
114    checkArgument(nodesIterator.hasNext(),
115        "'graph' is not a Hypergraph, and 'nodes' has < 1 elements: %s", nodes);
116    N node1 = nodesIterator.next();
117    N node2 = nodesIterator.hasNext() ? nodesIterator.next() : node1;
118    checkArgument(!nodesIterator.hasNext(),
119        "'graph' is not a Hypergraph, and 'nodes' has > 2 elements: %s", nodes);
120    return graph.addEdge(edge, node1, node2);
121  }

130  /**
131   * Creates a mutable copy of {@code graph}, using all of its elements that satisfy
132   * {@code nodePredicate} and {@code edgePredicate}.
133   */
134  public static <N> MutableGraph<N> copyOf(Graph<N> graph, Predicate<? super N> nodePredicate) {
135    checkNotNull(graph, "graph");
136    checkNotNull(nodePredicate, "nodePredicate");
137    MutableGraph<N> copy = GraphBuilder.from(graph).expectedNodeCount(graph.nodes().size()).build();
138
139    for (N node : graph.nodes()) {
140      if (nodePredicate.apply(node)) {
141        copy.addNode(node);
142        for (N successor : graph.successors(node)) {
143          if (nodePredicate.apply(successor)) {
144            copy.addEdge(node, successor);
145          }
146        }
147      }
148      // TODO(b/28087289): update this when parallel edges are permitted to ensure that the correct
149      // multiplicity is preserved.
150    }
151
152    return copy;
153  }

182  /**
183   * Creates a mutable copy of {@code graph}, using all of its elements that satisfy
184   * {@code nodePredicate} and {@code edgePredicate}.
185   */
186  public static <N, E> MutableNetwork<N, E> copyOf(
187      Network<N, E> graph,
188      Predicate<? super N> nodePredicate,
189      Predicate<? super E> edgePredicate) {
190    checkNotNull(graph, "graph");
191    checkNotNull(nodePredicate, "nodePredicate");
192    checkNotNull(edgePredicate, "edgePredicate");
193    MutableNetwork<N, E> copy = NetworkBuilder.from(graph)
194        .expectedNodeCount(graph.nodes().size()).expectedEdgeCount(graph.edges().size()).build();
195    mergeNodesFrom(graph, copy, nodePredicate);
196
197    // We can't just call mergeEdgesFrom(graph, copy, edgePredicate) because addEdge() can add
198    // the edge's incident nodes if they are not present.
199    for (E edge : graph.edges()) {
200      if (edgePredicate.apply(edge)) {
201        Set<N> incidentNodes = graph.incidentNodes(edge);
202        if (copy.nodes().containsAll(incidentNodes)) {
203          addEdge(copy, edge, incidentNodes);
204        }
205      }
206    }
207
208    return copy;
209  }

253  /**
254   * Returns true iff {@code graph1} and {@code graph2} have the same node connections.
255   *
256   * <p>Note: {@link Network} instances can only be equal to other {@link Network} instances.
257   * In particular, {@link Graph}s that are not also {@link Network}s cannot be equal
258   * to {@link Network}s.
259   *
260   * @see Network#equals(Object)
261   */
262  public static boolean equal(@Nullable Graph<?> graph1, @Nullable Graph<?> graph2) {
263    // If both graphs are Network instances, use equal(Network, Network) instead
264    if (graph1 instanceof Network && graph2 instanceof Network) {
265      return equal((Network<?, ?>) graph1, (Network<?, ?>) graph2);
266    }
267
268    // Otherwise, if either graph is a Network (but not both), they can't be equal.
269    if (graph1 instanceof Network || graph2 instanceof Network) {
270      return false;
271    }
272
273    if (graph1 == graph2) {
274      return true;
275    }
276
277    if (graph1 == null || graph2 == null) {
278      return false;
279    }
280
281    if (!graph1.nodes().equals(graph2.nodes())) {
282      return false;
283    }
284
285    for (Object node : graph1.nodes()) {
286      if (!graph1.successors(node).equals(graph2.successors(node))) {
287        return false;
288      }
289      boolean bothUndirected = !graph1.isDirected() && !graph2.isDirected();
290      if (!bothUndirected && !graph1.predecessors(node).equals(graph2.predecessors(node))) {
291        return false;
292      }
293    }
294
295    return true;
296  }

298  /**
299   * Returns true iff {@code graph1} and {@code graph2} have the same node/edge relationships.
300   *
301   * @see Network#equals(Object)
302   */
303  public static boolean equal(@Nullable Network<?, ?> graph1, @Nullable Network<?, ?> graph2) {
304    if (graph1 == graph2) {
305      return true;
306    }
307
308    if (graph1 == null || graph2 == null) {
309      return false;
310    }
311
312    if (graph1.edges().size() != graph2.edges().size()) {
313      return false;
314    }
315
316    if (!graph1.nodes().equals(graph2.nodes())) {
317      return false;
318    }
319
320    for (Object node : graph1.nodes()) {
321      if (!graph1.inEdges(node).equals(graph2.inEdges(node))) {
322        return false;
323      }
324      boolean bothUndirected = !graph1.isDirected() && !graph2.isDirected();
325      if (!bothUndirected && !graph1.outEdges(node).equals(graph2.outEdges(node))) {
326        return false;
327      }
328    }
329
330    return true;
331  }

353  /**
354   * Returns a string representation of {@code graph}. Encodes edge direction if {@code graph}
355   * is directed.
356   */
357  public static String toString(Graph<?> graph) {
358    if (graph instanceof Network) {
359      return toString((Network<?, ?>) graph);
360    }
361    return String.format(GRAPH_FORMAT,
362        getPropertiesString(graph),
363        graph.nodes(),
364        adjacentNodesString(graph));
365  }

367  /**
368   * Returns a string representation of {@code graph}. Encodes edge direction if {@code graph}
369   * is directed.
370   */
371  public static String toString(Network<?, ?> graph) {
372    return String.format(GRAPH_FORMAT,
373        getPropertiesString(graph),
374        graph.nodes(),
375        Maps.asMap(graph.edges(), edgeToIncidentNodesString(graph)));
376  }

378  /**
379   * Returns a String of the adjacent node relationships for {@code graph}.
380   */
381  private static <N> String adjacentNodesString(final Graph<N> graph) {
382    checkNotNull(graph, "graph");
383    List<String> adjacencies = new ArrayList<String>();
384    // This will list each undirected edge twice (once as [n1, n2] and once as [n2, n1]); this is OK
385    for (N node : graph.nodes()) {
386      for (N successor : graph.successors(node)) {
387        adjacencies.add(
388            String.format(
389                graph.isDirected() ? DIRECTED_FORMAT : UNDIRECTED_FORMAT,
390                node, successor));
391      }
392    }
393
394    return String.format("{%s}", Joiner.on(", ").join(adjacencies));
395  }

397  /**
398   * Returns a map that is a live view of {@code graph}, with nodes as keys
399   * and the set of incident edges as values.
400   */
401  private static <N, E> Map<N, Set<E>> nodeToIncidentEdges(final Network<N, E> graph) {
402    checkNotNull(graph, "graph");
403    return Maps.asMap(graph.nodes(), new Function<N, Set<E>>() {
404      @Override
405      public Set<E> apply(N node) {
406        return graph.incidentEdges(node);
407      }
408    });
409  }
410
411  private static <N> Map<N, Set<N>> nodeToAdjacentNodes(final Graph<N> graph) {
412    checkNotNull(graph, "graph");
413    return Maps.asMap(graph.nodes(), new Function<N, Set<N>>() {
414      @Override
415      public Set<N> apply(N node) {
416        return graph.adjacentNodes(node);
417      }
418    });
419  }

421  /**
422   * Returns a function that transforms an edge into a string representation of its incident nodes
423   * in {@code graph}. The function's {@code apply} method will throw an
424   * {@link IllegalArgumentException} if {@code graph} does not contain {@code edge}.
425   */
426  private static Function<Object, String> edgeToIncidentNodesString(final Network<?, ?> graph) {
427    if (graph.isDirected()) {
428      return new Function<Object, String>() {
429        @Override
430        public String apply(Object edge) {
431          return String.format("<%s -> %s>",
432              graph.source(edge), graph.target(edge));
433          }
434        };
435    }
436    return new Function<Object, String>() {
437      @Override
438      public String apply(Object edge) {
439        return graph.incidentNodes(edge).toString();
440      }
441    };
442 }
