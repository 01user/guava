048  /**
049   * Returns the node at the other end of {@code edge} from {@code node}.
050   *
051   * @throws UnsupportedOperationException if {@code graph} is a {@link Hypergraph}
052   * @throws IllegalArgumentException if {@code edge} is not incident to {@code node}
053   */
054  public static <N> N oppositeNode(Graph<N, ?> graph, Object edge, Object node) {
055    if (graph instanceof Hypergraph) {
056      throw new UnsupportedOperationException();
057    }
058
059    checkNotNull(node, "node");
060    Iterator<N> incidentNodesIterator = graph.incidentNodes(edge).iterator();
061    N node1 = incidentNodesIterator.next();
062    N node2 = incidentNodesIterator.hasNext() ? incidentNodesIterator.next() : node1;
063    if (node.equals(node1)) {
064      return node2;
065    } else {
066      checkArgument(node.equals(node2), "Edge %s is not incident to node %s", edge, node);
067      return node1;
068    }
069  }
070
071  /**
072   * Returns an unmodifiable view of edges that are parallel to {@code edge}, i.e. the set of edges
073   * that connect the same nodes in the same direction (if any). An edge is not parallel to itself.
074   *
075   * @throws UnsupportedOperationException if {@code graph} is a {@link Hypergraph}
076   * @throws IllegalArgumentException if {@code edge} is not present in {@code graph}
077   */
078  public static <N, E> Set<E> parallelEdges(Graph<N, E> graph, Object edge) {
079    if (graph instanceof Hypergraph) {
080      throw new UnsupportedOperationException();
081    }
082
083    Set<N> incidentNodes = graph.incidentNodes(edge); // Verifies that edge is in graph
084    if (!graph.allowsParallelEdges()) {
085      return ImmutableSet.of();
086    }
087    Iterator<N> incidentNodesIterator = incidentNodes.iterator();
088    N node1 = incidentNodesIterator.next();
089    N node2 = incidentNodesIterator.hasNext() ? incidentNodesIterator.next() : node1;
090    return Sets.difference(graph.edgesConnecting(node1, node2), ImmutableSet.of(edge));
091  }

093  /**
094   * Adds {@code edge} to {@code graph} with the specified incident {@code nodes}, in the order
095   * returned by {@code nodes}' iterator.
096   */
097  @CanIgnoreReturnValue
098  public static <N, E> boolean addEdge(Graph<N, E> graph, E edge, Iterable<N> nodes) {
099    checkNotNull(graph, "graph");
100    checkNotNull(edge, "edge");
101    checkNotNull(nodes, "nodes");
102    if (graph instanceof Hypergraph) {
103      return ((Hypergraph<N, E>) graph).addEdge(edge, nodes);
104    }
105
106    Iterator<N> nodesIterator = nodes.iterator();
107    checkArgument(nodesIterator.hasNext(),
108        "'graph' is not a Hypergraph, and 'nodes' has < 1 elements: %s", nodes);
109    N node1 = nodesIterator.next();
110    N node2 = nodesIterator.hasNext() ? nodesIterator.next() : node1;
111    checkArgument(!nodesIterator.hasNext(),
112        "'graph' is not a Hypergraph, and 'nodes' has > 2 elements: %s", nodes);
113    return graph.addEdge(edge, node1, node2);
114  }

116  /**
117   * Creates a mutable copy of {@code graph}, using the same node and edge elements.
118   */
119  public static <N, E> Graph<N, E> copyOf(Graph<N, E> graph) {
120    return copyOf(graph, Predicates.alwaysTrue(), Predicates.alwaysTrue());
121  }
122
123  /**
124   * Creates a mutable copy of {@code graph}, using all of its elements that satisfy
125   * {@code nodePredicate} and {@code edgePredicate}.
126   */
127  public static <N, E> Graph<N, E> copyOf(
128      Graph<N, E> graph,
129      Predicate<? super N> nodePredicate,
130      Predicate<? super E> edgePredicate) {
131    checkNotNull(graph, "graph");
132    checkNotNull(nodePredicate, "nodePredicate");
133    checkNotNull(edgePredicate, "edgePredicate");
134    Graph<N, E> copy = GraphBuilder.from(graph)
135        .expectedNodeCount(graph.nodes().size()).expectedEdgeCount(graph.edges().size()).build();
136    mergeNodesFrom(graph, copy, nodePredicate);
137
138    // We can't just call mergeEdgesFrom(graph, copy, edgePredicate) because addEdge() can add
139    // the edge's incident nodes if they are not present.
140    for (E edge : graph.edges()) {
141      if (edgePredicate.apply(edge)) {
142        Set<N> incidentNodes = graph.incidentNodes(edge);
143        if (copy.nodes().containsAll(incidentNodes)) {
144          addEdge(copy, edge, incidentNodes);
145        }
146      }
147    }
148
149    return copy;
150  }

152  /**
153   * Copies all nodes from {@code original} into {@code copy}.
154   */
155  public static <N, E> void mergeNodesFrom(Graph<N, E> original, Graph<N, E> copy) {
156    mergeNodesFrom(original, copy, Predicates.alwaysTrue());
157  }
158
159  /**
160   * Copies all nodes from {@code original} into {@code copy} that satisfy {@code nodePredicate}.
161   */
162  public static <N, E> void mergeNodesFrom(
163      Graph<N, E> original, Graph<N, E> copy, Predicate<? super N> nodePredicate) {
164    checkNotNull(original, "original");
165    checkNotNull(copy, "copy");
166    checkNotNull(nodePredicate, "nodePredicate");
167    for (N node : Sets.filter(original.nodes(), nodePredicate)) {
168      copy.addNode(node);
169    }
170  }
171
172  /**
173   * Copies all edges from {@code original} into {@code copy}. Also copies all nodes incident
174   * to these edges.
175   */
176  public static <N, E> void mergeEdgesFrom(Graph<N, E> original, Graph<N, E> copy) {
177    mergeEdgesFrom(original, copy, Predicates.alwaysTrue());
178  }
179
180  /**
181   * Copies all edges from {@code original} into {@code copy} that satisfy {@code edgePredicate}.
182   * Also copies all nodes incident to these edges.
183   */
184  public static <N, E> void mergeEdgesFrom(
185      Graph<N, E> original, Graph<N, E> copy, Predicate<? super E> edgePredicate) {
186    checkNotNull(original, "original");
187    checkNotNull(copy, "copy");
188    checkNotNull(edgePredicate, "edgePredicate");
189    for (E edge : Sets.filter(original.edges(), edgePredicate)) {
190      addEdge(copy, edge, original.incidentNodes(edge));
191    }
192  }

194  /**
195   * Returns true iff {@code graph1} and {@code graph2} have the same node/edge relationships.
196   *
197   * @see Graph#equals(Object)
198   */
199  public static boolean equal(@Nullable Graph<?, ?> graph1, @Nullable Graph<?, ?> graph2) {
200    if (graph1 == graph2) {
201      return true;
202    }
203
204    if (graph1 == null || graph2 == null) {
205      return false;
206    }
207
208    if (graph1.edges().size() != graph2.edges().size()) {
209      return false;
210    }
211
212    if (!graph1.nodes().equals(graph2.nodes())) {
213      return false;
214    }
215
216    for (Object node : graph1.nodes()) {
217      if (!graph1.inEdges(node).equals(graph2.inEdges(node))) {
218        return false;
219      }
220      boolean bothUndirected = !graph1.isDirected() && !graph2.isDirected();
221      if (!bothUndirected && !graph1.outEdges(node).equals(graph2.outEdges(node))) {
222        return false;
223      }
224    }
225
226    return true;
227  }

229  /**
230   * Returns the hash code of {@code graph}.
231   *
232   * @see Graph#hashCode()
233   */
234  public static int hashCode(Graph<?, ?> graph) {
235    return nodeToIncidentEdges(graph).hashCode();
236  }

238  /**
239   * Returns a string representation of {@code graph}. Encodes edge direction if {@code graph}
240   * is directed.
241   */
242  public static String toString(Graph<?, ?> graph) {
243    return String.format("%s, nodes: %s, edges: %s",
244        getPropertiesString(graph),
245        graph.nodes(),
246        Maps.asMap(graph.edges(), edgeToIncidentNodesString(graph)));
247  }

249  /**
250   * Returns a {@link Predicate} that returns {@code true} if the input edge is a self-loop in
251   * {@code graph}. A self-loop is defined as an edge whose set of incident nodes has exactly one
252   * element. The predicate's {@code apply} method will throw an {@link IllegalArgumentException} if
253   * {@code graph} does not contain {@code edge}.
254   */
255  public static <E> Predicate<E> selfLoopPredicate(final Graph<?, E> graph) {
256    checkNotNull(graph, "graph");
257    return new Predicate<E>() {
258      @Override
259      public boolean apply(E edge) {
260        return (graph.incidentNodes(edge).size() == 1);
261      }
262    };
263  }
264
265  /**
266   * Returns a map that is a live view of {@code graph}, with nodes as keys
267   * and the set of incident edges as values.
268   */
269  private static <N, E> Map<N, Set<E>> nodeToIncidentEdges(final Graph<N, E> graph) {
270    checkNotNull(graph, "graph");
271    return Maps.asMap(graph.nodes(), new Function<N, Set<E>>() {
272      @Override
273      public Set<E> apply(N node) {
274        return graph.incidentEdges(node);
275      }
276    });
277  }
278
279  /**
280   * Returns a function that transforms an edge into a string representation of its incident nodes
281   * in {@code graph}. The function's {@code apply} method will throw an
282   * {@link IllegalArgumentException} if {@code graph} does not contain {@code edge}.
283   */
284  private static Function<Object, String> edgeToIncidentNodesString(final Graph<?, ?> graph) {
285    if (graph.isDirected()) {
286      return new Function<Object, String>() {
287        @Override
288        public String apply(Object edge) {
289          return String.format("<%s -> %s>",
290              graph.source(edge), graph.target(edge));
291          }
292        };
293    }
294    return new Function<Object, String>() {
295      @Override
296      public String apply(Object edge) {
297        return graph.incidentNodes(edge).toString();
298      }
299    };
300  }
301
302  /**
303   * Returns a string representation of the properties of {@code graph}.
304   */
305  private static String getPropertiesString(Graph<?, ?> graph) {
306    return String.format("isDirected: %s, allowsParallelEdges: %s, allowsSelfLoops: %s",
307        graph.isDirected(), graph.allowsParallelEdges(), graph.allowsSelfLoops());
308  }
