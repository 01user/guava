049  /**
050   * Returns true iff {@code graph} has at least one cycle. A cycle is defined as a non-empty subset
051   * of edges in a graph arranged to form a path (a sequence of adjacent outgoing edges) starting
052   * and ending with the same node.
053   *
054   * <p>This method will detect any non-empty cycle, including self-loops (a cycle of length 1).
055   */
056  public static boolean hasCycle(Graph<?> graph) {
057    int numEdges = graph.edges().size();
058    if (numEdges == 0) {
059      return false; // An edge-free graph is acyclic by definition.
060    }
061    if (!graph.isDirected() && numEdges >= graph.nodes().size()) {
062      return true; // Optimization for the undirected case: at least one cycle must exist.
063    }
064
065    Map<Object, NodeVisitState> visitedNodes =
066        Maps.newHashMapWithExpectedSize(graph.nodes().size());
067    for (Object node : graph.nodes()) {
068      if (subgraphHasCycle(graph, visitedNodes, node, null)) {
069        return true;
070      }
071    }
072    return false;
073  }

075  /**
076   * Returns true iff {@code network} has at least one cycle. A cycle is defined as a non-empty
077   * subset of edges in a graph arranged to form a path (a sequence of adjacent outgoing edges)
078   * starting and ending with the same node.
079   *
080   * <p>This method will detect any non-empty cycle, including self-loops (a cycle of length 1).
081   */
082  public static boolean hasCycle(Network<?, ?> network) {
083    // In a directed graph, parallel edges cannot introduce a cycle in an acyclic graph.
084    // However, in an undirected graph, any parallel edge induces a cycle in the graph.
085    if (!network.isDirected()
086        && network.allowsParallelEdges()
087        && network.edges().size() > network.asGraph().edges().size()) {
088      return true;
089    }
090    return hasCycle(network.asGraph());
091  }

093  /**
094   * Performs a traversal of the nodes reachable from {@code node}. If we ever reach a node we've
095   * already visited (following only outgoing edges and without reusing edges), we know there's a
096   * cycle in the graph.
097   */
098  private static boolean subgraphHasCycle(
099      Graph<?> graph,
100      Map<Object, NodeVisitState> visitedNodes,
101      Object node,
102      @Nullable Object previousNode) {
103    NodeVisitState state = visitedNodes.get(node);
104    if (state == NodeVisitState.COMPLETE) {
105      return false;
106    }
107    if (state == NodeVisitState.PENDING) {
108      return true;
109    }
110
111    visitedNodes.put(node, NodeVisitState.PENDING);
112    for (Object nextNode : graph.successors(node)) {
113      if (canTraverseWithoutReusingEdge(graph, nextNode, previousNode)
114          && subgraphHasCycle(graph, visitedNodes, nextNode, node)) {
115        return true;
116      }
117    }
118    visitedNodes.put(node, NodeVisitState.COMPLETE);
119    return false;
120  }

122  /**
123   * Determines whether an edge has already been used during traversal. In the directed case a cycle
124   * is always detected before reusing an edge, so no special logic is required. In the undirected
125   * case, we must take care not to "backtrack" over an edge (i.e. going from A to B and then going
126   * from B to A).
127   */
128  private static boolean canTraverseWithoutReusingEdge(
129      Graph<?> graph, Object nextNode, @Nullable Object previousNode) {
130    if (graph.isDirected() || !Objects.equal(previousNode, nextNode)) {
131      return true;
132    }
133    // This falls into the undirected A->B->A case. The Graph interface does not support parallel
134    // edges, so this traversal would require reusing the undirected AB edge.
135    return false;
136  }

138  /**
139   * Returns the transitive closure of {@code graph}. The transitive closure of a graph is another
140   * graph with an edge connecting node A to node B iff node B is {@link #reachableNodes(Graph,
141   * Object) reachable} from node A.
142   *
143   * <p>This is a "snapshot" based on the current topology of {@code graph}, rather than a live view
144   * of the transitive closure of {@code graph}. In other words, the returned {@link Graph} will not
145   * be updated after modifications to {@code graph}.
146   */
147  public static <N> Graph<N> transitiveClosure(Graph<N> graph) {
148    MutableGraph<N> transitiveClosure = GraphBuilder.from(graph).allowsSelfLoops(true).build();
149    // Every node is, at a minimum, reachable from itself. Since the resulting transitive closure
150    // will have no isolated nodes, we can skip adding nodes explicitly and let putEdge() do it.
151
152    if (graph.isDirected()) {
153      // Note: works for both directed and undirected graphs, but we only use in the directed case.
154      for (N node : graph.nodes()) {
155        for (N reachableNode : reachableNodes(graph, node)) {
156          transitiveClosure.putEdge(node, reachableNode);
157        }
158      }
159    } else {
160      // An optimization for the undirected case: for every node B reachable from node A,
161      // node A and node B have the same reachability set.
162      Set<N> visitedNodes = new HashSet<N>();
163      for (N node : graph.nodes()) {
164        if (!visitedNodes.contains(node)) {
165          ImmutableList<N> reachableNodes = ImmutableList.copyOf(reachableNodes(graph, node));
166          visitedNodes.addAll(reachableNodes);
167          for (int a = 0; a < reachableNodes.size(); ++a) {
168            N nodeU = reachableNodes.get(a);
169            for (int b = a; b < reachableNodes.size(); ++b) {
170              N nodeV = reachableNodes.get(b);
171              transitiveClosure.putEdge(nodeU, nodeV);
172            }
173          }
174        }
175      }
176    }
177
178    return transitiveClosure;
179  }

181  /**
182   * Returns the set of nodes that are reachable from {@code node}. Node B is defined as reachable
183   * from node A if there exists a path (a sequence of adjacent outgoing edges) starting at node A
184   * and ending at node B. Note that a node is always reachable from itself via a zero-length path.
185   *
186   * <p>This is a "snapshot" based on the current topology of {@code graph}, rather than a live view
187   * of the set of nodes reachable from {@code node}. In other words, the returned {@link Set} will
188   * not be updated after modifications to {@code graph}.
189   *
190   * @throws IllegalArgumentException if {@code node} is not present in {@code graph}
191   */
192  @SuppressWarnings("unchecked") // Throws an exception if node is not an element of graph.
193  public static <N> Set<N> reachableNodes(Graph<N> graph, Object node) {
194    checkArgument(graph.nodes().contains(node), NODE_NOT_IN_GRAPH, node);
195    Set<N> visitedNodes = new HashSet<N>();
196    Queue<N> queuedNodes = new ArrayDeque<N>();
197    visitedNodes.add((N) node);
198    queuedNodes.add((N) node);
199    // Perform a breadth-first traversal rooted at the input node.
200    while (!queuedNodes.isEmpty()) {
201      N currentNode = queuedNodes.remove();
202      for (N successor : graph.successors(currentNode)) {
203        if (visitedNodes.add(successor)) {
204          queuedNodes.add(successor);
205        }
206      }
207    }
208    return Collections.unmodifiableSet(visitedNodes);
209  }

211  /**
212   * Returns {@code true} iff {@code graphA} and {@code graphB} have the same elements and the same
213   * relationships between elements, as exposed via the {@link Graph} interface.
214   *
215   * <p>Thus, two graphs A and B are equivalent if both are null or <b>all</b> of the following are
216   * true:
217   *
218   * <ul>
219   * <li>A and B have equal {@link Graph#isDirected() directedness}.
220   * <li>A and B have equal {@link Graph#nodes() node sets}.
221   * <li>A and B have equal {@link Graph#edges() edge sets}.
222   * </ul>
223   *
224   * <p>Graph properties besides {@link Graph#isDirected() directedness} do <b>not</b> affect
225   * equivalence. For example, two graphs may be considered equivalent even if one allows self-loops
226   * and the other doesn't. Additionally, the order in which nodes or edges are added to the graph,
227   * and the order in which they are iterated over, are irrelevant.
228   */
229  public static boolean equivalent(@Nullable Graph<?> graphA, @Nullable Graph<?> graphB) {
230    if (graphA == graphB) {
231      return true;
232    }
233    if (graphA == null || graphB == null) {
234      return false;
235    }
236
237    return graphA.isDirected() == graphB.isDirected()
238        && graphA.nodes().equals(graphB.nodes())
239        && graphA.edges().equals(graphB.edges());
240  }

252   * <li>A and B have equal {@link Graph#nodes() node sets}.
253   * <li>A and B have equal {@link Graph#edges() edge sets}.
254   * <li>Each edge in A has a {@link ValueGraph#edgeValue(Object, Object) value} equal to the {@link
255   *     ValueGraph#edgeValue(Object, Object) value} of the corresponding edge in B.
256   * </ul>
257   *
258   * <p>Graph properties besides {@link Graph#isDirected() directedness} do <b>not</b> affect
259   * equivalence. For example, two graphs may be considered equivalent even if one allows self-loops

261   * and the order in which they are iterated over, are irrelevant.
262   */
263  public static boolean equivalent(
264      @Nullable ValueGraph<?, ?> graphA, @Nullable ValueGraph<?, ?> graphB) {
265    if (graphA == graphB) {
266      return true;
267    }
268    if (graphA == null || graphB == null) {
269      return false;
270    }
271
272    return graphA.isDirected() == graphB.isDirected()
273        && graphA.nodes().equals(graphB.nodes())
274        && graphA.edgeValues().equals(graphB.edgeValues());

277  /**
278   * Returns {@code true} iff {@code networkA} and {@code networkB} have the same elements and the
279   * same relationships between elements, as exposed via the {@link Network} interface.
280   *
281   * <p>Thus, two networks A and B are equivalent if both are null or <b>all</b> of the following
282   * are true:
283   *
284   * <ul>
285   * <li>A and B have equal {@link Network#isDirected() directedness}.
286   * <li>A and B have equal {@link Network#nodes() node sets}.
287   * <li>A and B have equal {@link Network#edges() edge sets}.
288   * <li>Each edge in A connects the same nodes in the same direction (if any) as the corresponding
289   *     edge in B.
290   * </ul>
291   *
292   * <p>Network properties besides {@link Network#isDirected() directedness} do <b>not</b> affect
293   * equivalence. For example, two networks may be considered equal even if one allows parallel
294   * edges and the other doesn't. Additionally, the order in which nodes or edges are added to the
295   * network, and the order in which they are iterated over, are irrelevant.
296   */
297  public static boolean equivalent(
298      @Nullable Network<?, ?> networkA, @Nullable Network<?, ?> networkB) {
299    if (networkA == networkB) {
300      return true;
301    }
302    if (networkA == null || networkB == null) {
303      return false;
304    }
305
306    if (networkA.isDirected() != networkB.isDirected()
307        || !networkA.nodes().equals(networkB.nodes())
308        || !networkA.edges().equals(networkB.edges())) {
309      return false;
310    }
311
312    for (Object edge : networkA.edges()) {
313      if (!networkA.incidentNodes(edge).equals(networkB.incidentNodes(edge))) {
314        return false;
315      }
316    }
317
318    return true;
319  }

575  /**
576   * Returns an induced subgraph of {@code graph}. This subgraph is a new graph that contains all of
577   * the nodes in {@code nodes}, and all of the {@link Graph#edges() edges} from {@code graph} for
578   * which both nodes are contained by {@code nodes}.
579   *
580   * @throws IllegalArgumentException if any element in {@code nodes} is not a node in the graph
581   */
582  public static <N> MutableGraph<N> inducedSubgraph(Graph<N> graph, Iterable<? extends N> nodes) {
583    MutableGraph<N> subgraph = GraphBuilder.from(graph).build();
584    for (N node : nodes) {
585      subgraph.addNode(node);
586    }
587    for (N node : subgraph.nodes()) {
588      for (N successorNode : graph.successors(node)) {
589        if (subgraph.nodes().contains(successorNode)) {
590          subgraph.putEdge(node, successorNode);
591        }
592      }
593    }
594    return subgraph;
595  }

621   * Returns an induced subgraph of {@code network}. This subgraph is a new graph that contains all
622   * of the nodes in {@code nodes}, and all of the {@link Network#edges() edges} from {@code
623   * network} for which the {@link Network#incidentNodes(Object) incident nodes} are both contained
624   * by {@code nodes}.
625   *
626   * @throws IllegalArgumentException if any element in {@code nodes} is not a node in the graph
627   */
628  public static <N, E> MutableNetwork<N, E> inducedSubgraph(
629      Network<N, E> network, Iterable<? extends N> nodes) {
630    MutableNetwork<N, E> subgraph = NetworkBuilder.from(network).build();
631    for (N node : nodes) {
632      subgraph.addNode(node);
633    }
634    for (N node : subgraph.nodes()) {
635      for (E edge : network.outEdges(node)) {
636        N successorNode = network.incidentNodes(edge).adjacentNode(node);
637        if (subgraph.nodes().contains(successorNode)) {
638          subgraph.addEdge(node, successorNode, edge);
639        }
640      }
641    }
642    return subgraph;
643  }

670  /** Creates a mutable copy of {@code network} with the same nodes and edges. */
671  public static <N, E> MutableNetwork<N, E> copyOf(Network<N, E> network) {
672    MutableNetwork<N, E> copy =
673        NetworkBuilder.from(network)
674            .expectedNodeCount(network.nodes().size())
675            .expectedEdgeCount(network.edges().size())
676            .build();
677    for (N node : network.nodes()) {
678      copy.addNode(node);
679    }
680    for (E edge : network.edges()) {
681      EndpointPair<N> endpointPair = network.incidentNodes(edge);
682      copy.addEdge(endpointPair.nodeU(), endpointPair.nodeV(), edge);
683    }
684    return copy;
685  }
