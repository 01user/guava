098  /**
099   * Returns a {@link Stream} containing the elements of the first stream, followed by the elements
100   * of the second stream, and so on.
101   *
102   * <p>This is equivalent to {@code Stream.of(streams).flatMap(stream -> stream)}, but the returned
103   * stream may perform better.
104   *
105   * @see Stream#concat(Stream, Stream)
106   */
107  @SafeVarargs
108  public static <T> Stream<T> concat(Stream<? extends T>... streams) {
109    // TODO(lowasser): consider an implementation that can support SUBSIZED
110    boolean isParallel = false;
111    int characteristics = Spliterator.ORDERED | Spliterator.SIZED | Spliterator.NONNULL;
112    long estimatedSize = 0L;
113    ImmutableList.Builder<Spliterator<? extends T>> splitrsBuilder =
114        new ImmutableList.Builder<>(streams.length);
115    for (Stream<? extends T> stream : streams) {
116      isParallel |= stream.isParallel();
117      Spliterator<? extends T> splitr = stream.spliterator();
118      splitrsBuilder.add(splitr);
119      characteristics &= splitr.characteristics();
120      estimatedSize = LongMath.saturatedAdd(estimatedSize, splitr.estimateSize());
121    }
122    return StreamSupport.stream(
123        CollectSpliterators.flatMap(
124            splitrsBuilder.build().spliterator(),
125            splitr -> (Spliterator<T>) splitr,
126            characteristics,
127            estimatedSize),
128        isParallel);
129  }

203  /**
204   * Returns the last element of the specified stream, or {@link java.util.Optional#empty} if the
205   * stream is empty.
206   *
207   * <p>Equivalent to {@code stream.reduce((a, b) -> b)}, but may perform significantly better. This
208   * method's runtime will be between O(log n) and O(n), performing better on <a
209   * href="http://gee.cs.oswego.edu/dl/html/StreamParallelGuidance.html">efficiently splittable</a>
210   * streams.
211   *
212   * <p>If the stream has nondeterministic order, this has equivalent semantics to {@link
213   * Stream#findAny} (which you might as well use).
214   *
215   * @see Stream#findFirst()
216   * @throws NullPointerException if the last element of the stream is null
217   */
218  public static <T> java.util.Optional<T> findLast(Stream<T> stream) {
219    class OptionalState<T> {
220      boolean set = false;
221      T value = null;
222
223      void set(@Nullable T value) {
224        this.set = true;
225        this.value = value;
226      }
227
228      T get() {
229        checkState(set);
230        return value;
231      }
232    }
233    OptionalState<T> state = new OptionalState<>();
234
235    Deque<Spliterator<T>> splits = new ArrayDeque<>();
236    splits.addLast(stream.spliterator());
237
238    while (!splits.isEmpty()) {
239      Spliterator<T> spliterator = splits.removeLast();
240
241      if (spliterator.getExactSizeIfKnown() == 0) {
242        continue; // drop this split
243      }
244
245      // Many spliterators will have trySplits that are SUBSIZED even if they are not themselves
246      // SUBSIZED.
247      if (spliterator.hasCharacteristics(Spliterator.SUBSIZED)) {
248        // we can drill down to exactly the smallest nonempty spliterator
249        while (true) {
250          Spliterator<T> prefix = spliterator.trySplit();
251          if (prefix == null || prefix.getExactSizeIfKnown() == 0) {
252            break;
253          } else if (spliterator.getExactSizeIfKnown() == 0) {
254            spliterator = prefix;
255            break;
256          }
257        }
258
259        // spliterator is known to be nonempty now
260        spliterator.forEachRemaining(state::set);
261        return java.util.Optional.of(state.get());
262      }
263
264      Spliterator<T> prefix = spliterator.trySplit();
265      if (prefix == null || prefix.getExactSizeIfKnown() == 0) {
266        // we can't split this any further
267        spliterator.forEachRemaining(state::set);
268        if (state.set) {
269          return java.util.Optional.of(state.get());
270        }
271        // fall back to the last split
272        continue;
273      }
274      splits.addLast(prefix);
275      splits.addLast(spliterator);
276    }
277    return java.util.Optional.empty();
278  }

334  /**
335   * Returns a stream in which each element is the result of passing the corresponding element of
336   * each of {@code streamA} and {@code streamB} to {@code function}.
337   *
338   * <p>For example:
339   *
340   * <pre>{@code
341   * Streams.zip(
342   *   Stream.of("foo1", "foo2", "foo3"),
343   *   Stream.of("bar1", "bar2"),
344   *   (arg1, arg2) -> arg1 + ":" + arg2)
345   * }</pre>
346   *
347   * <p>will return {@code Stream.of("foo1:bar1", "foo2:bar2")}.
348   *
349   * <p>The resulting stream will only be as long as the shorter of the two input streams; if one
350   * stream is longer, its extra elements will be ignored.
351   *
352   * <p>The resulting stream is not <a
353   * href="http://gee.cs.oswego.edu/dl/html/StreamParallelGuidance.html">efficiently splittable</a>.
354   * This may harm parallel performance.
355   */
356  public static <A, B, R> Stream<R> zip(
357      Stream<A> streamA, Stream<B> streamB, BiFunction<? super A, ? super B, R> function) {
358    checkNotNull(streamA);
359    checkNotNull(streamB);
360    checkNotNull(function);
361    boolean isParallel = streamA.isParallel() || streamB.isParallel(); // same as Stream.concat
362    Spliterator<A> splitrA = streamA.spliterator();
363    Spliterator<B> splitrB = streamB.spliterator();
364    int characteristics =
365        splitrA.characteristics()
366            & splitrB.characteristics()
367            & (Spliterator.SIZED | Spliterator.ORDERED);
368    Iterator<A> itrA = Spliterators.iterator(splitrA);
369    Iterator<B> itrB = Spliterators.iterator(splitrB);
370    return StreamSupport.stream(
371        new AbstractSpliterator<R>(
372            Math.min(splitrA.estimateSize(), splitrB.estimateSize()), characteristics) {
373          @Override
374          public boolean tryAdvance(Consumer<? super R> action) {
375            if (itrA.hasNext() && itrB.hasNext()) {
376              action.accept(function.apply(itrA.next(), itrB.next()));
377              return true;
378            }
379            return false;
380          }
381        },
382        isParallel);
383  }
384  /**
385   * An analogue of {@link java.util.function.Function} also accepting an index.
386   *
387   * <p>This interface is only intended for use by callers of {@link #mapWithIndex}.
388   */
389  public interface FunctionWithIndex<T, R> {
390    /** Applies this function to the given argument and its index within a stream. */
391    R apply(T from, long index);
392  }
393
394  /**
395   * Returns a stream consisting of the results of applying the given function to the elements of
396   * {@code stream} and their indices in the stream. For example,
397   *
398   * <pre>{@code
399   * mapWithIndex(
400   *     Stream.of("a", "b", "c"),
401   *     (str, index) -> str + ":" + index)
402   * }</pre>
403   *
404   * <p>would return {@code Stream.of("a:0", "b:1", "c:2")}.
405   *
406   * <p>The resulting stream is <a
407   * href="http://gee.cs.oswego.edu/dl/html/StreamParallelGuidance.html">efficiently splittable</a>
408   * if and only if {@code stream} was efficiently splittable and its underlying spliterator
409   * reported {@link Spliterator#SUBSIZED}. This is generally the case if the underlying stream
410   * comes from a data structure supporting efficient indexed random access, typically an array or
411   * list.
412   *
413   * <p>The order of the resulting stream is defined if and only if the order of the original stream
414   * was defined.
415   */
416  public static <T, R> Stream<R> mapWithIndex(
417      Stream<T> stream, FunctionWithIndex<? super T, ? extends R> function) {
418    checkNotNull(stream);
419    checkNotNull(function);
420    boolean isParallel = stream.isParallel();
421    Spliterator<T> fromSpliterator = stream.spliterator();
422
423    if (!fromSpliterator.hasCharacteristics(Spliterator.SUBSIZED)) {
424      Iterator<T> fromIterator = Spliterators.iterator(fromSpliterator);
425      return StreamSupport.stream(
426          new AbstractSpliterator<R>(
427              fromSpliterator.estimateSize(),
428              fromSpliterator.characteristics() & (Spliterator.ORDERED | Spliterator.SIZED)) {
429            long index = 0;
430
431            @Override
432            public boolean tryAdvance(Consumer<? super R> action) {
433              if (fromIterator.hasNext()) {
434                action.accept(function.apply(fromIterator.next(), index++));
435                return true;
436              }
437              return false;
438            }
439          },
440          isParallel);
441    }
442    class Splitr implements Spliterator<R>, Consumer<T> {
443      final Spliterator<T> splitr;
444      long index;
445      T holder = null;
446
447      Splitr(Spliterator<T> splitr, long index) {
448        this.splitr = splitr;
449        this.index = index;
450      }
451
452      @Override
453      public void accept(@Nullable T t) {
454        this.holder = t;
455      }
456
457      @Override
458      public boolean tryAdvance(Consumer<? super R> action) {
459        if (splitr.tryAdvance(this)) {
460          try {
461            action.accept(function.apply(holder, index++));
462            return true;
463          } finally {
464            holder = null;
465          }
466        }
467        return false;
468      }
469
470      @Override
471      public Spliterator<R> trySplit() {
472        Spliterator<T> split = splitr.trySplit();
473        if (split == null) {
474          return null;
475        }
476        Spliterator<R> result = new Splitr(split, index);
477        this.index += split.getExactSizeIfKnown();
478        return result;
479      }
480
481      @Override
482      public long estimateSize() {
483        return splitr.estimateSize();
484      }
485
486      @Override
487      public int characteristics() {
488        return splitr.characteristics()
489            & (Spliterator.ORDERED | Spliterator.SIZED | Spliterator.SUBSIZED);
490      }
491    }
492    return StreamSupport.stream(new Splitr(fromSpliterator, 0), isParallel);
493  }
