017package com.google.common.graph;
018
019import com.google.common.annotations.Beta;
020import com.google.errorprone.annotations.CanIgnoreReturnValue;
021
022/**
023 * A subtype of {@link Network} which permits mutations.
024 * Users should generally use the {@link Network} interface where possible.
025 *
026 * @author Joshua O'Madadhain
027 * @param <N> Node parameter type
028 * @param <E> Edge parameter type
029 * @since 20.0
030 */
031@Beta
032public interface MutableNetwork<N, E> extends Network<N, E> {
033
034  /**
035   * Adds {@code node} to this graph (optional operation).
036   *
037   * <p><b>Nodes must be unique</b>, just as {@code Map} keys must be; they must also be non-null.
038   *
039   * @return {@code true} iff the graph was modified as a result of this call
040   * @throws UnsupportedOperationException if the add operation is not supported by this graph
041   */
042  @CanIgnoreReturnValue
043  boolean addNode(N node);
044
045  /**
046   * Adds {@code edge} to this graph, connecting {@code node1} to {@code node2}
047   * (optional operation).
048   *
049   * <p><b>Edges must be unique</b>, just as {@code Map} keys must be; they must also be non-null.
050   *
051   * <p>If {@code edge} already connects {@code node1} to {@code node2} in this graph
052   * (in the specified order if order is significant, as for directed graphs, else in any order),
053   * then this method will have no effect and will return {@code false}.
054   *
055   * <p>Behavior if {@code node1} and {@code node2} are not already elements of the graph is
056   * unspecified. Suggested behaviors include (a) silently adding {@code node1} and {@code node2}
057   * to the graph or (b) throwing {@code IllegalArgumentException}.
058   *
059   * @return {@code true} iff the graph was modified as a result of this call
060   * @throws IllegalArgumentException if {@code edge} already exists and connects nodes other than
061   *     {@code node1} and {@code node2}, or if the graph is not a multigraph and {@code node1} is
062   *     already connected to {@code node2}
063   * @throws UnsupportedOperationException if the add operation is not supported by this graph
064   */
065  @CanIgnoreReturnValue
066  boolean addEdge(E edge, N node1, N node2);
067
068  /**
069   * Removes {@code node} from this graph, if it is present (optional operation).
070   * In general, all edges incident to {@code node} in this graph will also be removed.
071   * (This is not true for hyperedges.)
072   *
073   * @return {@code true} iff the graph was modified as a result of this call
074   * @throws UnsupportedOperationException if the remove operation is not supported by this graph
075   */
076  @CanIgnoreReturnValue
077  boolean removeNode(Object node);
078
079  /**
080   * Removes {@code edge} from this graph, if it is present (optional operation).
081   * In general, nodes incident to {@code edge} are unaffected (although implementations may choose
082   * to disallow certain configurations, e.g., isolated nodes).
083   *
084   * @return {@code true} iff the graph was modified as a result of this call
085   * @throws UnsupportedOperationException if the remove operation is not supported by this graph
086   */
087  @CanIgnoreReturnValue
088  boolean removeEdge(Object edge);
089}
