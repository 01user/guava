015package com.google.common.collect;
016
017import static com.google.common.base.Preconditions.checkNotNull;
018
019import com.google.common.annotations.Beta;
020import com.google.common.annotations.GwtIncompatible;
021import com.google.common.annotations.VisibleForTesting;
022import com.google.common.base.Equivalence;
023import com.google.common.base.Function;
024import com.google.common.collect.MapMakerInternalMap.InternalEntry;
025import java.util.concurrent.ConcurrentMap;
026
027/**
028 * Contains static methods pertaining to instances of {@link Interner}.
029 *
030 * @author Kevin Bourrillion
031 * @since 3.0
032 */
033@Beta
034@GwtIncompatible
035public final class Interners {
036  private Interners() {}
037
038  /** Builder for {@link Interner} instances. */
039  public static class InternerBuilder {
040    private final MapMaker mapMaker = new MapMaker();
041    private boolean strong = true;
042
043    private InternerBuilder() {
044    }
045
046    /**
047     * Instructs the {@link InternerBuilder} to build a strong interner.
048     *
049     * @see Interners#newStrongInterner()
050     */
051    public InternerBuilder strong() {
052      this.strong = true;
053      return this;
054    }
055
056    /**
057     * Instructs the {@link InternerBuilder} to build a weak interner.
058     *
059     * @see Interners#newWeakInterner()
060     */
061    @GwtIncompatible("java.lang.ref.WeakReference")
062    public InternerBuilder weak() {
063      this.strong = false;
064      return this;
065    }
066
067    /**
068     * Sets the concurrency level that will be used by the to-be-built {@link Interner}.
069     *
070     * @see MapMaker#concurrencyLevel(int)
071     */
072    public InternerBuilder concurrencyLevel(int concurrencyLevel) {
073      this.mapMaker.concurrencyLevel(concurrencyLevel);
074      return this;
075    }
076
077    public <E> Interner<E> build() {
078      return strong ? new StrongInterner<E>(mapMaker) : new WeakInterner<E>(mapMaker);
079    }
080  }
081
082  /** Returns a fresh {@link InternerBuilder} instance. */
083  public static InternerBuilder newBuilder() {
084    return new InternerBuilder();
085  }
086
087  /**
088   * Returns a new thread-safe interner which retains a strong reference to each instance it has
089   * interned, thus preventing these instances from being garbage-collected. If this retention is
090   * acceptable, this implementation may perform better than {@link #newWeakInterner}.
091   */
092  public static <E> Interner<E> newStrongInterner() {
093    return newBuilder().strong().build();
094  }
095
096  /**
097   * Returns a new thread-safe interner which retains a weak reference to each instance it has
098   * interned, and so does not prevent these instances from being garbage-collected. This most
099   * likely does not perform as well as {@link #newStrongInterner}, but is the best alternative
100   * when the memory usage of that implementation is unacceptable.
101   */
102  @GwtIncompatible("java.lang.ref.WeakReference")
103  public static <E> Interner<E> newWeakInterner() {
104    return newBuilder().weak().build();
105  }
106
107  @VisibleForTesting
108  static final class StrongInterner<E> implements Interner<E> {
109    @VisibleForTesting
110    final ConcurrentMap<E, E> map;
111
112    private StrongInterner(MapMaker mapMaker) {
113      this.map = mapMaker.makeMap();
114    }
115
116    @Override
117    public E intern(E sample) {
118      E canonical = map.putIfAbsent(checkNotNull(sample), sample);
119      return (canonical == null) ? sample : canonical;
120    }
121  }
122
123  @VisibleForTesting
124  static final class WeakInterner<E> implements Interner<E> {
125    // MapMaker is our friend, we know about this type
126    @VisibleForTesting
127    final MapMakerInternalMap<E, Dummy, ?, ?> map;
128
129    private WeakInterner(MapMaker mapMaker) {
130      this.map = mapMaker.weakKeys().keyEquivalence(Equivalence.equals()).makeCustomMap();
131    }
132
133    @Override
134    public E intern(E sample) {
135      while (true) {
136        // trying to read the canonical...
137        InternalEntry<E, Dummy, ?> entry = map.getEntry(sample);
138        if (entry != null) {
139          E canonical = entry.getKey();
140          if (canonical != null) { // only matters if weak/soft keys are used
141            return canonical;
142          }
143        }
144
145        // didn't see it, trying to put it instead...
146        Dummy sneaky = map.putIfAbsent(sample, Dummy.VALUE);
147        if (sneaky == null) {
148          return sample;
149        } else {
150          /* Someone beat us to it! Trying again...
151           *
152           * Technically this loop not guaranteed to terminate, so theoretically (extremely
153           * unlikely) this thread might starve, but even then, there is always going to be another
154           * thread doing progress here.
155           */
156        }
157      }
158    }
159
160    private enum Dummy {
161      VALUE
162    }
163  }
164
165  /**
166   * Returns a function that delegates to the {@link Interner#intern} method of the given interner.
167   *
168   * @since 8.0
169   */
170  public static <E> Function<E, E> asFunction(Interner<E> interner) {
171    return new InternerFunction<E>(checkNotNull(interner));
172  }
173
174  private static class InternerFunction<E> implements Function<E, E> {
175
176    private final Interner<E> interner;
177
178    public InternerFunction(Interner<E> interner) {
179      this.interner = interner;
180    }
181
182    @Override
183    public E apply(E input) {
184      return interner.intern(input);
185    }
186
187    @Override
188    public int hashCode() {
189      return interner.hashCode();
190    }
191
192    @Override
193    public boolean equals(Object other) {
194      if (other instanceof InternerFunction) {
195        InternerFunction<?> that = (InternerFunction<?>) other;
196        return interner.equals(that.interner);
197      }
198
199      return false;
200    }
201  }
202}
