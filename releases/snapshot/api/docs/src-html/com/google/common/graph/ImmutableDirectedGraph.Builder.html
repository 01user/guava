019import static com.google.common.base.Preconditions.checkArgument;
020import static com.google.common.base.Preconditions.checkNotNull;
021import static com.google.common.graph.GraphErrorMessageUtils.EDGE_NOT_IN_GRAPH;
022import static com.google.common.graph.GraphErrorMessageUtils.NODE_NOT_IN_GRAPH;
023
024import com.google.common.annotations.Beta;
025import com.google.common.base.Objects;
026import com.google.common.collect.ImmutableMap;
027import com.google.common.collect.ImmutableSet;
028import com.google.common.collect.Sets;
029import com.google.errorprone.annotations.CanIgnoreReturnValue;
030
031import java.util.Set;
032
033import javax.annotation.Nullable;

035/**
036 * Implementation of an immutable directed graph consisting of nodes
037 * of type N and edges of type E.
038 *
039 * <p>This class maintains the following data structures:
040 * <ul>
041 * <li>For each node: sets of incoming and outgoing edges.
042 * <li>For each edge: references to the source and target nodes.
043 * </ul>
044 *
045 * <p>Some invariants/assumptions are maintained in this implementation:
046 * <ul>
047 * <li>An edge has exactly two end-points (source node and target node), which
048 *     may or may not be distinct.
049 * <li>By default, this is not a multigraph, that is, parallel edges (multiple
050 *     edges directed from node1 to node2) are not allowed.  If you want a multigraph,
051 *     build the graph with the 'multigraph' option:
052 *     <pre>ImmutableDirectedGraph.builder(Graphs.config().multigraph()).build();</pre>
053 * <li>Anti-parallel edges (same incident nodes but in opposite direction,
054 *     e.g. (node1, node2) and (node2, node1)) are always allowed.
055 * <li>Edges are not adjacent to themselves by definition. In the case of a
056 *     self-loop, a node can be adjacent to itself, but an edge will never be.
057 * </ul>
058 *
059 * <p>The time complexity of all {@code Set}-returning accessors is O(1), since we
060 * are returning views. An exception to this is {@code edgesConnecting(node1, node2)},
061 * which is O(min(outD_node1, inD_node2)).
062 *
063 * <p>All other accessors have a time complexity of O(1), except for {@code degree(node)},
064 * whose time complexity is O(outD_node).
065 *
066 * @author Joshua O'Madadhain
067 * @param <N> Node parameter type
068 * @param <E> Edge parameter type
069 * @see Graphs
070 * @since 20.0
071 */
072// TODO(user): Add support for sorted nodes/edges and/or hypergraphs.
073@Beta
074public final class ImmutableDirectedGraph<N, E> extends AbstractImmutableGraph<N, E>
075    implements DirectedGraph<N, E> {
076
077  // All nodes in the graph exist in this map
078  private final ImmutableMap<N, NodeConnections<N, E>> nodeConnections;
079  // All edges in the graph exist in this map
080  private final ImmutableMap<E, DirectedIncidentNodes<N>> edgeToIncidentNodes;
081  private final GraphConfig config;
082
083  private ImmutableDirectedGraph(Builder<N, E> builder) {
084    DirectedGraph<N, E> directedGraph = builder.directedGraph;
085    ImmutableMap.Builder<N, NodeConnections<N, E>> nodeConnectionsBuilder =
086        ImmutableMap.builder();
087    for (N node : directedGraph.nodes()) {
088      NodeConnections<N, E> connections = DirectedNodeConnections.ofImmutable(
089          directedGraph.predecessors(node), directedGraph.successors(node),
090          directedGraph.inEdges(node), directedGraph.outEdges(node));
091      nodeConnectionsBuilder.put(node, connections);
092    }
093    this.nodeConnections = nodeConnectionsBuilder.build();
094    ImmutableMap.Builder<E, DirectedIncidentNodes<N>> edgeToIncidentNodesBuilder =
095        ImmutableMap.builder();
096    for (E edge : directedGraph.edges()) {
097      DirectedIncidentNodes<N> incidentNodes = DirectedIncidentNodes.of(
098          directedGraph.source(edge), directedGraph.target(edge));
099      edgeToIncidentNodesBuilder.put(edge, incidentNodes);
100    }
101    this.edgeToIncidentNodes = edgeToIncidentNodesBuilder.build();
102    this.config = directedGraph.config();
103  }
104
105  @Override
106  public Set<N> nodes() {
107    return nodeConnections.keySet();
108  }
109
110  @Override
111  public Set<E> edges() {
112    return edgeToIncidentNodes.keySet();
113  }
114
115  @Override
116  public GraphConfig config() {
117    return config;
118  }
119
120  @Override
121  public Set<E> incidentEdges(Object node) {
122    return Sets.union(inEdges(node), outEdges(node));
123  }
124
125  @Override
126  public Set<N> incidentNodes(Object edge) {
127    return checkedEndpoints(edge).asImmutableSet();
128  }
129
130  @Override
131  public Set<N> adjacentNodes(Object node) {
132    return Sets.union(predecessors(node), successors(node));
133  }
134
135  @Override
136  public Set<E> adjacentEdges(Object edge) {
137    ImmutableSet.Builder<E> adjacentEdges = ImmutableSet.builder();
138    for (N endpoint : incidentNodes(edge)) {
139      for (E adjacentEdge : incidentEdges(endpoint)) {
140        if (!edge.equals(adjacentEdge)) { // Edges are not adjacent to themselves by definition.
141          adjacentEdges.add(adjacentEdge);
142        }
143      }
144    }
145    return adjacentEdges.build();
146  }
147
148  /**
149   * Returns the intersection of these two sets, using {@link Sets#intersection}:
150   * <ol>
151   * <li>Outgoing edges of {@code node1}.
152   * <li>Incoming edges of {@code node2}.
153   * </ol>
154   */
155  @Override
156  public Set<E> edgesConnecting(Object node1, Object node2) {
157    Set<E> sourceOutEdges = outEdges(node1); // Verifies that node1 is in graph
158    if (!config.isSelfLoopsAllowed() && node1.equals(node2)) {
159      return ImmutableSet.of();
160    }
161    Set<E> targetInEdges = inEdges(node2);
162    return (sourceOutEdges.size() <= targetInEdges.size())
163        ? Sets.intersection(sourceOutEdges, targetInEdges).immutableCopy()
164        : Sets.intersection(targetInEdges, sourceOutEdges).immutableCopy();
165  }
166
167  @Override
168  public Set<E> inEdges(Object node) {
169    return checkedConnections(node).inEdges();
170  }
171
172  @Override
173  public Set<E> outEdges(Object node) {
174    return checkedConnections(node).outEdges();
175  }
176
177  @Override
178  public Set<N> predecessors(Object node) {
179    return checkedConnections(node).predecessors();
180  }
181
182  @Override
183  public Set<N> successors(Object node) {
184    return checkedConnections(node).successors();
185  }
186
187  @Override
188  public long degree(Object node) {
189    return incidentEdges(node).size();
190  }
191
192  @Override
193  public long inDegree(Object node) {
194    return inEdges(node).size();
195  }
196
197  @Override
198  public long outDegree(Object node) {
199    return outEdges(node).size();
200  }
201
202  @Override
203  public N source(Object edge) {
204    return checkedEndpoints(edge).source();
205  }
206
207  @Override
208  public N target(Object edge) {
209    return checkedEndpoints(edge).target();
210  }
211
212  @Override
213  public boolean equals(@Nullable Object object) {
214    return (object instanceof DirectedGraph) && Graphs.equal(this, (DirectedGraph<?, ?>) object);
215  }
216
217  @Override
218  public int hashCode() {
219    // The node set is included in the hash to differentiate between graphs with isolated nodes.
220    return Objects.hashCode(nodes(), edgeToIncidentNodes);
221  }
222
223  @Override
224  public String toString() {
225    return String.format("config: %s, nodes: %s, edges: %s",
226        config,
227        nodes(),
228        edgeToIncidentNodes);
229  }
230
231  private NodeConnections<N, E> checkedConnections(Object node) {
232    checkNotNull(node, "node");
233    NodeConnections<N, E> connections = nodeConnections.get(node);
234    checkArgument(connections != null, NODE_NOT_IN_GRAPH, node);
235    return connections;
236  }
237
238  private DirectedIncidentNodes<N> checkedEndpoints(Object edge) {
239    checkNotNull(edge, "edge");
240    DirectedIncidentNodes<N> endpoints = edgeToIncidentNodes.get(edge);
241    checkArgument(endpoints != null, EDGE_NOT_IN_GRAPH, edge);
242    return endpoints;
243  }
244
245  /**
246   * Returns a new builder. The generated builder is equivalent to the builder
247   * created by the {@code Builder} constructor.
248   */
249  public static <N, E> Builder<N, E> builder() {
250    return new Builder<N, E>();
251  }
252
253  /**
254   * Returns a new builder. The generated builder is equivalent to the builder
255   * created by the {@code Builder} constructor.
256   *
257   * @param config an instance of {@code GraphConfig} with the intended
258   *        graph configuration.
259   */
260  public static <N, E> Builder<N, E> builder(GraphConfig config) {
261    return new Builder<N, E>(config);
262  }
263
264  /**
265   * Returns an immutable copy of the input graph.
266   */
267  public static <N, E> ImmutableDirectedGraph<N, E> copyOf(DirectedGraph<N, E> graph) {
268    return new Builder<N, E>(graph).build();
269  }

271  /**
272   * A builder for creating immutable directed graph instances.
273   *
274   * @param <N> Node parameter type
275   * @param <E> edge parameter type
276   * @see GraphConfig
277   */
278  public static final class Builder<N, E> implements AbstractImmutableGraph.Builder<N, E> {
279
280    private final DirectedGraph<N, E> directedGraph;
281
282    /**
283     * Creates a new builder with the default graph configuration.
284     */
285    public Builder() {
286      this(Graphs.<N, E>createDirected());
287    }
288
289    /**
290     * Creates a new builder with the specified graph configuration.
291     */
292    public Builder(GraphConfig config) {
293      this(Graphs.<N, E>createDirected(config));
294    }
295
296    /**
297     * Creates a new builder whose internal state is that of {@code graph}.
298     *
299     * <p>NOTE: This constructor should only be used in the case where it will be immediately
300     * followed by a call to {@code build}, to ensure that the input graph will not be modified.
301     * Currently the only such context is {@code Immutable*Graph.copyOf()}, which use these
302     * constructors to avoid making an extra copy of the graph state.
303     * @see ImmutableDirectedGraph#copyOf(DirectedGraph)
304     */
305    private Builder(DirectedGraph<N, E> graph) {
306      this.directedGraph = checkNotNull(graph, "graph");
307    }
308
309    @Override
310    @CanIgnoreReturnValue
311    public Builder<N, E> addNode(N node) {
312      directedGraph.addNode(node);
313      return this;
314    }
315
316    @Override
317    @CanIgnoreReturnValue
318    public Builder<N, E> addEdge(E edge, N node1, N node2) {
319      directedGraph.addEdge(edge, node1, node2);
320      return this;
321    }
322
323    /**
324     * Adds all elements of {@code graph} to the graph being built.
325     *
326     * @throws IllegalArgumentException under either of two conditions:
327     *     (1) the {@code GraphConfig} objects held by the graph being built and by {@code graph}
328     *     are not compatible
329     *     (2) calling {@code Graph.addEdge(e, n1, n2)} on the graph being built throws IAE
330     * @see Graph#addEdge(e, n1, n2)
331     */
332    @CanIgnoreReturnValue
333    public Builder<N, E> addGraph(DirectedGraph<N, E> graph) {
334      checkArgument(
335          directedGraph.config().compatibleWith(graph.config()),
336          "GraphConfigs for input and for graph being built are not compatible: input: %s, "
337              + "this graph: %s",
338          graph.config(),
339          directedGraph.config());
340
341      for (N node : graph.nodes()) {
342        directedGraph.addNode(node);
343      }
344      for (E edge : graph.edges()) {
345        directedGraph.addEdge(edge, graph.source(edge), graph.target(edge));
346      }
347
348      return this;
349    }
350
351    @Override
352    public ImmutableDirectedGraph<N, E> build() {
353      return new ImmutableDirectedGraph<N, E>(this);
354    }
355  }
