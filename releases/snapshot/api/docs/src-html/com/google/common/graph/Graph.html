017package com.google.common.graph;
018
019import com.google.common.annotations.Beta;
020import java.util.ConcurrentModificationException;
021import java.util.Map;
022import java.util.Set;
023import javax.annotation.Nullable;
024
025/**
026 * An interface for <a href="https://en.wikipedia.org/wiki/Graph_(discrete_mathematics)">graph</a>
027 * data structures. A graph is composed of a set of nodes (sometimes called vertices) and a set of
028 * edges connecting pairs of nodes. Graphs are useful for modeling many kinds of relations. If the
029 * relation to be modeled is symmetric (such as "distance between cities"), that can be represented
030 * with an undirected graph, where an edge that connects node A to node B also connects node B to
031 * node A. If the relation to be modeled is asymmetric (such as "employees managed"), that can be
032 * represented with a directed graph, where edges are strictly one-way.
033 *
034 * <p>There are three main interfaces provided to represent graphs. In order of increasing
035 * complexity they are: {@link BasicGraph}, {@link Graph}, and {@link Network}. You should generally
036 * prefer the simplest interface that satisfies your use case.
037 *
038 * <p>To choose the right interface, answer these questions:
039 *
040 * <ol>
041 * <li>Do you have data (objects) that you wish to associate with edges?
042 *     <p>Yes: Go to question 2. No: Use {@link BasicGraph}.
043 * <li>Are the objects you wish to associate with edges unique within the scope of a graph? That is,
044 *     no two objects would be {@link Object#equals(Object) equal} to each other. A common example
045 *     where this would <i>not</i> be the case is with weighted graphs.
046 *     <p>Yes: Go to question 3. No: Use {@link Graph}.
047 * <li>Do you need to be able to query the graph for an edge associated with a particular object?
048 *     For example, do you need to query what nodes an edge associated with a particular object
049 *     connects, or whether an edge associated with that object exists in the graph?
050 *     <p>Yes: Use {@link Network}. No: Go to question 4.
051 * <li>Do you need explicit support for parallel edges? For example, do you need to remove one edge
052 *     connecting a pair of nodes while leaving other edges connecting those same nodes intact?
053 *     <p>Yes: Use {@link Network}. No: Use {@link Graph}.
054 * </ol>
055 *
056 * <p>Although {@link MutableGraph} and {@link MutableNetwork} both require users to provide objects
057 * to associate with edges when adding them, the differentiating factor is that in {@link Graph}s,
058 * these objects can be any arbitrary data. Like the values in a {@link Map}, they do not have to be
059 * unique, and can be mutated while in the graph. In a {@link Network}, these objects serve as keys
060 * into the data structure. Like the keys in a {@link Map}, they must be unique, and cannot be
061 * mutated in a way that affects their equals/hashcode or the data structure will become corrupted.
062 *
063 * <p>In all three interfaces, nodes have all the same requirements as keys in a {@link Map}.
064 *
065 * <p>All mutation methods live on the subinterface {@link MutableGraph}. If you do not need to
066 * mutate a graph (e.g. if you write a method than runs a read-only algorithm on the graph), you
067 * should prefer the non-mutating {@link Graph} interface.
068 *
069 * <p>We provide an efficient implementation of this interface via {@link GraphBuilder}. When using
070 * the implementation provided, all {@link Set}-returning methods provide live, unmodifiable views
071 * of the graph. In other words, you cannot add an element to the {@link Set}, but if an element is
072 * added to the {@link Graph} that would affect the result of that set, it will be updated
073 * automatically. This also means that you cannot modify a {@link Graph} in a way that would affect
074 * a {#link Set} while iterating over that set. For example, you cannot remove the nodes from a
075 * {@link Graph} while iterating over {@link #nodes} (unless you first make a copy of the nodes),
076 * just as you could not remove the keys from a {@link Map} while iterating over its {@link
077 * Map#keySet()}. This will either throw a {@link ConcurrentModificationException} or risk undefined
078 * behavior.
079 *
080 * <p>Example of use:
081 *
082 * <pre><code>
083 * MutableGraph<String, Double> synonymGraph = GraphBuilder.undirected().build();
084 * synonymGraph.putEdgeValue("large", "big", 0.9);
085 * synonymGraph.putEdgeValue("large", "huge", 0.9);
086 * synonymGraph.putEdgeValue("large", "grand", 0.6);
087 * synonymGraph.putEdgeValue("large", "cold", 0.0);
088 * synonymGraph.putEdgeValue("large", "small", -1.0);
089 * for (String word : synonymGraph.adjacentNodes("large")) {
090 *   if (synonymGraph.edgeValue(word, "large") > 0.5) {
091 *     System.out.println(word + " is a synonym for large");
092 *   }
093 * }
094 * </code></pre>
095 *
096 * @author James Sexton
097 * @param <N> Node parameter type
098 * @param <V> Value parameter type
099 * @since 20.0
100 */
101@Beta
102public interface Graph<N, V> {
103  //
104  // Graph-level accessors
105  //
106
107  /** Returns all nodes in this graph, in the order specified by {@link #nodeOrder()}. */
108  Set<N> nodes();
109
110  /** Returns all edges in this graph. */
111  Set<Endpoints<N>> edges();
112
113  //
114  // Graph properties
115  //
116
117  /**
118   * Returns true if the edges in this graph are directed. Directed edges connect a {@link
119   * Endpoints#source() source node} to a {@link Endpoints#target() target node}, while undirected
120   * edges connect a pair of nodes to each other.
121   */
122  boolean isDirected();
123
124  /**
125   * Returns true if this graph allows self-loops (edges that connect a node to itself). Attempting
126   * to add a self-loop to a graph that does not allow them will throw an {@link
127   * UnsupportedOperationException}.
128   */
129  boolean allowsSelfLoops();
130
131  /** Returns the order of iteration for the elements of {@link #nodes()}. */
132  ElementOrder<N> nodeOrder();
133
134  //
135  // Element-level accessors
136  //
137
138  /**
139   * Returns the nodes which have an incident edge in common with {@code node} in this graph.
140   *
141   * @throws IllegalArgumentException if {@code node} is not an element of this graph
142   */
143  Set<N> adjacentNodes(Object node);
144
145  /**
146   * Returns all nodes in this graph adjacent to {@code node} which can be reached by traversing
147   * {@code node}'s incoming edges <i>against</i> the direction (if any) of the edge.
148   *
149   * <p>In an undirected graph, this is equivalent to {@link #adjacentNodes(Object)}.
150   *
151   * @throws IllegalArgumentException if {@code node} is not an element of this graph
152   */
153  Set<N> predecessors(Object node);
154
155  /**
156   * Returns all nodes in this graph adjacent to {@code node} which can be reached by traversing
157   * {@code node}'s outgoing edges in the direction (if any) of the edge.
158   *
159   * <p>In an undirected graph, this is equivalent to {@link #adjacentNodes(Object)}.
160   *
161   * <p>This is <i>not</i> the same as "all nodes reachable from {@code node} by following outgoing
162   * edges". For that functionality, see {@link Graphs#reachableNodes(Graph, Object)}.
163   *
164   * @throws IllegalArgumentException if {@code node} is not an element of this graph
165   */
166  Set<N> successors(Object node);
167
168  /**
169   * Returns the count of {@code node}'s incident edges, counting self-loops twice (equivalently,
170   * the number of times an edge touches {@code node}).
171   *
172   * <p>For directed graphs, this is equal to {@code inDegree(node) + outDegree(node)}.
173   *
174   * <p>For undirected graphs, this is equal to {@code adjacentNodes(node).size()} + (1 if {@code
175   * node} has an incident self-loop, 0 otherwise).
176   *
177   * <p>If the count is greater than {@code Integer.MAX_VALUE}, returns {@code Integer.MAX_VALUE}.
178   *
179   * @throws IllegalArgumentException if {@code node} is not an element of this graph
180   */
181  int degree(Object node);
182
183  /**
184   * Returns the count of {@code node}'s incoming edges (equal to {@code predecessors(node).size()})
185   * in a directed graph. In an undirected graph, returns the {@link #degree(Object)}.
186   *
187   * <p>If the count is greater than {@code Integer.MAX_VALUE}, returns {@code Integer.MAX_VALUE}.
188   *
189   * @throws IllegalArgumentException if {@code node} is not an element of this graph
190   */
191  int inDegree(Object node);
192
193  /**
194   * Returns the count of {@code node}'s outgoing edges (equal to {@code successors(node).size()})
195   * in a directed graph. In an undirected graph, returns the {@link #degree(Object)}.
196   *
197   * <p>If the count is greater than {@code Integer.MAX_VALUE}, returns {@code Integer.MAX_VALUE}.
198   *
199   * @throws IllegalArgumentException if {@code node} is not an element of this graph
200   */
201  int outDegree(Object node);
202
203  /**
204   * If there is an edge connecting {@code nodeA} to {@code nodeB}, returns the non-null value
205   * associated with that edge.
206   *
207   * <p>In an undirected graph, this is equal to {@code edgeValue(nodeB, nodeA)}.
208   *
209   * @throws IllegalArgumentException if there is no edge connecting {@code nodeA} to {@code nodeB}
210   */
211  V edgeValue(Object nodeA, Object nodeB);
212
213  /**
214   * If there is an edge connecting {@code nodeA} to {@code nodeB}, returns the non-null value
215   * associated with that edge; otherwise, returns {@code defaultValue}.
216   *
217   * <p>In an undirected graph, this is equal to {@code edgeValueOrDefault(nodeB, nodeA,
218   * defaultValue)}.
219   *
220   * @throws IllegalArgumentException if {@code nodeA} or {@code nodeB} is not an element of this
221   *     graph
222   */
223  V edgeValueOrDefault(Object nodeA, Object nodeB, @Nullable V defaultValue);
224
225  //
226  // Graph identity
227  //
228
229  /**
230   * Returns {@code true} iff {@code object} is a {@link Graph} that has the same elements and the
231   * same structural relationships as those in this graph.
232   *
233   * <p>Thus, two graphs A and B are equal if <b>all</b> of the following are true:
234   *
235   * <ul>
236   * <li>A and B have equal {@link #isDirected() directedness}.
237   * <li>A and B have equal {@link #nodes() node sets}.
238   * <li>A and B have equal {@link #edges() edge sets}.
239   * <li>Every edge in A and B are associated with equal {@link #edgeValue(Object, Object) values}.
240   * </ul>
241   *
242   * <p>Graph properties besides {@link #isDirected() directedness} do <b>not</b> affect equality.
243   * For example, two graphs may be considered equal even if one allows self-loops and the other
244   * doesn't. Additionally, the order in which nodes or edges are added to the graph, and the order
245   * in which they are iterated over, are irrelevant.
246   *
247   * <p>A reference implementation of this is provided by {@link AbstractGraph#equals(Object)}.
248   */
249  @Override
250  boolean equals(@Nullable Object object);
251
252  /**
253   * Returns the hash code for this graph. The hash code of a graph is defined as the hash code of a
254   * map from each of its {@link #edges() edges} to the associated {@link #edgeValue(Object, Object)
255   * edge value}.
256   *
257   * <p>A reference implementation of this is provided by {@link AbstractGraph#hashCode()}.
258   */
259  @Override
260  int hashCode();
261}
