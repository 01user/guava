017package com.google.common.base;
018
019import com.google.common.annotations.GwtCompatible;
020import com.google.errorprone.annotations.CanIgnoreReturnValue;
021
022import javax.annotation.Nullable;
023
024/**
025 * Determines an output value based on an input value; a pre-Java-8 version of {@code
026 * java.util.function.Function}.
027 *
028 * <p>The {@link Functions} class provides common functions and related utilites.
029 *
030 * <p>See the Guava User Guide article on <a href=
031 * "https://github.com/google/guava/wiki/FunctionalExplained">the use of {@code
032 * Function}</a>.
033 *
034 * <h3>For Java 8+ users</h3>
035 *
036 * <p>This interface is now a legacy type. Use {@code java.util.function.Function} (or the
037 * appropriate primitive specialization such as {@code ToIntFunction}) instead whenever possible.
038 * Otherwise, reducing <i>explicit</i> dependencies on this type by using lambda expressions or
039 * method references instead of classes will leave your code easier to migrate in the future.
040 *
041 * <p>To use an existing function (say, named {@code function}) in a context where the <i>other
042 * type</i> of function is expected, use {@code function::apply}. A future version of
043 * {@code com.google.common.base.Function} will be made to <i>extend</i>
044 * {@code java.util.function.Function}, making conversion code necessary only in one direction.
045 * At that time, we plan to also deprecate this interface.
046 *
047 * @author Kevin Bourrillion
048 * @since 2.0
049 */
050@GwtCompatible
051public interface Function<F, T> {
052  /**
053   * Returns the result of applying this function to {@code input}. This method is <i>generally
054   * expected</i>, but not absolutely required, to have the following properties:
055   *
056   * <ul>
057   * <li>Its execution does not cause any observable side effects.
058   * <li>The computation is <i>consistent with equals</i>; that is, {@link Objects#equal
059   *     Objects.equal}{@code (a, b)} implies that {@code Objects.equal(function.apply(a),
060   *     function.apply(b))}.
061   * </ul>
062   *
063   * @throws NullPointerException if {@code input} is null and this function does not accept null
064   *     arguments
065   */
066  @Nullable
067  @CanIgnoreReturnValue
068  T apply(@Nullable F input);
069
070  /**
071   * Indicates whether another object is equal to this function.
072   *
073   * <p>Most implementations will have no reason to override the behavior of {@link Object#equals}.
074   * However, an implementation may also choose to return {@code true} whenever {@code object} is a
075   * {@link Function} that it considers <i>interchangeable</i> with this one. "Interchangeable"
076   * <i>typically</i> means that {@code Objects.equal(this.apply(f), that.apply(f))} is true for all
077   * {@code f} of type {@code F}. Note that a {@code false} result from this method does not imply
078   * that the functions are known <i>not</i> to be interchangeable.
079   */
080  @Override
081  boolean equals(@Nullable Object object);
082}
