017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkState;
020import static com.google.common.graph.GraphConstants.GRAPH_STRING_FORMAT;
021
022import com.google.common.annotations.Beta;
023import com.google.common.base.Function;
024import com.google.common.collect.Maps;
025import com.google.common.math.IntMath;
026import com.google.common.primitives.Ints;
027import java.util.AbstractSet;
028import java.util.Iterator;
029import java.util.Map;
030import java.util.Set;
031import javax.annotation.Nullable;
032
033/**
034 * This class provides a skeletal implementation of {@link Graph}. It is recommended to extend
035 * this class rather than implement {@link Graph} directly, to ensure consistent {@link
036 * #equals(Object)} and {@link #hashCode()} results across different graph implementations.
037 *
038 * @author James Sexton
039 * @param <N> Node parameter type
040 * @param <V> Value parameter type
041 * @since 20.0
042 */
043@Beta
044public abstract class AbstractGraph<N, V> implements Graph<N, V> {
045
046  /**
047   * Returns the number of edges in this graph; used to calculate the size of {@link #edges()}.
048   * The default implementation is O(|N|). You can manually keep track of the number of edges and
049   * override this method for better performance.
050   */
051  protected long edgeCount() {
052    long degreeSum = 0L;
053    for (N node : nodes()) {
054      degreeSum += degree(node);
055    }
056    // According to the degree sum formula, this is equal to twice the number of edges.
057    checkState((degreeSum & 1) == 0);
058    return degreeSum >>> 1;
059  }
060
061  /**
062   * A reasonable default implementation of {@link Graph#edges()} defined in terms of
063   * {@link #nodes()} and {@link #successors(Object)}.
064   */
065  @Override
066  public Set<Endpoints<N>> edges() {
067    return new AbstractSet<Endpoints<N>>() {
068      @Override
069      public Iterator<Endpoints<N>> iterator() {
070        return EndpointsIterator.of(AbstractGraph.this);
071      }
072
073      @Override
074      public int size() {
075        return Ints.saturatedCast(edgeCount());
076      }
077
078      @Override
079      public boolean contains(Object obj) {
080        if (!(obj instanceof Endpoints)) {
081          return false;
082        }
083        Endpoints<?> endpoints = (Endpoints<?>) obj;
084        return isDirected() == endpoints.isDirected()
085            && nodes().contains(endpoints.nodeA())
086            && successors(endpoints.nodeA()).contains(endpoints.nodeB());
087      }
088    };
089  }
090
091  @Override
092  public int degree(Object node) {
093    if (isDirected()) {
094      return IntMath.saturatedAdd(predecessors(node).size(), successors(node).size());
095    } else {
096      Set<N> neighbors = adjacentNodes(node);
097      int selfLoop = (allowsSelfLoops() && neighbors.contains(node)) ? 1 : 0;
098      return IntMath.saturatedAdd(neighbors.size(), selfLoop);
099    }
100  }
101
102  @Override
103  public int inDegree(Object node) {
104    return isDirected() ? predecessors(node).size() : degree(node);
105  }
106
107  @Override
108  public int outDegree(Object node) {
109    return isDirected() ? successors(node).size() : degree(node);
110  }
111
112  @Override
113  public final boolean equals(@Nullable Object obj) {
114    if (obj == this) {
115      return true;
116    }
117    if (!(obj instanceof Graph)) {
118      return false;
119    }
120    Graph<?, ?> other = (Graph<?, ?>) obj;
121
122    if (isDirected() != other.isDirected()
123        || !nodes().equals(other.nodes())
124        || !edges().equals(other.edges())) {
125      return false;
126    }
127
128    for (Endpoints<N> edge : edges()) {
129      if (!edgeValue(edge.nodeA(), edge.nodeB()).equals(
130          other.edgeValue(edge.nodeA(), edge.nodeB()))) {
131        return false;
132      }
133    }
134
135    return true;
136  }
137
138  @Override
139  public final int hashCode() {
140    return edgeValueMap().hashCode();
141  }
142
143  /**
144   * Returns a string representation of this graph.
145   */
146  @Override
147  public String toString() {
148    String propertiesString = String.format(
149        "isDirected: %s, allowsSelfLoops: %s", isDirected(), allowsSelfLoops());
150    return String.format(GRAPH_STRING_FORMAT,
151        propertiesString,
152        nodes(),
153        edgeValueMap());
154  }
155
156  private Map<Endpoints<N>, V> edgeValueMap() {
157    Function<Endpoints<N>, V> edgeToValueFn = new Function<Endpoints<N>, V>() {
158      @Override
159      public V apply(Endpoints<N> edge) {
160        return edgeValue(edge.nodeA(), edge.nodeB());
161      }
162    };
163    return Maps.asMap(edges(), edgeToValueFn);
164  }
165}
