017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkNotNull;
020
021import com.google.common.collect.ImmutableMap;
022import com.google.common.collect.ImmutableSet;
023
024import java.util.Map;
025import java.util.Set;
026
027/**
028 * A {@link Network} whose relationships are constant. Instances of this class may be obtained
029 * with {@link #copyOf(Network)}.
030 *
031 * <p>The time complexity of {@code edgesConnecting(node1, node2)} is O(min(outD_node1, inD_node2)).
032 *
033 * @author James Sexton
034 * @author Joshua O'Madadhain
035 * @author Omar Darwish
036 * @param <N> Node parameter type
037 * @param <E> Edge parameter type
038 */
039public final class ImmutableNetwork<N, E> extends AbstractConfigurableNetwork<N, E> {
040
041  private ImmutableNetwork(Network<N, E> graph) {
042    super(NetworkBuilder.from(graph), getNodeConnections(graph), getEdgeToIncidentNodes(graph));
043  }
044
045  /**
046   * Returns an immutable copy of {@code graph}.
047   */
048  public static <N, E> ImmutableNetwork<N, E> copyOf(Network<N, E> graph) {
049    return (graph instanceof ImmutableNetwork)
050        ? (ImmutableNetwork<N, E>) graph
051        : new ImmutableNetwork<N, E>(graph);
052  }
053
054  /**
055   * Simply returns its argument.
056   *
057   * @deprecated no need to use this
058   */
059  @Deprecated
060  public static <N, E> ImmutableNetwork<N, E> copyOf(ImmutableNetwork<N, E> graph) {
061    return checkNotNull(graph);
062  }
063
064  @Override
065  public Set<E> edgesConnecting(Object node1, Object node2) {
066    // This set is calculated as the intersection of two sets, and is likely to be small.
067    // As an optimization, copy it to an ImmutableSet so re-iterating is fast.
068    return ImmutableSet.copyOf(super.edgesConnecting(node1, node2));
069  }
070
071  private static <N, E> Map<N, NodeConnections<N, E>> getNodeConnections(Network<N, E> graph) {
072    ImmutableMap.Builder<N, NodeConnections<N, E>> nodeConnections = ImmutableMap.builder();
073    for (N node : graph.nodes()) {
074      nodeConnections.put(node, nodeConnectionsOf(graph, node));
075    }
076    return nodeConnections.build();
077  }
078
079  private static <N, E> Map<E, IncidentNodes<N>> getEdgeToIncidentNodes(Network<N, E> graph) {
080    ImmutableMap.Builder<E, IncidentNodes<N>> edgeToIncidentNodes = ImmutableMap.builder();
081    for (E edge : graph.edges()) {
082      edgeToIncidentNodes.put(edge, IncidentNodes.of(graph.incidentNodes(edge)));
083    }
084    return edgeToIncidentNodes.build();
085  }
086
087  private static <N, E> NodeConnections<N, E> nodeConnectionsOf(Network<N, E> graph, N node) {
088    return graph.isDirected()
089        ? DirectedNodeConnections.ofImmutable(
090            graph.predecessors(node), graph.successors(node),
091            graph.inEdges(node), graph.outEdges(node))
092        : UndirectedNodeConnections.ofImmutable(
093            graph.adjacentNodes(node), graph.incidentEdges(node));
094  }
095}
