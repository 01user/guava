017package com.google.common.graph;
018
019import static com.google.common.base.Preconditions.checkArgument;
020
021import com.google.common.base.Optional;
022
023import java.util.Comparator;
024
025/**
026 * A builder for constructing instances of {@link Graph} with user-defined properties.
027 *
028 * <p>A graph built by this class will have the following properties by default:
029 * <ul>
030 * <li>does not allow parallel edges
031 * <li>allows self-loops
032 * </ul>
033 *
034 * @author James Sexton
035 * @author Joshua O'Madadhain
036 * @since 20.0
037 */
038// TODO(b/24620028): Add support for sorted nodes/edges. Use the same pattern as CacheBuilder
039// to narrow the generic <N, E> type when Comparators are provided.
040public final class GraphBuilder<N, E> {
041  Boolean directed = null; // No default value to enforce that this is set before building
042  boolean allowsParallelEdges = false;
043  boolean allowsSelfLoops = true;
044  Comparator<N> nodeComparator = null;
045  Comparator<E> edgeComparator = null;
046  Optional<Integer> expectedNodeCount = Optional.absent();
047  Optional<Integer> expectedEdgeCount = Optional.absent();
048
049  private GraphBuilder() {}
050
051  /**
052   * Returns a {@link GraphBuilder} for building directed graphs.
053   */
054  public static GraphBuilder<Object, Object> directed() {
055    return new GraphBuilder<Object, Object>().directed(true);
056  }
057
058  /**
059   * Returns a {@link GraphBuilder} for building undirected graphs.
060   */
061  public static GraphBuilder<Object, Object> undirected() {
062    return new GraphBuilder<Object, Object>().directed(false);
063  }
064
065  /**
066   * Returns a {@link GraphBuilder} initialized with all properties queryable from {@code graph}.
067   *
068   * <p>The "queryable" properties are those that are exposed through the {@link Graph} interface,
069   * such as {@link Graph#isDirected()}. Other properties, such as {@link #expectedNodeCount(int)},
070   * are not set in the new builder.
071   */
072  public static <N, E> GraphBuilder<N, E> from(Graph<N, E> graph) {
073    return new GraphBuilder<N, E>()
074        .directed(graph.isDirected())
075        .allowsParallelEdges(graph.allowsParallelEdges())
076        .allowsSelfLoops(graph.allowsSelfLoops());
077  }
078
079  /**
080   * This value should be set by {@link #directed()}, {@link #undirected()},
081   * or {@link #from(Graph)}.
082   */
083  private GraphBuilder<N, E> directed(boolean directed) {
084    this.directed = directed;
085    return this;
086  }
087
088  /**
089   * Specifies whether the graph will allow parallel edges. Attempting to add a parallel edge to
090   * a graph that does not allow them will throw an {@link UnsupportedOperationException}.
091   */
092  public GraphBuilder<N, E> allowsParallelEdges(boolean allowsParallelEdges) {
093    this.allowsParallelEdges = allowsParallelEdges;
094    return this;
095  }
096
097  /**
098   * Specifies whether the graph will allow self-loops (edges that connect a node to itself).
099   * Attempting to add a self-loop to a graph that does not allow them will throw an
100   * {@link UnsupportedOperationException}.
101   */
102  public GraphBuilder<N, E> allowsSelfLoops(boolean allowsSelfLoops) {
103    this.allowsSelfLoops = allowsSelfLoops;
104    return this;
105  }
106
107  /**
108   * Specifies the expected number of nodes in the graph.
109   *
110   * @throws IllegalArgumentException if {@code expectedNodeCount} is negative
111   */
112  public GraphBuilder<N, E> expectedNodeCount(int expectedNodeCount) {
113    checkArgument(expectedNodeCount >= 0, "The expected number of nodes can't be negative: %s",
114        expectedNodeCount);
115    this.expectedNodeCount = Optional.of(expectedNodeCount);
116    return this;
117  }
118
119  /**
120   * Specifies the expected number of edges in the graph.
121   *
122   * @throws IllegalArgumentException if {@code expectedEdgeCount} is negative
123   */
124  public GraphBuilder<N, E> expectedEdgeCount(int expectedEdgeCount) {
125    checkArgument(expectedEdgeCount >= 0, "The expected number of edges can't be negative: %s",
126        expectedEdgeCount);
127    this.expectedEdgeCount = Optional.of(expectedEdgeCount);
128    return this;
129  }
130
131  /**
132   * Returns an empty mutable {@link Graph} with the properties of this {@link GraphBuilder}.
133   */
134  public <N1 extends N, E1 extends E> Graph<N1, E1> build() {
135    return new ConfigurableGraph<N1, E1>(this);
136  }
137}
