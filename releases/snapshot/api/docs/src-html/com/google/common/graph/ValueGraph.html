017package com.google.common.graph;
018
019import com.google.common.annotations.Beta;
020import javax.annotation.Nullable;
021
022/**
023 * An interface for <a
024 * href="https://en.wikipedia.org/wiki/Graph_(discrete_mathematics)">graph</a>-structured data,
025 * whose edges have associated non-unique values.
026 *
027 * <p>A graph is composed of a set of nodes and a set of edges connecting pairs of nodes.
028 *
029 * <p>There are three main interfaces provided to represent graphs. In order of increasing
030 * complexity they are: {@link Graph}, {@link ValueGraph}, and {@link Network}. You should generally
031 * prefer the simplest interface that satisfies your use case. See the <a
032 * href="https://github.com/google/guava/wiki/GraphsExplained#choosing-the-right-graph-type">
033 * "Choosing the right graph type"</a> section of the Guava User Guide for more details.
034 *
035 * <h3>Capabilities</h3>
036 *
037 * <p>{@code ValueGraph} supports the following use cases (<a
038 * href="https://github.com/google/guava/wiki/GraphsExplained#definitions">definitions of
039 * terms</a>):
040 *
041 * <ul>
042 *   <li>directed graphs
043 *   <li>undirected graphs
044 *   <li>graphs that do/don't allow self-loops
045 *   <li>graphs whose nodes/edges are insertion-ordered, sorted, or unordered
046 *   <li>graphs whose edges have associated values
047 * </ul>
048 *
049 * <p>{@code ValueGraph}, as a subtype of {@code Graph}, explicitly does not support parallel edges,
050 * and forbids implementations or extensions with parallel edges. If you need parallel edges, use
051 * {@link Network}. (You can use a positive {@code Integer} edge value as a loose representation of
052 * edge multiplicity, but the {@code *degree()} and mutation methods will not reflect your
053 * interpretation of the edge value as its multiplicity.)
054 *
055 * <h3>Building a {@code ValueGraph}</h3>
056 *
057 * <p>The implementation classes that `common.graph` provides are not public, by design. To create
058 * an instance of one of the built-in implementations of {@code ValueGraph}, use the {@link
059 * ValueGraphBuilder} class:
060 *
061 * <pre>{@code
062 *   MutableValueGraph<Integer, Double> graph = ValueGraphBuilder.directed().build();
063 * }</pre>
064 *
065 * <p>{@link ValueGraphBuilder#build()} returns an instance of {@link MutableValueGraph}, which is a
066 * subtype of {@code ValueGraph} that provides methods for adding and removing nodes and edges. If
067 * you do not need to mutate a graph (e.g. if you write a method than runs a read-only algorithm on
068 * the graph), you should use the non-mutating {@link ValueGraph} interface, or an {@link
069 * ImmutableValueGraph}.
070 *
071 * <p>You can create an immutable copy of an existing {@code ValueGraph} using {@link
072 * ImmutableValueGraph#copyOf(ValueGraph)}:
073 *
074 * <pre>{@code
075 *   ImmutableValueGraph<Integer, Double> immutableGraph = ImmutableValueGraph.copyOf(graph);
076 * }</pre>
077 *
078 * <p>Instances of {@link ImmutableValueGraph} do not implement {@link MutableValueGraph}
079 * (obviously!) and are contractually guaranteed to be unmodifiable and thread-safe.
080 *
081 * <p>The Guava User Guide has <a
082 * href="https://github.com/google/guava/wiki/GraphsExplained#building-graph-instances">more
083 * information on (and examples of) building graphs</a>.
084 *
085 * <h3>Additional documentation</h3>
086 *
087 * <p>See the Guava User Guide for the {@code common.graph} package (<a
088 * href="https://github.com/google/guava/wiki/GraphsExplained">"Graphs Explained"</a>) for
089 * additional documentation, including:
090 *
091 * <ul>
092 *   <li><a
093 *       href="https://github.com/google/guava/wiki/GraphsExplained#equals-hashcode-and-graph-equivalence">
094 *       {@code equals()}, {@code hashCode()}, and graph equivalence</a>
095 *   <li><a href="https://github.com/google/guava/wiki/GraphsExplained#synchronization">
096 *       Synchronization policy</a>
097 *   <li><a href="https://github.com/google/guava/wiki/GraphsExplained#notes-for-implementors">Notes
098 *       for implementors</a>
099 * </ul>
100 *
101 * @author James Sexton
102 * @author Joshua O'Madadhain
103 * @param <N> Node parameter type
104 * @param <V> Value parameter type
105 * @since 20.0
106 */
107@Beta
108public interface ValueGraph<N, V> extends Graph<N> {
109
110  /**
111   * If there is an edge connecting {@code nodeU} to {@code nodeV}, returns the non-null value
112   * associated with that edge.
113   *
114   * <p>In an undirected graph, this is equal to {@code edgeValue(nodeV, nodeU)}.
115   *
116   * @throws IllegalArgumentException if there is no edge connecting {@code nodeU} to {@code nodeV}.
117   */
118  V edgeValue(Object nodeU, Object nodeV);
119
120  /**
121   * If there is an edge connecting {@code nodeU} to {@code nodeV}, returns the non-null value
122   * associated with that edge; otherwise, returns {@code defaultValue}.
123   *
124   * <p>In an undirected graph, this is equal to {@code edgeValueOrDefault(nodeV, nodeU,
125   * defaultValue)}.
126   */
127  V edgeValueOrDefault(Object nodeU, Object nodeV, @Nullable V defaultValue);
128
129  //
130  // ValueGraph identity
131  //
132
133  /**
134   * For the default {@link ValueGraph} implementations, returns true if {@code this == object}
135   * (reference equality). External implementations are free to define this method as they see fit,
136   * as long as they satisfy the {@link Object#equals(Object)} contract.
137   *
138   * <p>To compare two {@link ValueGraph}s based on their contents rather than their references, see
139   * {@link Graphs#equivalent(ValueGraph, ValueGraph)}.
140   */
141  @Override
142  boolean equals(@Nullable Object object);
143
144  /**
145   * For the default {@link ValueGraph} implementations, returns {@code
146   * System.identityHashCode(this)}. External implementations are free to define this method as they
147   * see fit, as long as they satisfy the {@link Object#hashCode()} contract.
148   */
149  @Override
150  int hashCode();
151}
