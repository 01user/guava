037/**
038 * A high-performance, immutable {@code Set} with reliable, user-specified
039 * iteration order. Does not permit null elements.
040 *
041 * <p>Unlike {@link Collections#unmodifiableSet}, which is a <i>view</i> of a
042 * separate collection that can still change, an instance of this class contains
043 * its own private data and will <i>never</i> change. This class is convenient
044 * for {@code public static final} sets ("constant sets") and also lets you
045 * easily make a "defensive copy" of a set provided to your class by a caller.
046 *
047 * <p><b>Warning:</b> Like most sets, an {@code ImmutableSet} will not function
048 * correctly if an element is modified after being placed in the set. For this
049 * reason, and to avoid general confusion, it is strongly recommended to place
050 * only immutable objects into this collection.
051 *
052 * <p>This class has been observed to perform significantly better than {@link
053 * HashSet} for objects with very fast {@link Object#hashCode} implementations
054 * (as a well-behaved immutable object should). While this class's factory
055 * methods create hash-based instances, the {@link ImmutableSortedSet} subclass
056 * performs binary searches instead.
057 *
058 * <p><b>Note:</b> Although this class is not final, it cannot be subclassed
059 * outside its package as it has no public or protected constructors. Thus,
060 * instances of this type are guaranteed to be immutable.
061 *
062 * <p>See the Guava User Guide article on <a href=
063 * "http://code.google.com/p/guava-libraries/wiki/ImmutableCollectionsExplained">
064 * immutable collections</a>.
065 *
066 * @see ImmutableList
067 * @see ImmutableMap
068 * @author Kevin Bourrillion
069 * @author Nick Kralevich
070 * @since 2.0 (imported from Google Collections Library)
071 */
072@GwtCompatible(serializable = true, emulated = true)
073@SuppressWarnings("serial") // we're overriding default serialization
074public abstract class ImmutableSet<E> extends ImmutableCollection<E>
075    implements Set<E> {

141  /**
142   * Returns an immutable set containing the given elements, in order. Repeated
143   * occurrences of an element (according to {@link Object#equals}) after the
144   * first are ignored.
145   *
146   * @throws NullPointerException if any element is null
147   * @since 3.0 (source-compatible since 2.0)
148   */
149  public static <E> ImmutableSet<E> of(E e1, E e2, E e3, E e4, E e5, E e6,
150      E... others) {
151    final int paramCount = 6;
152    Object[] elements = new Object[paramCount + others.length];
153    elements[0] = e1;
154    elements[1] = e2;
155    elements[2] = e3;
156    elements[3] = e4;
157    elements[4] = e5;
158    elements[5] = e6;
159    System.arraycopy(others, 0, elements, paramCount, others.length);
160    return construct(elements.length, elements);
161  }

163  /**
164   * Constructs an {@code ImmutableSet} from the first {@code n} elements of the specified array.
165   * If {@code k} is the size of the returned {@code ImmutableSet}, then the unique elements of
166   * {@code elements} will be in the first {@code k} positions, and {@code elements[i] == null} for
167   * {@code k <= i < n}.
168   *
169   * <p>This may modify {@code elements}.  Additionally, if {@code n == elements.length} and
170   * {@code elements} contains no duplicates, {@code elements} may be used without copying in the
171   * returned {@code ImmutableSet}, in which case it may no longer be modified.
172   *
173   * <p>{@code elements} may contain only values of type {@code E}.
174   *
175   * @throws NullPointerException if any of the first {@code n} elements of {@code elements} is
176   *          null
177   */
178  private static <E> ImmutableSet<E> construct(int n, Object... elements) {
179    switch (n) {
180      case 0:
181        return of();
182      case 1: {
183        @SuppressWarnings("unchecked") // safe; elements contains only E's
184        E elem = (E) elements[0];
185        return of(elem);
186      }
187    }
188    int tableSize = chooseTableSize(n);
189    Object[] table = new Object[tableSize];
190    int mask = tableSize - 1;
191    int hashCode = 0;
192    int uniques = 0;
193    for (int i = 0; i < n; i++) {
194      Object element = ObjectArrays.checkElementNotNull(elements[i], i);
195      int hash = element.hashCode();
196      for (int j = Hashing.smear(hash); ; j++) {
197        int index = j & mask;
198        Object value = table[index];
199        if (value == null) {
200          // Came to an empty slot. Put the element here.
201          elements[uniques++] = element;
202          table[index] = element;
203          hashCode += hash;
204          break;
205        } else if (value.equals(element)) {
206          break;
207        }
208      }
209    }
210    Arrays.fill(elements, uniques, n, null);
211    if (uniques == 1) {
212      // There is only one element or elements are all duplicates
213      @SuppressWarnings("unchecked") // we are careful to only pass in E
214      E element = (E) elements[0];
215      return new SingletonImmutableSet<E>(element, hashCode);
216    } else if (tableSize != chooseTableSize(uniques)) {
217      // Resize the table when the array includes too many duplicates.
218      // when this happens, we have already made a copy
219      return construct(uniques, elements);
220    } else {
221      Object[] uniqueElements = (uniques < elements.length)
222          ? ObjectArrays.arraysCopyOf(elements, uniques)
223          : elements;
224      return new RegularImmutableSet<E>(uniqueElements, hashCode, table, mask);
225    }
226  }

228  // We use power-of-2 tables, and this is the highest int that's a power of 2
229  static final int MAX_TABLE_SIZE = Ints.MAX_POWER_OF_TWO;
230
231  // Represents how tightly we can pack things, as a maximum.
232  private static final double DESIRED_LOAD_FACTOR = 0.7;
233
234  // If the set has this many elements, it will "max out" the table size
235  private static final int CUTOFF =
236      (int) Math.floor(MAX_TABLE_SIZE * DESIRED_LOAD_FACTOR);
237
238  /**
239   * Returns an array size suitable for the backing array of a hash table that
240   * uses open addressing with linear probing in its implementation.  The
241   * returned size is the smallest power of two that can hold setSize elements
242   * with the desired load factor.
243   *
244   * <p>Do not call this method with setSize < 2.
245   */
246  @VisibleForTesting static int chooseTableSize(int setSize) {
247    // Correct the size for open addressing to match desired load factor.
248    if (setSize < CUTOFF) {
249      // Round up to the next highest power of 2.
250      int tableSize = Integer.highestOneBit(setSize - 1) << 1;
251      while (tableSize * DESIRED_LOAD_FACTOR < setSize) {
252        tableSize <<= 1;
253      }
254      return tableSize;
255    }
256
257    // The table can't be completely full or we'll get infinite reprobes
258    checkArgument(setSize < MAX_TABLE_SIZE, "collection too large");
259    return MAX_TABLE_SIZE;
260  }

262  /**
263   * Returns an immutable set containing the given elements, in order. Repeated
264   * occurrences of an element (according to {@link Object#equals}) after the
265   * first are ignored.
266   *
267   * @throws NullPointerException if any of {@code elements} is null
268   * @since 3.0
269   */
270  public static <E> ImmutableSet<E> copyOf(E[] elements) {
271    // TODO(benyu): could we delegate to
272    // copyFromCollection(Arrays.asList(elements))?
273    switch (elements.length) {
274      case 0:
275        return of();
276      case 1:
277        return of(elements[0]);
278      default:
279        return construct(elements.length, elements.clone());
280    }
281  }

283  /**
284   * Returns an immutable set containing the given elements, in order. Repeated
285   * occurrences of an element (according to {@link Object#equals}) after the
286   * first are ignored. This method iterates over {@code elements} at most once.
287   *
288   * <p>Note that if {@code s} is a {@code Set<String>}, then {@code
289   * ImmutableSet.copyOf(s)} returns an {@code ImmutableSet<String>} containing
290   * each of the strings in {@code s}, while {@code ImmutableSet.of(s)} returns
291   * a {@code ImmutableSet<Set<String>>} containing one element (the given set
292   * itself).
293   *
294   * <p>Despite the method name, this method attempts to avoid actually copying
295   * the data when it is safe to do so. The exact circumstances under which a
296   * copy will or will not be performed are undocumented and subject to change.
297   *
298   * @throws NullPointerException if any of {@code elements} is null
299   */
300  public static <E> ImmutableSet<E> copyOf(Iterable<? extends E> elements) {
301    return (elements instanceof Collection)
302        ? copyOf(Collections2.cast(elements))
303        : copyOf(elements.iterator());
304  }

306  /**
307   * Returns an immutable set containing the given elements, in order. Repeated
308   * occurrences of an element (according to {@link Object#equals}) after the
309   * first are ignored.
310   *
311   * @throws NullPointerException if any of {@code elements} is null
312   */
313  public static <E> ImmutableSet<E> copyOf(Iterator<? extends E> elements) {
314    // We special-case for 0 or 1 elements, but anything further is madness.
315    if (!elements.hasNext()) {
316      return of();
317    }
318    E first = elements.next();
319    if (!elements.hasNext()) {
320      return of(first);
321    } else {
322      return new ImmutableSet.Builder<E>()
323          .add(first)
324          .addAll(elements)
325          .build();
326    }
327  }

329  /**
330   * Returns an immutable set containing the given elements, in order. Repeated
331   * occurrences of an element (according to {@link Object#equals}) after the
332   * first are ignored. This method iterates over {@code elements} at most
333   * once.
334   *
335   * <p>Note that if {@code s} is a {@code Set<String>}, then {@code
336   * ImmutableSet.copyOf(s)} returns an {@code ImmutableSet<String>} containing
337   * each of the strings in {@code s}, while {@code ImmutableSet.of(s)} returns
338   * a {@code ImmutableSet<Set<String>>} containing one element (the given set
339   * itself).
340   *
341   * <p><b>Note:</b> Despite what the method name suggests, {@code copyOf} will
342   * return constant-space views, rather than linear-space copies, of some
343   * inputs known to be immutable. For some other immutable inputs, such as key
344   * sets of an {@code ImmutableMap}, it still performs a copy in order to avoid
345   * holding references to the values of the map. The heuristics used in this
346   * decision are undocumented and subject to change except that:
347   * <ul>
348   * <li>A full copy will be done of any {@code ImmutableSortedSet}.</li>
349   * <li>{@code ImmutableSet.copyOf()} is idempotent with respect to pointer
350   * equality.</li>
351   * </ul>
352   *
353   * <p>This method is safe to use even when {@code elements} is a synchronized
354   * or concurrent collection that is currently being modified by another
355   * thread.
356   *
357   * @throws NullPointerException if any of {@code elements} is null
358   * @since 7.0 (source-compatible since 2.0)
359   */
360  public static <E> ImmutableSet<E> copyOf(Collection<? extends E> elements) {
361    if (elements instanceof ImmutableSet
362        && !(elements instanceof ImmutableSortedSet)) {
363      @SuppressWarnings("unchecked") // all supported methods are covariant
364      ImmutableSet<E> set = (ImmutableSet<E>) elements;
365      if (!set.isPartialView()) {
366        return set;
367      }
368    } else if (elements instanceof EnumSet) {
369      EnumSet<?> enumSet = EnumSet.copyOf((EnumSet<?>) elements);
370      @SuppressWarnings("unchecked")
371      // immutable collections are safe for covariant casts
372      ImmutableSet<E> result = (ImmutableSet<E>) ImmutableEnumSet.asImmutable(enumSet);
373      return result;
374    }
375    return copyFromCollection(elements);
376  }
377
378  private static <E> ImmutableSet<E> copyFromCollection(
379      Collection<? extends E> collection) {
380    Object[] elements = collection.toArray();
381    switch (elements.length) {
382      case 0:
383        return of();
384      case 1:
385        @SuppressWarnings("unchecked") // collection had only Es in it
386        E onlyElement = (E) elements[0];
387        return of(onlyElement);
388      default:
389        // safe to use the array without copying it
390        // as specified by Collection.toArray().
391        return construct(elements.length, elements);
392    }
393  }

398  boolean isHashCodeFast() {
399    return false;

402  @Override public boolean equals(@Nullable Object object) {
403    if (object == this) {
404      return true;
405    }
406    if (object instanceof ImmutableSet
407        && isHashCodeFast()
408        && ((ImmutableSet<?>) object).isHashCodeFast()
409        && hashCode() != object.hashCode()) {
410      return false;
411    }
412    return Sets.equalsImpl(this, object);
413  }

416    return Sets.hashCodeImpl(this);

423  abstract static class ArrayImmutableSet<E> extends ImmutableSet<E> {
424    // the elements (two or more) in the desired order.
425    final transient Object[] elements;
426
427    ArrayImmutableSet(Object[] elements) {
428      this.elements = elements;
429    }
430
431    @Override
432    public int size() {
433      return elements.length;
434    }
435
436    @Override public boolean isEmpty() {
437      return false;
438    }
439
440    @Override public UnmodifiableIterator<E> iterator() {
441      return asList().iterator();
442    }
443
444    @Override public Object[] toArray() {
445      return asList().toArray();
446    }
447
448    @Override public <T> T[] toArray(T[] array) {
449      return asList().toArray(array);
450    }
451
452    @Override public boolean containsAll(Collection<?> targets) {
453      if (targets == this) {
454        return true;
455      }
456      if (!(targets instanceof ArrayImmutableSet)) {
457        return super.containsAll(targets);
458      }
459      if (targets.size() > size()) {
460        return false;
461      }
462      for (Object target : ((ArrayImmutableSet<?>) targets).elements) {
463        if (!contains(target)) {
464          return false;
465        }
466      }
467      return true;
468    }
469
470    @Override boolean isPartialView() {
471      return false;
472    }
473
474    @Override ImmutableList<E> createAsList() {
475      return new RegularImmutableAsList<E>(this, elements);
476    }
477  }

479  /*
480   * This class is used to serialize all ImmutableSet instances, except for
481   * ImmutableEnumSet/ImmutableSortedSet, regardless of implementation type. It
482   * captures their "logical contents" and they are reconstructed using public
483   * static factories. This is necessary to ensure that the existence of a
484   * particular implementation type is an implementation detail.
485   */
486  private static class SerializedForm implements Serializable {
487    final Object[] elements;
488    SerializedForm(Object[] elements) {
489      this.elements = elements;
490    }
491    Object readResolve() {
492      return copyOf(elements);
493    }
494    private static final long serialVersionUID = 0;
495  }
496
497  @Override Object writeReplace() {
498    return new SerializedForm(toArray());
499  }

509  /**
510   * A builder for creating immutable set instances, especially {@code public
511   * static final} sets ("constant sets"). Example: <pre>   {@code
512   *
513   *   public static final ImmutableSet<Color> GOOGLE_COLORS =
514   *       new ImmutableSet.Builder<Color>()
515   *           .addAll(WEBSAFE_COLORS)
516   *           .add(new Color(0, 191, 255))
517   *           .build();}</pre>
518   *
519   * Builder instances can be reused; it is safe to call {@link #build} multiple
520   * times to build multiple sets in series. Each set is a superset of the set
521   * created before it.
522   *
523   * @since 2.0 (imported from Google Collections Library)
524   */
525  public static class Builder<E> extends ImmutableCollection.Builder<E> {
526    Object[] contents;
527    int size;
528
529    /**
530     * Creates a new builder. The returned builder is equivalent to the builder
531     * generated by {@link ImmutableSet#builder}.
532     */
533    public Builder() {
534      this(DEFAULT_INITIAL_CAPACITY);
535    }
536
537    Builder(int capacity) {
538      checkArgument(capacity >= 0, "capacity must be >= 0 but was %s", capacity);
539      this.contents = new Object[capacity];
540      this.size = 0;
541    }
542
543    /**
544     * Expand the absolute capacity of the builder so it can accept at least
545     * the specified number of elements without being resized.
546     */
547    Builder<E> ensureCapacity(int minCapacity) {
548      if (contents.length < minCapacity) {
549        contents = ObjectArrays.arraysCopyOf(
550            contents, expandedCapacity(contents.length, minCapacity));
551      }
552      return this;
553    }
554
555    /**
556     * Adds {@code element} to the {@code ImmutableSet}.  If the {@code
557     * ImmutableSet} already contains {@code element}, then {@code add} has no
558     * effect (only the previously added element is retained).
559     *
560     * @param element the element to add
561     * @return this {@code Builder} object
562     * @throws NullPointerException if {@code element} is null
563     */
564    @Override public Builder<E> add(E element) {
565      ensureCapacity(size + 1);
566      contents[size++] = checkNotNull(element);
567      return this;
568    }
569
570    /**
571     * Adds each element of {@code elements} to the {@code ImmutableSet},
572     * ignoring duplicate elements (only the first duplicate element is added).
573     *
574     * @param elements the elements to add
575     * @return this {@code Builder} object
576     * @throws NullPointerException if {@code elements} is null or contains a
577     *     null element
578     */
579    @Override public Builder<E> add(E... elements) {
580      for (int i = 0; i < elements.length; i++) {
581        ObjectArrays.checkElementNotNull(elements[i], i);
582      }
583      ensureCapacity(size + elements.length);
584      System.arraycopy(elements, 0, contents, size, elements.length);
585      size += elements.length;
586      return this;
587    }
588
589    /**
590     * Adds each element of {@code elements} to the {@code ImmutableSet},
591     * ignoring duplicate elements (only the first duplicate element is added).
592     *
593     * @param elements the {@code Iterable} to add to the {@code ImmutableSet}
594     * @return this {@code Builder} object
595     * @throws NullPointerException if {@code elements} is null or contains a
596     *     null element
597     */
598    @Override public Builder<E> addAll(Iterable<? extends E> elements) {
599      if (elements instanceof Collection) {
600        Collection<?> collection = (Collection<?>) elements;
601        ensureCapacity(size + collection.size());
602      }
603      super.addAll(elements);
604      return this;
605    }
606
607    /**
608     * Adds each element of {@code elements} to the {@code ImmutableSet},
609     * ignoring duplicate elements (only the first duplicate element is added).
610     *
611     * @param elements the elements to add to the {@code ImmutableSet}
612     * @return this {@code Builder} object
613     * @throws NullPointerException if {@code elements} is null or contains a
614     *     null element
615     */
616    @Override public Builder<E> addAll(Iterator<? extends E> elements) {
617      super.addAll(elements);
618      return this;
619    }
620
621    /**
622     * Returns a newly-created {@code ImmutableSet} based on the contents of
623     * the {@code Builder}.
624     */
625    @Override public ImmutableSet<E> build() {
626      ImmutableSet<E> result = construct(size, contents);
627      // construct has the side effect of deduping contents, so we update size
628      // accordingly.
629      size = result.size();
630      return result;
631    }
632  }
